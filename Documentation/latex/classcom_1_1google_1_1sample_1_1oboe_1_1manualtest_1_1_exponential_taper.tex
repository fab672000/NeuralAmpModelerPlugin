\doxysubsection{Detailed Description}
Maps integer range info to a double value along an exponential scale.


\begin{DoxyPre}\end{DoxyPre}



\begin{DoxyPre}  x = ival / mResolution
  f(x) = a*(root**bx)
  f(0.0) = dmin
  f(1.0) = dmax\end{DoxyPre}



\begin{DoxyPre}  f(0.0) = a * 1.0 => a = dmin
  f(1.0) = dmin * root**b = dmax
  b = log(dmax / dmin) / \mbox{\hyperlink{structlog}{log(root)}}\end{DoxyPre}
