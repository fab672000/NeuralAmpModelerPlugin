38 template <typename Visitor, typename Derived, int UnrollCount, bool Vectorize, bool ShortCircuitEvaluation>
39 struct visitor_impl<Visitor, Derived, UnrollCount, Vectorize, false, ShortCircuitEvaluation> {
40  // don't use short circuit evaulation for unrolled version
41  using Scalar = typename Derived::Scalar;
42  using Packet = typename packet_traits<Scalar>::type;
43  static constexpr bool RowMajor = Derived::IsRowMajor;
44  static constexpr int RowsAtCompileTime = Derived::RowsAtCompileTime;
45  static constexpr int ColsAtCompileTime = Derived::ColsAtCompileTime;
46  static constexpr int PacketSize = packet_traits<Scalar>::size;
47 
48  static constexpr bool CanVectorize(int K) {
49  constexpr int InnerSizeAtCompileTime = RowMajor ? ColsAtCompileTime : RowsAtCompileTime;
50  return Vectorize && (InnerSizeAtCompileTime - (K % InnerSizeAtCompileTime) >= PacketSize);
51  }
52 
53  template <int K = 0,
54  bool Empty = (K == UnrollCount),
55  std::enable_if_t<Empty, bool> = true>
56  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived&, Visitor&) {}
57 
58  template <int K = 0,
59  bool Empty = (K == UnrollCount),
60  bool Initialize = (K == 0),
61  bool DoVectorOp = CanVectorize(K),
62  std::enable_if_t<!Empty && Initialize && !DoVectorOp, bool> = true>
63  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor)
64  {
65  visitor.init(mat.coeff(0, 0), 0, 0);
66  run<1>(mat, visitor);
67  }
68 
69  template <int K = 0,
70  bool Empty = (K == UnrollCount),
71  bool Initialize = (K == 0),
72  bool DoVectorOp = CanVectorize(K),
73  std::enable_if_t<!Empty && !Initialize && !DoVectorOp, bool> = true>
74  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor)
75  {
76  static constexpr int R = RowMajor ? (K / ColsAtCompileTime) : (K % RowsAtCompileTime);
77  static constexpr int C = RowMajor ? (K % ColsAtCompileTime) : (K / RowsAtCompileTime);
78  visitor(mat.coeff(R, C), R, C);
79  run<K + 1>(mat, visitor);
80  }

83  bool Empty = (K == UnrollCount),
84  bool Initialize = (K == 0),
85  bool DoVectorOp = CanVectorize(K),
86  std::enable_if_t<!Empty && Initialize && DoVectorOp, bool> = true>
87  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor)
88  {
89  Packet P = mat.template packet<Packet>(0, 0);
90  visitor.initpacket(P, 0, 0);
91  run<PacketSize>(mat, visitor);
92  }

95  bool Empty = (K == UnrollCount),

98  std::enable_if_t<!Empty && !Initialize && DoVectorOp, bool> = true>
99  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor)
100  {
101  static constexpr int R = RowMajor ? (K / ColsAtCompileTime) : (K % RowsAtCompileTime);
102  static constexpr int C = RowMajor ? (K % ColsAtCompileTime) : (K / RowsAtCompileTime);
103  Packet P = mat.template packet<Packet>(R, C);
104  visitor.packet(P, R, C);
105  run<K + PacketSize>(mat, visitor);
106  }

175 template <typename Visitor, typename Derived, bool ShortCircuitEvaluation>
176 struct visitor_impl<Visitor, Derived, Dynamic, /*Vectorize=*/false, /*LinearAccess=*/false, ShortCircuitEvaluation> {
177  using short_circuit = short_circuit_eval_impl<Visitor, ShortCircuitEvaluation>;
178  static constexpr bool RowMajor = Derived::IsRowMajor;
179 
180  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor) {
181  const Index innerSize = RowMajor ? mat.cols() : mat.rows();
182  const Index outerSize = RowMajor ? mat.rows() : mat.cols();
183  if (innerSize == 0 || outerSize == 0) return;
184  {
185  visitor.init(mat.coeff(0, 0), 0, 0);
186  if (short_circuit::run(visitor)) return;
187  for (Index i = 1; i < innerSize; ++i) {
188  Index r = RowMajor ? 0 : i;
189  Index c = RowMajor ? i : 0;
190  visitor(mat.coeff(r, c), r, c);
191  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
192  }
193  }
194  for (Index j = 1; j < outerSize; j++) {
195  for (Index i = 0; i < innerSize; ++i) {
196  Index r = RowMajor ? j : i;
197  Index c = RowMajor ? i : j;
198  visitor(mat.coeff(r, c), r, c);
199  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
200  }
201  }
202  }
203 };

205 // dynamic vectorized outer-inner traversal
206 template <typename Visitor, typename Derived, bool ShortCircuitEvaluation>
207 struct visitor_impl<Visitor, Derived, Dynamic, /*Vectorize=*/true, /*LinearAccess=*/false, ShortCircuitEvaluation> {
208  using Scalar = typename Derived::Scalar;
209  using Packet = typename packet_traits<Scalar>::type;
210  static constexpr int PacketSize = packet_traits<Scalar>::size;
211  using short_circuit = short_circuit_eval_impl<Visitor, ShortCircuitEvaluation>;
212  static constexpr bool RowMajor = Derived::IsRowMajor;
213 
214  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor) {
215  const Index innerSize = RowMajor ? mat.cols() : mat.rows();
216  const Index outerSize = RowMajor ? mat.rows() : mat.cols();
217  if (innerSize == 0 || outerSize == 0) return;
218  {
219  Index i = 0;
220  if (innerSize < PacketSize) {
221  visitor.init(mat.coeff(0, 0), 0, 0);
222  i = 1;
223  } else {
224  Packet p = mat.template packet<Packet>(0, 0);
225  visitor.initpacket(p, 0, 0);
226  i = PacketSize;
227  }
228  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
229  for (; i + PacketSize - 1 < innerSize; i += PacketSize) {
230  Index r = RowMajor ? 0 : i;
231  Index c = RowMajor ? i : 0;
232  Packet p = mat.template packet<Packet>(r, c);
233  visitor.packet(p, r, c);
234  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
235  }
236  for (; i < innerSize; ++i) {
237  Index r = RowMajor ? 0 : i;
238  Index c = RowMajor ? i : 0;
239  visitor(mat.coeff(r, c), r, c);
240  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
241  }
242  }
243  for (Index j = 1; j < outerSize; j++) {
244  Index i = 0;
245  for (; i + PacketSize - 1 < innerSize; i += PacketSize) {
246  Index r = RowMajor ? j : i;
247  Index c = RowMajor ? i : j;
248  Packet p = mat.template packet<Packet>(r, c);
249  visitor.packet(p, r, c);
250  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
251  }
252  for (; i < innerSize; ++i) {
253  Index r = RowMajor ? j : i;
254  Index c = RowMajor ? i : j;
255  visitor(mat.coeff(r, c), r, c);
256  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
257  }
258  }
259  }
260 };

280 template <typename Visitor, typename Derived, bool ShortCircuitEvaluation>
281 struct visitor_impl<Visitor, Derived, Dynamic, /*Vectorize=*/true, /*LinearAccess=*/true, ShortCircuitEvaluation> {
282  using Scalar = typename Derived::Scalar;
283  using Packet = typename packet_traits<Scalar>::type;
284  static constexpr int PacketSize = packet_traits<Scalar>::size;
285  using short_circuit = short_circuit_eval_impl<Visitor, ShortCircuitEvaluation>;
286 
287  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const Derived& mat, Visitor& visitor) {
288  const Index size = mat.size();
289  if (size == 0) return;
290  Index k = 0;
291  if (size < PacketSize) {
292  visitor.init(mat.coeff(0), 0);
293  k = 1;
294  } else {
295  Packet p = mat.template packet<Packet>(k);
296  visitor.initpacket(p, k);
297  k = PacketSize;
298  }
299  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
300  for (; k + PacketSize - 1 < size; k += PacketSize) {
301  Packet p = mat.template packet<Packet>(k);
302  visitor.packet(p, k);
303  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
304  }
305  for (; k < size; k++) {
306  visitor(mat.coeff(k), k);
307  if EIGEN_PREDICT_FALSE(short_circuit::run(visitor)) return;
308  }
309  }

354 template <typename Derived, typename Visitor, bool ShortCircuitEvaulation>
355 struct visit_impl {
356  using Evaluator = visitor_evaluator<Derived>;
357  using Scalar = typename DenseBase<Derived>::Scalar;
358 
359  static constexpr bool IsRowMajor = DenseBase<Derived>::IsRowMajor;
360  static constexpr int SizeAtCompileTime = DenseBase<Derived>::SizeAtCompileTime;
361  static constexpr int RowsAtCompileTime = DenseBase<Derived>::RowsAtCompileTime;
362  static constexpr int ColsAtCompileTime = DenseBase<Derived>::ColsAtCompileTime;
363  static constexpr int InnerSizeAtCompileTime = IsRowMajor ? ColsAtCompileTime : RowsAtCompileTime;
364  static constexpr int OuterSizeAtCompileTime = IsRowMajor ? RowsAtCompileTime : ColsAtCompileTime;
365 
366  static constexpr bool LinearAccess = Evaluator::LinearAccess && static_cast<bool>(functor_traits<Visitor>::LinearAccess);
367  static constexpr bool Vectorize = Evaluator::PacketAccess && static_cast<bool>(functor_traits<Visitor>::PacketAccess);
368 
369  static constexpr int PacketSize = packet_traits<Scalar>::size;
370  static constexpr int VectorOps = Vectorize ? (LinearAccess ? (SizeAtCompileTime / PacketSize) : (OuterSizeAtCompileTime * (InnerSizeAtCompileTime / PacketSize))) : 0;
371  static constexpr int ScalarOps = SizeAtCompileTime - (VectorOps * PacketSize);
372  // treat vector op and scalar op as same cost for unroll logic
373  static constexpr int TotalOps = VectorOps + ScalarOps;
374 
375  static constexpr int UnrollCost = int(Evaluator::CoeffReadCost) + int(functor_traits<Visitor>::Cost);
376  static constexpr bool Unroll = (SizeAtCompileTime != Dynamic) && ((TotalOps * UnrollCost) <= EIGEN_UNROLLING_LIMIT);
377  static constexpr int UnrollCount = Unroll ? int(SizeAtCompileTime) : Dynamic;
378 
379 
380  using impl = visitor_impl<Visitor, Evaluator, UnrollCount, Vectorize, LinearAccess, ShortCircuitEvaulation>;
381 
382  static EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE void run(const DenseBase<Derived>& mat, Visitor& visitor) {
383  Evaluator evaluator(mat.derived());
384  impl::run(evaluator, visitor);
385  }
386 };

456 // Default implementation used by non-floating types, where we do not
457 // need special logic for NaN handling.
458 template <typename Derived, bool is_min, int NaNPropagation,
459  bool isInt = NumTraits<typename Derived::Scalar>::IsInteger>
460 struct minmax_coeff_visitor : coeff_visitor<Derived> {
461  using Scalar = typename Derived::Scalar;
462  using Packet = typename packet_traits<Scalar>::type;
463  using Comparator = minmax_compare<Scalar, NaNPropagation, is_min>;
464  static constexpr Index PacketSize = packet_traits<Scalar>::size;
465 
466  EIGEN_DEVICE_FUNC inline void operator()(const Scalar& value, Index i, Index j) {
467  if (Comparator::compare(value, this->res)) {
468  this->res = value;
469  this->row = i;
470  this->col = j;
471  }
472  }
473  EIGEN_DEVICE_FUNC inline void packet(const Packet& p, Index i, Index j) {
474  Scalar value = Comparator::predux(p);
475  if (Comparator::compare(value, this->res)) {
476  const Packet range = preverse(plset<Packet>(Scalar(1)));
477  Packet mask = pcmp_eq(pset1<Packet>(value), p);
478  Index max_idx = PacketSize - static_cast<Index>(predux_max(pand(range, mask)));
479  this->res = value;
480  this->row = Derived::IsRowMajor ? i : i + max_idx;
481  this->col = Derived::IsRowMajor ? j + max_idx : j;
482  }
483  }
484  EIGEN_DEVICE_FUNC inline void initpacket(const Packet& p, Index i, Index j) {
485  Scalar value = Comparator::predux(p);
486  const Packet range = preverse(plset<Packet>(Scalar(1)));
487  Packet mask = pcmp_eq(pset1<Packet>(value), p);
488  Index max_idx = PacketSize - static_cast<Index>(predux_max(pand(range, mask)));
489  this->res = value;
490  this->row = Derived::IsRowMajor ? i : i + max_idx;
491  this->col = Derived::IsRowMajor ? j + max_idx : j;
492  }

504  if ((!(numext::isnan)(value) && (numext::isnan)(this->res)) || Comparator::compare(value, this->res)) {
505  this->res = value;
506  this->row = i;
507  this->col = j;
508  }
509  }
510  EIGEN_DEVICE_FUNC inline void packet(const Packet& p, Index i, Index j) {

542 // Propagate NaNs. If the matrix contains NaN, the location of the first NaN
543 // will be returned in row and col.
544 template <typename Derived, bool is_min, int NaNPropagation>
545  struct minmax_coeff_visitor<Derived, is_min, NaNPropagation, false> : coeff_visitor<Derived> {
546  typedef typename Derived::Scalar Scalar;
547  using Packet = typename packet_traits<Scalar>::type;
548  using Comparator = minmax_compare<Scalar, PropagateNaN, is_min>;
549 
550  EIGEN_DEVICE_FUNC inline void operator()(const Scalar& value, Index i, Index j) {
551  const bool value_is_nan = (numext::isnan)(value);
552  if ((value_is_nan && !(numext::isnan)(this->res)) || Comparator::compare(value, this->res)) {
553  this->res = value;

567  this->res = value;
