60  void readFile(Soundfile* soundfile, const std::string& path_name, int part, int& offset, int max_chan)
61  {
62  std::unique_ptr<AudioFormatReader> formatReader (fFormatManager.createReaderFor (File (path_name)));
63 
64  soundfile->fLength[part] = int(formatReader->lengthInSamples);
65  soundfile->fSR[part] = int(formatReader->sampleRate);
66  soundfile->fOffset[part] = offset;
67 
68  FAUSTFLOAT** buffers = static_cast<FAUSTFLOAT**>(alloca(soundfile->fChannels * sizeof(FAUSTFLOAT*)));
69 
70  getBuffersOffset(soundfile, buffers, offset);
71 
72  if (formatReader->read(reinterpret_cast<int *const *>(buffers), int(formatReader->numChannels), 0, int(formatReader->lengthInSamples), false)) {
73 
74  // Possibly concert samples
75  if (!formatReader->usesFloatingPointData) {
76  for (int chan = 0; chan < int(formatReader->numChannels); ++chan) {
77  FAUSTFLOAT* buffer = &soundfile->fBuffers[chan][soundfile->fOffset[part]];
78  FloatVectorOperations::convertFixedToFloat(buffer, reinterpret_cast<const int*>(buffer), 1.0f/0x7fffffff, int(formatReader->lengthInSamples));
79  }
80  }
81 
82  } else {
83  std::cerr << "Error reading the file : " << path_name << std::endl;
84  }
85 
86  // Update offset
87  offset += soundfile->fLength[part];
