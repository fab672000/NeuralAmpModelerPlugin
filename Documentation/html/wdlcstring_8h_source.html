240  _WDL_CSTRING_PREFIX char *WDL_remove_trailing_decimal_zeros(char *str, unsigned int keep)
241  // returns pointer to decimal point or end of string. removes final zeros after final decimal point only, keep=0 makes min length X, keep=1 X., keep=2 X.0, keep=3 X.00 etc
242  // treats commas as decimal points
243  {
244  char *end = str, *decimal, *last_z=NULL;
245  while (*end) end++;
246  decimal = end;
247  while (--decimal >= str && *decimal >= '0' && *decimal <= '9') if (!last_z && *decimal != '0') last_z = decimal+1;
248  if (decimal < str || (*decimal != '.' && *decimal != ',')) return end;
249  if (!last_z || last_z < decimal+keep) last_z = decimal+keep;
250  if (last_z < end)
251  {
252  *last_z=0;
253  if (!str[0] || ((str[0] == '.' || str[0] == ',') && !str[1]))
254  {
255  str[0]='0';
256  str[1]=0;
257  return str+1;
258  }
259  }
260  return decimal;
261  }

270  {
271  // converts = and leading [ to _. removes leading/trailing whitespace. if extra_filters&1, converts all [] to _.
272  char *w=buf;
273  lstrcpyn_safe(buf,WDL_sanitize_ini_key_start(p),bufsz);
274  while (*w)
275  {
276  if (*w == '=' || ((extra_filters&1) && (*w=='[' || *w == ']'))) *w = '_';
277  w++;
278  }
279  while (w > buf && (w[-1] == ' ' || w[-1] == '\t' || w[-1] == '\r' || w[-1] == '\n')) *--w = 0;
280  while (--w >= buf) if (*w == '\r' || *w == '\n' || *w == '\t') *w = '_';
281  return buf;

305  static WDL_STATICFUNC_UNUSED int logical_char_order(int ch, int case_sensitive)
306  {
307  // _-<>etc, numbers, utf-8 chars, alpha chars
308  if (ch<0) return ch + 384; // utf-8 maps to 256..383
309  if (ch >= '0' && ch <= '9') return ch + 128; // numbers map to 128+'0' etc
310  if (ch >= 'A' && ch <= 'Z') return ch + 384; // alpha goes to 384+'A' or 384+'a' if not ignoring case
311  if (ch >= 'a' && ch <= 'z') return case_sensitive ? (ch + 384) : (ch + 'A' - 'a' + 384);
312  return ch;
313  }

315  // flags: WDL_STRCMP_LOGICAL_EX_FLAG_OLDSORT
316  _WDL_CSTRING_PREFIX int WDL_strcmp_logical_ex(const char *s1, const char *s2, int case_sensitive, int flags)
317  {
318  for (;;)
319  {
320  if (*s1 >= '0' && *s1 <= '9' && *s2 >= '0' && *s2 <= '9')
321  {
322  int lzdiff=0, len1=0, len2=0;
323 
324  while (*s1 == '0') { s1++; lzdiff--; }
325  while (*s2 == '0') { s2++; lzdiff++; }
326 
327  while (s1[len1] >= '0' && s1[len1] <= '9') len1++;
328  while (s2[len2] >= '0' && s2[len2] <= '9') len2++;
329 
330  if (len1 != len2) return len1-len2;
331 
332  while (len1--)
333  {
334  const int d = *s1++ - *s2++;
335  if (d) return d;
336  }
337 
338  if (lzdiff) return lzdiff;
339  }
340  else
341  {
342  int c1 = *s1++, c2 = *s2++;
343  if (c1 != c2)
344  {
345  if (flags & WDL_STRCMP_LOGICAL_EX_FLAG_OLDSORT)
346  {
347  if (!case_sensitive)
348  {
349  if (c1>='a' && c1<='z') c1+='A'-'a';
350  if (c2>='a' && c2<='z') c2+='A'-'a';
351  }
352  }
353  else
354  {
355  c1 = logical_char_order(c1, case_sensitive);
356  c2 = logical_char_order(c2, case_sensitive);
357  }
358  if (c1 != c2) return c1-c2;
359  }
360  else if (!c1) return 0;
361  }
362  }
363  }
