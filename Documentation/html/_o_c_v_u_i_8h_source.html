154  bool parser(const std::string& string2parse, metadata* pmeta)
155  {
156  int SPACE = 32; // Parameters separator
157  std::vector<std::string> parameters(0);
158 
159  // String analysis
160  for (int i = 0 ; i < string2parse.size() ; i++)
161  {
162  if (string2parse[i] == SPACE)
163  {
164  std::string oneParameter= string2parse.substr(0, i);
165  parameters.push_back(oneParameter);
166  string2parse.erase(string2parse.begin(), string2parse.begin()+i+1);
167  i = 0;
168  }
169  }
170  std::string lastParameter = string2parse;
171 
172  parameters.push_back(lastParameter);
173 
174  // Store Parameters in a Metadata Structure
175 
176  // Parameters count must be 2
177  if (parameters.size() == 2)
178  {
186  if (parameters[0]=="red")
187  {
188  pmeta->color = 1;
189  }
190  else if (parameters[0]=="yellow")
191  {
192  pmeta->color = 2;
193  }
194  else if (parameters[0]=="green")
195  {
196  pmeta->color = 3;
197  }
198  else if (parameters[0]=="cyan")
199  {
200  pmeta->color = 4;
201  }
202  else if (parameters[0]=="blue")
203  {
204  pmeta->color = 5;
205  }
206  else if (parameters[0]=="magenta")
207  {
208  pmeta->color = 6;
209  }
210  pmeta->param = parameters[1];
211  pmeta->used = false;
212  return true;
213  }
214  else
215  {
216  std::cout << "Wrong count of parameters. Please check if the OpenCV"
217  << "metadata is correctly defined" << std::endl;

264  void contoursProcess(std::vector<std::vector<cv::Point> > contours, int color)
265  {
266  int tempArea = 0;
267  cv::Rect myRect;
268  for (int j = 0; j<contours.size(); j++)
269  {
270  std::vector<std::vector<cv::Point> > contours_poly( contours.size() );
271  std::vector<cv::Rect> boundRect( contours.size() );
272 
273  if (contours[j].size() > 40) // Do not take care about small
274  // contours
275  {
276  approxPolyDP( cv::Mat(contours[j]), contours_poly[j], 3, true );// Approximate contours to
277  // a polygone
278  boundRect[j] = cv::boundingRect( cv::Mat(contours_poly[j]) ); // Bound a contour in a
279  // rectangle
280  if ((int)boundRect[j].area()>tempArea)
281  {
282  tempArea=(int)boundRect[j].area();
283  myRect = boundRect[j];
284  }
285  }
286  }
287  if (tempArea != 0)
288  {
289  // Create a new object structure to store the object properties
290  object newObject;
291  newObject.color = color;
292  newObject.centerX = myRect.x+myRect.width/2;
293  newObject.centerY = myRect.y+myRect.height/2;
294  newObject.area = 1.5*(float)tempArea/(width_*height_);
295  newObject.radius= (int)MIN(myRect.width/2, myRect.height/2);
296 
297  // Put the new object in the objects storage.
298  objects_storage_.push_back(newObject);
299  }

312  void erodeAndDilate(cv::Mat image)
313  {
314  cv::Mat element;
315  element = cv::getStructuringElement(cv::MORPH_ELLIPSE, cv::Size(3, 3));
316 
317  // Erase small alone pixels
318  // http://docs.opencv.org/modules/imgproc/doc/filtering.html#dilate
319  for (int i = 0; i < 2 ; i++)

321  cv::erode(image, image, element);

328  cv::dilate(image, image, element);

398  void imageProcessing(cv::Mat BGRImage)
399  {
400  height_ = BGRImage.rows;
401  width_ = BGRImage.cols;
402 
403  cv::Mat HsvImage;
404  cv::cvtColor(BGRImage, HsvImage, CV_BGR2HSV); // Convert frame to HSV format
405  // in order to use "inRange"
406 
407  // Mask matrices (red, yellow, green, cyan, blue and magenta)
408  cv::Mat r_mask, y_mask, g_mask, c_mask, b_mask, m_mask;
409 
410  // Objects contours
411  std::vector<std::vector<cv::Point> > r_contours, y_contours, g_contours,
412  c_contours, b_contours, m_contours;
413  std::vector<cv::Vec4i> hierarchy;
414 
415  // Get every pixel whose value is between _min and _max
416  // and put it into a mask
417  cv::inRange(BGRImage, red_min, red_max, r_mask);
418  cv::inRange(BGRImage, yellow_min, yellow_max, y_mask);
419  cv::inRange(BGRImage, green_min, green_max, g_mask);
420  cv::inRange(BGRImage, cyan_min, cyan_max, c_mask);
421  cv::inRange(BGRImage, blue_min, blue_max, b_mask);
422  cv::inRange(BGRImage, magenta_min, magenta_max, m_mask);
423 
424  // Improve masks shapes
425  erodeAndDilate(r_mask);
426  erodeAndDilate(y_mask);
427  erodeAndDilate(g_mask);
428  erodeAndDilate(c_mask);
429  erodeAndDilate(b_mask);
430  erodeAndDilate(m_mask);
431 
432  // Get the shapes contours from masks
433  cv::findContours(r_mask, r_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
434  cv::findContours(y_mask, y_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
435  cv::findContours(g_mask, g_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
436  cv::findContours(c_mask, c_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
437  cv::findContours(b_mask, b_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
438  cv::findContours(m_mask, m_contours, hierarchy, CV_RETR_TREE, CV_CHAIN_APPROX_SIMPLE, cv::Point(0, 0) );
439 
440  // Process every contour. Note that color is taken in account.
441  for (int i = 1; i <= 6; i++)
442  {
443  switch (i)
444  {
445  case 1: // RED
446  contoursProcess(r_contours, 1);
447 
448  break;
449 
450  case 2: // YELLOW
451  contoursProcess(y_contours, 2);
452 
453  break;
454 
455  case 3: // GREEN
456  contoursProcess(g_contours, 3);
457 
458  break;
459 
460  case 4: // CYAN
461  contoursProcess(c_contours, 4);
462 
463  break;
464 
465  case 5: // BLUE
466  contoursProcess(b_contours, 5);
467 
468  break;
469 
470  case 6: // MAGENTA
471  contoursProcess(m_contours, 6);
472 
473  break;
474 
475  default: // You'll have to add a new color...
476  break;
477  }
478  }
479 
480  // Audio parameters setting
481  for (int i = 0; i<objects_storage_.size(); i++)
482  {
483  for (int j = 0; j < parameters_storage_.size(); j++)
484  {
485  if (objects_storage_[i].color == parameters_storage_[j].color
486  && !parameters_storage_[j].used)
487  {
488  if (parameters_storage_[j].param=="color")
489  {
490  *parameters_storage_[j].zone=(float)objects_storage_[i].color;
491  }
492  else if (parameters_storage_[j].param=="x")
493  {
494  *parameters_storage_[j].zone=objects_storage_[i].centerX/width_;
495  }
496  else if (parameters_storage_[j].param=="y")
497  {
498  *parameters_storage_[j].zone=1-(objects_storage_[i].centerY/height_);
499  }
500  else if (parameters_storage_[j].param=="area")
501  {
502  *parameters_storage_[j].zone=(float)objects_storage_[i].area;
503  }
504  parameters_storage_[j].used=true;
505 
506  drawCircle(objects_storage_[i], BGRImage);
507  }
508  }
509  }

668
