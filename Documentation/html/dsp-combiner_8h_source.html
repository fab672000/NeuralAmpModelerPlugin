385  fDSP1Inputs[chan][0] = fDSP2Outputs[chan][0];

436 static dsp* createDSPSplitter(dsp* dsp1, dsp* dsp2, std::string& error)
437 {
438  if (dsp1->getNumOutputs() == 0) {
439  error = "Connection error in dsp_splitter : the first expression has no outputs\n";
440  return nullptr;
441  } else if (dsp2->getNumInputs() == 0) {
442  error = "Connection error in dsp_splitter : the second expression has no inputs\n";
443  return nullptr;
444  } else if (dsp2->getNumInputs() % dsp1->getNumOutputs() != 0) {
445  std::stringstream error_aux;
446  error_aux << "Connection error in dsp_splitter : the number of outputs (" << dsp1->getNumOutputs()
447  << ") of the first expression should be a divisor of the number of inputs ("
448  << dsp2->getNumInputs()
449  << ") of the second expression" << std::endl;
450  error = error_aux.str();
451  return nullptr;
452  } else if (dsp2->getNumInputs() == dsp1->getNumOutputs()) {
453  return new dsp_sequencer(dsp1, dsp2);
