67  void audioIn(ofSoundBuffer& input)

69  // Keep the input buffer to be used in 'audioOut' for the same audio cycle

73  void audioOut(ofSoundBuffer& output)
74  {
75  AVOIDDENORMALS;
76 
77  // Interleave input (= fInBuffer)
78  for (int chan = 0; chan < fDSP->getNumInputs(); chan++) {
79  for (int frame = 0; frame < output.getNumFrames(); frame++) {
80  fNIInputs[chan][frame] = fInBuffer[chan + frame * fDSP->getNumInputs()];
81  }
82  }
83 
84  fDSP->compute(output.getNumFrames(), fNIInputs, fNIOutputs);
85 
86  // Interleave output
87  for (int chan = 0; chan < fDSP->getNumOutputs(); chan++) {
88  for (int frame = 0; frame < output.getNumFrames(); frame++) {
89  output[chan + frame * fDSP->getNumOutputs()] = fNIOutputs[chan][frame];
90  }
91  }
92  }
