127 // Runs a kernel via serialized buffer. Does this by deserializing the buffer
128 // to construct the arguments, calling the kernel, then re-serialing the outputs.
129 // The buffer contains
130 // [ input_buffer_size, args ]
131 // After the kernel call, it is then populated with
132 // [ output_buffer_size, output_parameters, return_value ]
133 // If the output_buffer_size exceeds the buffer's capacity, then only the
134 // output_buffer_size is populated.
135 template<typename Kernel, typename... Args, size_t... Indices, size_t... OutputIndices>
136 EIGEN_DEVICE_FUNC
137 void run_serialized(std::index_sequence<Indices...>, std::index_sequence<OutputIndices...>,
138  Kernel kernel, uint8_t* buffer, size_t capacity) {
139  using test_detail::get;
140  using test_detail::make_tuple;
141  using test_detail::tuple;
142  // Deserialize input size and inputs.
143  size_t input_size;
144  const uint8_t* read_ptr = buffer;
145  const uint8_t* read_end = buffer + capacity;
146  read_ptr = Eigen::deserialize(read_ptr, read_end, input_size);
147  // Create value-type instances to populate.
148  auto args = make_tuple(typename std::decay<Args>::type{}...);
149  EIGEN_UNUSED_VARIABLE(args) // Avoid NVCC compile warning.
150  // NVCC 9.1 requires us to spell out the template parameters explicitly.
151  read_ptr = Eigen::deserialize(read_ptr, read_end, get<Indices, typename std::decay<Args>::type...>(args)...);
152 
153  // Call function, with void->Void conversion so we are guaranteed a complete
154  // output type.
155  auto result = void_helper::call(kernel, get<Indices, typename std::decay<Args>::type...>(args)...);
156 
157  // Determine required output size.
158  size_t output_size = Eigen::serialize_size(capacity);
159  output_size += Eigen::serialize_size(get<OutputIndices, typename std::decay<Args>::type...>(args)...);
160  output_size += Eigen::serialize_size(result);
161 
162  // Always serialize required buffer size.
163  uint8_t* write_ptr = buffer;
164  uint8_t* write_end = buffer + capacity;
165  write_ptr = Eigen::serialize(write_ptr, write_end, output_size);
166  // Null `write_ptr` can be safely passed along.
167  // Serialize outputs if they fit in the buffer.
168  if (output_size <= capacity) {
169  // Collect outputs and result.
170  write_ptr = Eigen::serialize(write_ptr, write_end, get<OutputIndices, typename std::decay<Args>::type...>(args)...);
171  write_ptr = Eigen::serialize(write_ptr, write_end, result);
172  }
173 }

203 // Runs kernel(args...) on the GPU via the serialization mechanism.
204 //
205 // Note: this may end up calling the kernel multiple times if the initial output
206 // buffer is not large enough to hold the outputs.
207 template<typename Kernel, typename... Args, size_t... Indices, size_t... OutputIndices>
208 auto run_serialized_on_gpu(size_t buffer_capacity_hint,
209  std::index_sequence<Indices...>,
210  std::index_sequence<OutputIndices...>,
211  Kernel kernel, Args&&... args) -> decltype(kernel(args...)) {
212  // Compute the required serialization buffer capacity.
213  // Round up input size to next power of two to give a little extra room
214  // for outputs.
215  size_t input_data_size = sizeof(size_t) + Eigen::serialize_size(args...);
216 
217  size_t capacity;
218  if (buffer_capacity_hint == 0) {
219  // Estimate as the power of two larger than the total input size.
220  capacity = sizeof(size_t);
221  while (capacity <= input_data_size) {
222  capacity *= 2;
223  }
224  } else {
225  // Use the larger of the hint and the total input size.
226  // Add sizeof(size_t) to the hint to account for storing the buffer capacity
227  // itself so the user doesn't need to think about this.
228  capacity = std::max<size_t>(buffer_capacity_hint + sizeof(size_t),
229  input_data_size);
230  }
231  std::vector<uint8_t> buffer(capacity);
232 
233  uint8_t* host_data = nullptr;
234  uint8_t* host_data_end = nullptr;
235  uint8_t* host_ptr = nullptr;
236  uint8_t* device_data = nullptr;
237  size_t output_data_size = 0;
238 
239  // Allocate buffers and copy input data.
240  capacity = std::max<size_t>(capacity, output_data_size);
241  buffer.resize(capacity);
242  host_data = buffer.data();
243  host_data_end = buffer.data() + capacity;
244  host_ptr = Eigen::serialize(host_data, host_data_end, input_data_size);
245  host_ptr = Eigen::serialize(host_ptr, host_data_end, args...);
246 
247  // Copy inputs to host.
248  gpuMalloc((void**)(&device_data), capacity);
249  gpuMemcpy(device_data, buffer.data(), input_data_size, gpuMemcpyHostToDevice);
250  GPU_CHECK(gpuDeviceSynchronize());
251 
252  // Run kernel.
253  #ifdef EIGEN_USE_HIP
254  hipLaunchKernelGGL(
255  HIP_KERNEL_NAME(run_serialized_on_gpu_meta_kernel<Kernel, Args...>),
256  1, 1, 0, 0, kernel, device_data, capacity);
257  #else
258  run_serialized_on_gpu_meta_kernel<Kernel, Args...><<<1,1>>>(
259  kernel, device_data, capacity);
260  #endif
261  // Check pre-launch and kernel execution errors.
262  GPU_CHECK(gpuGetLastError());
263  GPU_CHECK(gpuDeviceSynchronize());
264  // Copy back new output to host.
265  gpuMemcpy(host_data, device_data, capacity, gpuMemcpyDeviceToHost);
266  gpuFree(device_data);
267  GPU_CHECK(gpuDeviceSynchronize());
268 
269  // Determine output buffer size.
270  const uint8_t* c_host_ptr = Eigen::deserialize(host_data, host_data_end, output_data_size);
271  // If the output doesn't fit in the buffer, spit out warning and fail.
272  if (output_data_size > capacity) {
273  std::cerr << "The serialized output does not fit in the output buffer, "
274  << output_data_size << " vs capacity " << capacity << "."
275  << std::endl
276  << "Try specifying a minimum buffer capacity: " << std::endl
277  << " run_with_hint(" << output_data_size << ", ...)"
278  << std::endl;
279  VERIFY(false);
280  }
281 
282  // Deserialize outputs.
283  auto args_tuple = test_detail::tie(args...);
284  EIGEN_UNUSED_VARIABLE(args_tuple) // Avoid NVCC compile warning.
285  c_host_ptr = Eigen::deserialize(c_host_ptr, host_data_end, test_detail::get<OutputIndices, Args&...>(args_tuple)...);
286 
287  // Maybe deserialize return value, properly handling void.
288  typename void_helper::ReturnType<decltype(kernel(args...))> result;
289  c_host_ptr = Eigen::deserialize(c_host_ptr, host_data_end, result);
290  return void_helper::restore(result);
291 }
