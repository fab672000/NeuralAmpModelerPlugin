481  // Ptr and UPtr are an index into the registers args[]. The two styles of using args are
482  // varyings and uniforms. Varyings use Ptr, have a stride associated with them, and are
483  // evaluated everytime through the loop. Uniforms use UPtr, don't have a stride, and are
484  // usually hoisted above the loop.
485  struct Ptr {
486  Ptr() = default;
487  Ptr(int ix_) : ix(ix_) {}
488  int ix;
