46 /*
47 
48  Note about the OSC addresses and the Faust UI names:
49  ----------------------------------------------------
50  There are potential conflicts between the Faust UI objects naming scheme and
51  the OSC address space. An OSC symbolic names is an ASCII string consisting of
52  printable characters other than the following:
53  space
54  # number sign
55  * asterisk
56  , comma
57  / forward
58  ? question mark
59  [ open bracket
60  ] close bracket
61  { open curly brace
62  } close curly brace
63 
64  a simple solution to address the problem consists in replacing
65  space or tabulation with '_' (underscore)
66  all the other osc excluded characters with '-' (hyphen)
67 
68  This solution is implemented in the proposed OSC UI;
69  */
70

80  const char* tr(const char* label) const
81  {
82  static char buffer[1024];
83  char * ptr = buffer; int n=1;
84  while (*label && (n++ < 1024)) {
85  switch (*label) {
86  case ' ': case ' ':
87  *ptr++ = '_';
88  break;
89  case '#': case '*': case ',': case '/': case '?':
90  case '[': case ']': case '{': case '}': case '(': case ')':
91  *ptr++ = '_';
92  break;
93  default:
94  *ptr++ = *label;
95  }
96  label++;
97  }
98  *ptr = 0;
99  return buffer;
100  }
