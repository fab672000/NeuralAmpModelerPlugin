79 class MIDIOutputCallbackHelper
80 {
81 public:
82  MIDIOutputCallbackHelper ()
83  {
84  mMIDIMessageList.reserve (16);
85  mMIDICallbackStruct.midiOutputCallback = NULL;
86  }
87  virtual ~MIDIOutputCallbackHelper () {};
88 
89  void SetCallbackInfo (AUMIDIOutputCallback callback, void* userData)
90  {
91  mMIDICallbackStruct.midiOutputCallback = callback;
92  mMIDICallbackStruct.userData = userData;
93  }
94 
95  void AddEvent (UInt8 status, UInt8 channel, UInt8 data1, UInt8 data2, UInt32 inStartFrame)
96  {
97  MIDIMessageInfoStruct info = {status, channel, data1, data2, inStartFrame};
98  mMIDIMessageList.push_back (info);
99  }
100 
101  void FireAtTimeStamp (const AudioTimeStamp& inTimeStamp)
102  {
103  if (!mMIDIMessageList.empty () && mMIDICallbackStruct.midiOutputCallback != 0)
104  {
105  // synthesize the packet list and call the MIDIOutputCallback
106  // iterate through the vector and get each item
107  std::vector<MIDIMessageInfoStruct>::iterator myIterator;
108  MIDIPacketList* pktlist = PacketList ();
109 
110  for (myIterator = mMIDIMessageList.begin (); myIterator != mMIDIMessageList.end ();
111  myIterator++)
112  {
113  MIDIMessageInfoStruct item = *myIterator;
114 
115  MIDIPacket* pkt = MIDIPacketListInit (pktlist);
116  bool tooBig = false;
117  Byte data[3] = {item.status, item.data1, item.data2};
118  if ((pkt = MIDIPacketListAdd (pktlist, sizeof (mBuffersAllocated), pkt,
119  item.startFrame, 4, const_cast<Byte*> (data))) ==
120  NULL)
121  tooBig = true;
122 
123  if (tooBig)
124  { // send what we have and then clear the buffer and send again
125  // issue the callback with what we got
126  OSStatus result = mMIDICallbackStruct.midiOutputCallback (
127  mMIDICallbackStruct.userData, &inTimeStamp, 0, pktlist);
128  if (result != noErr)
129  printf ("error calling output callback: %d", (int)result);
130 
131  // clear stuff we've already processed, and fire again
132  mMIDIMessageList.erase (mMIDIMessageList.begin (), myIterator);
133  this->FireAtTimeStamp (inTimeStamp);
134  return;
135  }
136  }
137  // fire callback
138  OSStatus result = mMIDICallbackStruct.midiOutputCallback (mMIDICallbackStruct.userData,
139  &inTimeStamp, 0, pktlist);
140  if (result != noErr)
141  printf ("error calling output callback: %d", (int)result);
142 
143  mMIDIMessageList.clear ();
144  }
145  }
146 
147 protected:
148  typedef std::vector<MIDIMessageInfoStruct> MIDIMessageList;
149 
150 private:
151  MIDIPacketList* PacketList () { return (MIDIPacketList*)mBuffersAllocated; }
152 
153  Byte mBuffersAllocated[1024];
154  AUMIDIOutputCallbackStruct mMIDICallbackStruct;
155  MIDIMessageList mMIDIMessageList;
156 };
