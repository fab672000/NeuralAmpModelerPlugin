82 /*
83  * A segment of user/application data that a QR Code symbol can convey.
84  * Each segment has a mode, a character count, and character/general data that is
85  * already encoded as a sequence of bits. The maximum allowed bit length is 32767,
86  * because even the largest QR Code (version 40) has only 31329 modules.
87  */
88 struct qrcodegen_Segment {
89  // The mode indicator for this segment.
90  enum qrcodegen_Mode mode;
91 
92  // The length of this segment's unencoded data. Always in the range [0, 32767].
93  // For numeric, alphanumeric, and kanji modes, this measures in Unicode code points.
94  // For byte mode, this measures in bytes (raw binary data, text in UTF-8, or other encodings).
95  // For ECI mode, this is always zero.
96  int numChars;
97 
98  // The data bits of this segment, packed in bitwise big endian.
99  // Can be null if the bit length is zero.
100  uint8_t *data;
101 
102  // The number of valid data bits used in the buffer. Requires
103  // 0 <= bitLength <= 32767, and bitLength <= (capacity of data array) * 8.
104  int bitLength;
105 };

111 // The minimum and maximum defined QR Code version numbers for Model 2.
112 #define qrcodegen_VERSION_MIN 1
113 #define qrcodegen_VERSION_MAX 40
114 
115 // Calculates the number of bytes needed to store any QR Code up to and including the given version number,
116 // as a compile-time constant. For example, 'uint8_t buffer[qrcodegen_BUFFER_LEN_FOR_VERSION(25)];'
117 // can store any single QR Code from version 1 to 25, inclusive.
118 // Requires qrcodegen_VERSION_MIN <= n <= qrcodegen_VERSION_MAX.
119 #define qrcodegen_BUFFER_LEN_FOR_VERSION(n) ((((n) * 4 + 17) * ((n) * 4 + 17) + 7) / 8 + 1)
120 
121 // The worst-case number of bytes needed to store one QR Code, up to and including
122 // version 40. This value equals 3918, which is just under 4 kilobytes.
123 // Use this more convenient value to avoid calculating tighter memory bounds for buffers.
124 #define qrcodegen_BUFFER_LEN_MAX qrcodegen_BUFFER_LEN_FOR_VERSION(qrcodegen_VERSION_MAX)

132 /*
133  * Encodes the given text string to a QR Code symbol, returning true if encoding succeeded.
134  * If the data is too long to fit in any version in the given range
135  * at the given ECC level, then false is returned.
136  * - The input text must be encoded in UTF-8 and contain no NULs.
137  * - The variables ecl and mask must correspond to enum constant values.
138  * - Requires 1 <= minVersion <= maxVersion <= 40.
139  * - The arrays tempBuffer and qrcode must each have a length
140  * of at least qrcodegen_BUFFER_LEN_FOR_VERSION(maxVersion).
141  * - After the function returns, tempBuffer contains no useful data.
142  * - If successful, the resulting QR Code may use numeric,
143  * alphanumeric, or byte mode to encode the text.
144  * - In the most optimistic case, a QR Code at version 40 with low ECC
145  * can hold any UTF-8 string up to 2953 bytes, or any alphanumeric string
146  * up to 4296 characters, or any digit string up to 7089 characters.
147  * These numbers represent the hard upper limit of the QR Code standard.
148  * - Please consult the QR Code specification for information on
149  * data capacities per version, ECC level, and text encoding mode.
150  */
151 static bool qrcodegen_encodeText(const char *text, uint8_t tempBuffer[], uint8_t qrcode[],
152  enum qrcodegen_Ecc ecl, int minVersion, int maxVersion, enum qrcodegen_Mask mask, bool boostEcl);

155 /*
156  * Encodes the given binary data to a QR Code symbol, returning true if encoding succeeded.
157  * If the data is too long to fit in any version in the given range
158  * at the given ECC level, then false is returned.
159  * - The input array range dataAndTemp[0 : dataLen] should normally be
160  * valid UTF-8 text, but is not required by the QR Code standard.
161  * - The variables ecl and mask must correspond to enum constant values.
162  * - Requires 1 <= minVersion <= maxVersion <= 40.
163  * - The arrays dataAndTemp and qrcode must each have a length
164  * of at least qrcodegen_BUFFER_LEN_FOR_VERSION(maxVersion).
165  * - After the function returns, the contents of dataAndTemp may have changed,
166  * and does not represent useful data anymore.
167  * - If successful, the resulting QR Code will use byte mode to encode the data.
168  * - In the most optimistic case, a QR Code at version 40 with low ECC can hold any byte
169  * sequence up to length 2953. This is the hard upper limit of the QR Code standard.
170  * - Please consult the QR Code specification for information on
171  * data capacities per version, ECC level, and text encoding mode.
172  */

189 /*
190  * Returns the number of bytes (uint8_t) needed for the data buffer of a segment
191  * containing the given number of characters using the given mode. Notes:
192  * - Returns SIZE_MAX on failure, i.e. numChars > INT16_MAX or
193  * the number of needed bits exceeds INT16_MAX (i.e. 32767).
194  * - Otherwise, all valid results are in the range [0, ceil(INT16_MAX / 8)], i.e. at most 4096.
195  * - It is okay for the user to allocate more bytes for the buffer than needed.
196  * - For byte mode, numChars measures the number of bytes, not Unicode code points.
197  * - For ECI mode, numChars must be 0, and the worst-case number of bytes is returned.
198  * An actual ECI segment can have shorter data. For non-ECI modes, the result is exact.
199  */
200 static size_t qrcodegen_calcSegmentBufferSize(enum qrcodegen_Mode mode, size_t numChars);

230 /*
231  * Renders a QR Code symbol representing the given data segments at the given error correction
232  * level or higher. The smallest possible QR Code version is automatically chosen for the output.
233  * Returns true if QR Code creation succeeded, or false if the data is too long to fit in any version.
234  * This function allows the user to create a custom sequence of segments that switches
235  * between modes (such as alphanumeric and binary) to encode text more efficiently.
236  * This function is considered to be lower level than simply encoding text or binary data.
237  * To save memory, the segments' data buffers can alias/overlap tempBuffer, and will
