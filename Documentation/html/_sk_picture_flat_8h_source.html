154 // clipparams are packed in 5 bits
155 // doAA:1 | clipOp:4

157 // Newly serialized pictures will only write kIntersect or kDifference.
158 static inline uint32_t ClipParams_pack(SkClipOp op, bool doAA) {
159  unsigned doAABit = doAA ? 1 : 0;
160  return (doAABit << 4) | static_cast<int>(op);
161 }

163 // But old SKPs may have been serialized with the SK_SUPPORT_DEPRECATED_CLIPOP flag, so might
164 // encounter expanding clip ops. Thus, this returns the clip op as the more general Region::Op.
165 static inline SkRegion::Op ClipParams_unpackRegionOp(SkReadBuffer* buffer, uint32_t packed) {
166  uint32_t unpacked = packed & 0xF;
167  if (buffer->validate(unpacked <= SkRegion::kIntersect_Op ||
168  (unpacked <= SkRegion::kReplace_Op &&
169  buffer->isVersionLT(SkPicturePriv::kNoExpandingClipOps)))) {
170  return static_cast<SkRegion::Op>(unpacked);
171  }
172  return SkRegion::kIntersect_Op;
173 }

175 static inline bool ClipParams_unpackDoAA(uint32_t packed) {
176  return SkToBool((packed >> 4) & 1);
