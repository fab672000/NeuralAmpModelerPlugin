216 SMTG_ALWAYS_INLINE ParamValue Parameter::advance (int32 numSamples) noexcept
217 {
218  if (pointCount < 0)
219  return currentValue;
220  while (valuePoint.sampleOffset >= 0 && valuePoint.sampleOffset < numSamples)
221  {
222  sampleCounter += valuePoint.sampleOffset;
223  numSamples -= valuePoint.sampleOffset;
224  currentValue = valuePoint.value;
225  valuePoint = processNextValuePoint ();
226  }
227  currentValue += (valuePoint.rampPerSample * numSamples);
228  valuePoint.sampleOffset -= numSamples;
229  sampleCounter += numSamples;
230  return currentValue;
231 }

284 SMTG_ALWAYS_INLINE auto Parameter::processNextValuePoint () noexcept -> ValuePoint
285 {
286  ValuePoint nv;
287  if (pointCount == 0 || queue->getPoint (pointIndex, nv.sampleOffset, nv.value) != kResultTrue)
288  {
289  pointCount = -1;
290  return {currentValue, 0., -1};
291  }
292  nv.sampleOffset -= sampleCounter;
293  ++pointIndex;
294  --pointCount;
295  if (nv.sampleOffset == 0)
296  nv.rampPerSample = (nv.value - currentValue);
297  else
298  nv.rampPerSample = (nv.value - currentValue) / static_cast<double> (nv.sampleOffset);
299  return nv;
300 }
