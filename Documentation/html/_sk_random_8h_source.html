127  // Initialize state variables with LCG.
128  // We must ensure that both J and K are non-zero, otherwise the
129  // multiply-with-carry step will forevermore return zero.
130  void init(uint32_t seed) {
131  fK = NextLCG(seed);
132  if (0 == fK) {
133  fK = NextLCG(fK);
134  }
135  fJ = NextLCG(fK);

138  }
