76 struct CircularBuffer {

78  short* fBuffer;
79  int fReadIndex;
80  int fWriteIndex;
81  int fSize;

83 
84  CircularBuffer(int frames, int chan)
85  {
86  fBuffer = new short[frames * chan];
87  memset(fBuffer, 0, sizeof(short) * frames * chan);
88  fSize = frames;
89  fChan = chan;
90  fReadIndex = 0;
91  fWriteIndex = frames/2; // Set write index in the middle
92  }
93 
94  ~CircularBuffer()
95  {
96  delete [] fBuffer;
97  }
98 
99  short* getWritePtr() { return &fBuffer[fWriteIndex * fChan]; }
100  short* getReadPtr() { return &fBuffer[fReadIndex * fChan]; }
101 
102  void moveWritePtr(int frames)
103  {
104  //__android_log_print(ANDROID_LOG_ERROR, "Faust", "moveWritePtr %x fWriteIndex = %ld", this, fWriteIndex);
105  fWriteIndex = (fWriteIndex + frames) % fSize;
106  }
107  void moveReadPtr(int frames)
108  {
109  //__android_log_print(ANDROID_LOG_ERROR, "Faust", "moveReadPtr %x fReadIndex = %ld", this, fReadIndex);
110  fReadIndex = (fReadIndex + frames) % fSize;
111  }
112 
113 };

151  void processAudio()
152  {
153  int64_t t1 = getTimeUsec();
154 
155  // Converting short input to float
156  if (fNumInChans > 0) {
157  short* input = fOpenSLInputs.getReadPtr();
158  for (int chan = 0; chan < NUM_INPUTS; chan++) {
159  for (int i = 0; i < fBufferSize; i++) {
160  fInputs[chan][i] = float(input[i * NUM_INPUTS + chan] * CONVMYFLT);
161  }
162  }
163  fOpenSLInputs.moveReadPtr(fBufferSize);
164  }
165 
166  // Compute DSP
167  fDSP->compute(fBufferSize, fInputs, fOutputs);
168 
169  runControlCallbacks();
170 
171  // Converting float to short output
172  if (fNumOutChans > 0) {
173  short* output = fOpenSLOutputs.getWritePtr();
174  for (int chan = 0; chan < NUM_OUTPUTS; chan++) {
175  for (int i = 0; i < fBufferSize; i++) {
176  output[i * NUM_OUTPUTS + chan] = short(min(1.f, max(-1.f, fOutputs[chan][i])) * CONV16BIT);
177  }
178  }
179  fOpenSLOutputs.moveWritePtr(fBufferSize);
180  }
181 
182  int64_t t2 = getTimeUsec();
183  fCPUTable[(fCPUTableIndex++)&(CPU_TABLE_SIZE-1)] = t2 - t1;
184  }
