22 template<typename Index, int Mode, typename LhsScalar, bool ConjLhs, typename RhsScalar>
23 struct band_solve_triangular_selector<Index,Mode,LhsScalar,ConjLhs,RhsScalar,RowMajor>
24 {
25  typedef Map<const Matrix<LhsScalar,Dynamic,Dynamic,RowMajor>, 0, OuterStride<> > LhsMap;
26  typedef Map<Matrix<RhsScalar,Dynamic,1> > RhsMap;
27  enum { IsLower = (Mode&Lower) ? 1 : 0 };
28  static void run(Index size, Index k, const LhsScalar* _lhs, Index lhsStride, RhsScalar* _other)
29  {
30  const LhsMap lhs(_lhs,size,k+1,OuterStride<>(lhsStride));
31  RhsMap other(_other,size,1);
32  std::conditional_t<ConjLhs,
33  const CwiseUnaryOp<typename internal::scalar_conjugate_op<LhsScalar>,LhsMap>,
34  const LhsMap&> cjLhs(lhs);
35 
36  for(int col=0 ; col<other.cols() ; ++col)
37  {
38  for(int ii=0; ii<size; ++ii)
39  {
40  int i = IsLower ? ii : size-ii-1;
41  int actual_k = (std::min)(k,ii);
42  int actual_start = IsLower ? k-actual_k : 1;
43 
44  if(actual_k>0)
45  other.coeffRef(i,col) -= cjLhs.row(i).segment(actual_start,actual_k).transpose()
46  .cwiseProduct(other.col(col).segment(IsLower ? i-actual_k : i+1,actual_k)).sum();
47 
48  if((Mode&UnitDiag)==0)
49  other.coeffRef(i,col) /= cjLhs(i,IsLower ? k : 0);
50  }
51  }
52  }
53 
54 };

62  static void run(Index size, Index k, const LhsScalar* _lhs, Index lhsStride, RhsScalar* _other)
63  {
64  const LhsMap lhs(_lhs,k+1,size,OuterStride<>(lhsStride));
65  RhsMap other(_other,size,1);
66  std::conditional_t<ConjLhs,
67  const CwiseUnaryOp<typename internal::scalar_conjugate_op<LhsScalar>,LhsMap>,
68  const LhsMap&> cjLhs(lhs);
69 
70  for(int col=0 ; col<other.cols() ; ++col)
71  {
72  for(int ii=0; ii<size; ++ii)
73  {
74  int i = IsLower ? ii : size-ii-1;
75  int actual_k = (std::min)(k,size-ii-1);
76  int actual_start = IsLower ? 1 : k-actual_k;
77 
78  if((Mode&UnitDiag)==0)
79  other.coeffRef(i,col) /= cjLhs(IsLower ? 0 : k, i);
80 
81  if(actual_k>0)
82  other.col(col).segment(IsLower ? i+1 : i-actual_k, actual_k)
83  -= other.coeff(i,col) * cjLhs.col(i).segment(actual_start,actual_k);
84 
85  }
