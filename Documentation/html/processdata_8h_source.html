107 inline bool HostProcessData::isValidBus (BusDirection dir, int32 busIndex) const
108 {
109  if (dir == kInput && (!inputs || busIndex >= numInputs))
110  return false;
111  if (dir == kOutput && (!outputs || busIndex >= numOutputs))
112  return false;
113  return true;
