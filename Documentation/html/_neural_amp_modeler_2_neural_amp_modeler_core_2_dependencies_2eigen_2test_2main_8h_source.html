25 // The following includes of STL headers have to be done _before_ the
26 // definition of macros min() and max(). The reason is that many STL
27 // implementations will not work properly as the min and max symbols collide
28 // with the STL functions std::min() and std::max(). The STL headers may check
29 // for the macro definition of min/max and issue a warning or undefine the
30 // macros.
31 //
32 // Still, Windows defines min() and max() in windef.h as part of the regular
33 // Windows system interfaces and many other Windows APIs depend on these
34 // macros being available. To prevent the macro expansion of min/max and to
35 // make Eigen compatible with the Windows environment all function calls of
36 // std::min() and std::max() have to be written with parenthesis around the
37 // function name.
38 //
39 // All STL headers used by Eigen should be included here. Because main.h is
40 // included before any Eigen header and because the STL headers are guarded
41 // against multiple inclusions, no STL header will see our own min/max macro
42 // definitions.

89 // To test that all calls from Eigen code to std::min() and std::max() are

491 // test_relative_error returns the relative difference between a and b as a real scalar as used in isApprox.
492 template<typename T1,typename T2>
493 typename NumTraits<typename T1::RealScalar>::NonInteger test_relative_error(const EigenBase<T1> &a, const EigenBase<T2> &b)
494 {
495  using std::sqrt;
496  typedef typename NumTraits<typename T1::RealScalar>::NonInteger RealScalar;
497  typename internal::nested_eval<T1,2>::type ea(a.derived());
498  typename internal::nested_eval<T2,2>::type eb(b.derived());
499  return sqrt(RealScalar((ea.matrix()-eb.matrix()).cwiseAbs2().sum()) / RealScalar((std::min)(eb.cwiseAbs2().sum(),ea.cwiseAbs2().sum())));
500 }

652 // Checks component-wise, works with infs and nans.
653 template<typename Derived1, typename Derived2>
654 bool test_isCwiseApprox(const DenseBase<Derived1>& m1,
655  const DenseBase<Derived2>& m2,
656  bool exact) {
657  if (m1.rows() != m2.rows()) {
658  return false;
659  }
660  if (m1.cols() != m2.cols()) {
661  return false;
662  }
663  for (Index r = 0; r < m1.rows(); ++r) {
664  for (Index c = 0; c < m1.cols(); ++c) {
665  if (m1(r, c) != m2(r, c)
666  && !((numext::isnan)(m1(r, c)) && (numext::isnan)(m2(r, c)))
667  && (exact || !test_isApprox(m1(r, c), m2(r, c)))) {
668  return false;
669  }
670  }
671  }
672  return true;
673 }
