79  template <SymbolicSampleSizes SampleSize, typename DoProcessCallback>
80  void process (ProcessData& data, DoProcessCallback doProcessing) noexcept
81  {
82  stopIt = false;
83  auto numSamples = data.numSamples;
84  auto samplesLeft = data.numSamples;
85  while (samplesLeft > 0 && !stopIt)
86  {
87  auto currentSliceSize = samplesLeft > sliceSize ? sliceSize : samplesLeft;
88 
89  data.numSamples = currentSliceSize;
90  doProcessing (data);
91 
92  advanceBuffers<SampleSize> (data.inputs, data.numInputs, currentSliceSize);
93  advanceBuffers<SampleSize> (data.outputs, data.numOutputs, currentSliceSize);
94  samplesLeft -= currentSliceSize;
95  }
96  // revert buffer pointers (otherwise some hosts may use these wrong pointers)
97  advanceBuffers<SampleSize> (data.inputs, data.numInputs, -(numSamples - samplesLeft));
98  advanceBuffers<SampleSize> (data.outputs, data.numOutputs, -(numSamples - samplesLeft));
99  data.numSamples = numSamples;
100  }
101 
107  void stop () noexcept { stopIt = true; }
108 
109 private:
110  template <SymbolicSampleSizes SampleSize>
111  void advanceBuffers (AudioBusBuffers* buffers, int32 numBuffers, int32 numSamples) const
112  noexcept
113  {
114  for (auto index = 0; index < numBuffers; ++index)
115  {
116  for (auto channelIndex = 0; channelIndex < buffers[index].numChannels; ++channelIndex)
117  {
118  if (SampleSize == SymbolicSampleSizes::kSample32)
119  buffers[index].channelBuffers32[channelIndex] += numSamples;
120  else
121  buffers[index].channelBuffers64[channelIndex] += numSamples;
122  }
123  }
124  }
125  int32 sliceSize;
126  bool stopIt {false};
127 };
