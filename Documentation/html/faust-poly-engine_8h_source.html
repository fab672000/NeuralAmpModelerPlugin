65  void init(dsp* mono_dsp, audio* driver, midi_handler* midi)
66  {
67  bool midi_sync = false;
68  int nvoices = 0;
69  fRunning = false;
70 
71  MidiMeta::analyse(mono_dsp, midi_sync, nvoices);
72 
73  // Getting the UI JSON
74  JSONUI jsonui1(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
75  mono_dsp->buildUserInterface(&jsonui1);
76  fJSONUI = jsonui1.JSON();
77 
78  // Getting the metadata JSON
79  JSONUI jsonui1M(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
80  mono_dsp->metadata(&jsonui1M);
81  fJSONMeta = jsonui1M.JSON();
82 
83  if (nvoices > 0) {
84 
85  fPolyDSP = new mydsp_poly(mono_dsp, nvoices, true);
86  fMidiHandler.addMidiIn(fPolyDSP);
87  if (midi) midi->addMidiIn(fPolyDSP);
88 
89  #if POLY2
90  fFinalDSP = new dsp_sequencer(fPolyDSP, new effect());
91  #else
92  fFinalDSP = fPolyDSP;
93  #endif
94 
95  // Update JSONs with Poly version
96  JSONUI jsonui2(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
97  fFinalDSP->buildUserInterface(&jsonui2);
98  fJSONUI = jsonui2.JSON();
99 
100  JSONUI jsonui2M(mono_dsp->getNumInputs(), mono_dsp->getNumOutputs());
101  fFinalDSP->metadata(&jsonui2M);
102  fJSONMeta = jsonui2M.JSON();
103 
104  } else {
105  fPolyDSP = NULL;
106  fFinalDSP = mono_dsp;
107  }
108 
109  fFinalDSP->buildUserInterface(&fMidiUI);
110  fFinalDSP->buildUserInterface(&fAPIUI);

112  // Retrieving DSP object name
113  struct MyMeta : public Meta
114  {
115  string fName;
116  void declare(const char* key, const char* value)
117  {
118  if (strcmp(key, "name") == 0) fName = value;
119  }
120  MyMeta():fName("Dummy"){}
121  };
122 
123  MyMeta meta;
124  fFinalDSP->metadata(&meta);
125  if (midi) midi->setName(meta.fName);
126 
127  if (driver) {
128  // If driver cannot be initialized, start will fail later on...
129  if (!driver->init(meta.fName.c_str(), fFinalDSP)) {
130  delete driver;
131  fDriver = NULL;
132  } else {
133  fDriver = driver;
134  }
135  } else {
136  fDriver = NULL;
137  }

193  * Instantiates a new polyphonic voice where velocity
