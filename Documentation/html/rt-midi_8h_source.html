48  static void midiCallback(double time, std::vector<unsigned char>* message, void* arg)
49  {
50  rt_midi* midi = static_cast<rt_midi*>(arg);
51  size_t nBytes = message->size();
52  int type = (int)message->at(0) & 0xf0;
53  int channel = (int)message->at(0) & 0x0f;
54 
55  // MIDI sync
56  if (nBytes == 1) {
57  midi->handleSync(time, (int)message->at(0));
58  // One data byte messages
59  } else if (nBytes == 2) {
60  midi->handleData1(time, type, channel, (int)message->at(1));
61  // Two data bytes messages
62  } else if (nBytes == 3) {
63  midi->handleData2(time, type, channel, (int)message->at(1), (int)message->at(2));
64  } else {
65  midi->handleMessage(time, (int)message->at(0), *message);
66  }
67  }

69  bool openMidiInputPorts()
70  {
71  // Get number of input ports
72  RtMidiIn midi_in;
73  unsigned nInPorts = midi_in.getPortCount();
74  if (nInPorts == 0) {
75  std::cout << "No input ports available!" << std::endl;
76  return false;
77  }
78 
79  // Then open all of them
80  for (unsigned int i = 0; i < nInPorts; i++) {
81  RtMidiIn* midi_in = new RtMidiIn();
82  midi_in->ignoreTypes(true, false, true);
83  fInput.push_back(midi_in);
84  midi_in->openPort(i);
85  if (!fPolling) {
86  midi_in->setCallback(&midiCallback, this);
87  }
88  //std::cout << "Input port #" << i << ": " << midi_in->getPortName(i) << '\n';
89  }
90  return true;
91  }

188  // To be used in polling mode
189  int recvMessages(std::vector<MIDIMessage>* messages)
190  {
191  int count = 0;
192  double first_time_stamp = 0.;
193  for (auto& it : fInput) {
194  std::vector<unsigned char> message;
195  double time_stamp = (uint32_t)it->getMessage(&message);
196  // Small messages
197  if (message.size() > 0) {
198  if (count == 0) first_time_stamp = time_stamp;
199  MIDIMessage& mes = messages->at(count++);
200  mes.frameIndex = (uint32_t)(time_stamp - first_time_stamp);
201  mes.byte0 = message[0];
202  mes.byte1 = message[1];
203  mes.byte2 = message[2];
204  }
205  }
206  return count;
207  }

209  void sendMessages(std::vector<MIDIMessage>* messages, int count)
210  {
211  for (int i = 0; i < count; ++i) {
212  MIDIMessage mes1 = (*messages)[i];
213  std::vector<unsigned char> mes2;
214  mes2.push_back(mes1.byte0);
215  mes2.push_back(mes1.byte1);
216  mes2.push_back(mes1.byte2);
217  sendMessage(mes2);
218  }
