107  void setThreadAffinity() {
108 
109  pid_t current_thread_id = gettid();
110  cpu_set_t cpu_set;
111  CPU_ZERO(&cpu_set);
112 
113  // If the callback cpu ids aren't specified then bind to the current cpu
114  if (mCpuIds.empty()) {
115  int current_cpu_id = sched_getcpu();
116  LOGD("Binding to current CPU ID %d", current_cpu_id);
117  CPU_SET(current_cpu_id, &cpu_set);
118  } else {
119  LOGD("Binding to %d CPU IDs", static_cast<int>(mCpuIds.size()));
120  for (size_t i = 0; i < mCpuIds.size(); i++) {
121  int cpu_id = mCpuIds.at(i);
122  LOGD("CPU ID %d added to cores set", cpu_id);
123  CPU_SET(cpu_id, &cpu_set);
124  }
125  }
126 
127  int result = sched_setaffinity(current_thread_id, sizeof(cpu_set_t), &cpu_set);
128  if (result == 0) {
129  LOGV("Thread affinity set");
130  } else {
131  LOGW("Error setting thread affinity. Error no: %d", result);
132  }
133 
134  mIsThreadAffinitySet = true;
135  }
