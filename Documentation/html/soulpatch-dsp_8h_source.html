281  virtual void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
282  {
283  // Update inputs control
284  if (fDecoder) {
285  for (auto& i : fDecoder->getInputControls()) {
286  i->reflectZone();
287  }
288  } else {
289  for (auto& i : fInputsControl) {
290  i->reflectZone();
291  }
292  }
293 
294  // MIDI input handling
295  if (fMIDIHander) {
296  fRenderContext.numMIDIMessagesIn = fMIDIHander->recvMessages(reinterpret_cast<std::vector<MIDIMessage>*>(&fMIDIInputMessages));
297  if (fRenderContext.numMIDIMessagesIn > 1024) {
298  std::cerr << "MIDI input overflow\n";
299  }
300  fRenderContext.incomingMIDI = std::addressof(fMIDIInputMessages[0]);
301  fRenderContext.outgoingMIDI = std::addressof(fMIDIOutputMessages[0]);
302  fRenderContext.maximumMIDIMessagesOut = (uint32_t)fMIDIOutputMessages.size();
303  fRenderContext.numMIDIMessagesOut = 0;
304  } else {
305  fRenderContext.incomingMIDI = nullptr;
306  fRenderContext.incomingMIDI = nullptr;
307  fRenderContext.numMIDIMessagesIn = 0;
308  }
309 
310  // Setup audio buffers
311  fRenderContext.inputChannels = (const FAUSTFLOAT**)inputs;
312  fRenderContext.outputChannels = outputs;
313  fRenderContext.numFrames = count;
314 
315  // DSP compute
316  fPlayer->render(fRenderContext);
317 
318  // MIDI output handling
319  if (fMIDIHander && fRenderContext.numMIDIMessagesOut != 0) {
320  if (fRenderContext.numMIDIMessagesOut > fRenderContext.maximumMIDIMessagesOut) {
321  std::cerr << "MIDI output overflow\n";
322  }
323  int numMessagesOut = std::min(fRenderContext.numMIDIMessagesOut, fRenderContext.maximumMIDIMessagesOut);
324  fMIDIHander->sendMessages(reinterpret_cast<std::vector<MIDIMessage>*>(&fMIDIOutputMessages), numMessagesOut);
325  }
326 
327  // Update outputs control
328  if (fDecoder) {
329  for (auto& i : fDecoder->getOutputControls()) {
330  i->modifyZone();
331  }
332  } else {
333  for (auto& i : fOutputsControl) {
334  i->modifyZone();
335  }
336  }
337  }

542  std::map <std::string, std::string> extractFaustBlocks(std::istream& in, std::stringstream& res_file)
543  {
544  std::string line;
545  std::stringstream faust_block;
546  bool is_faust_block = false;
547  int brackets = 0;
548  std::map <std::string, std::string> faust_blocks; // name, code
549  std::map <std::string, std::string>::iterator cur_faust_block;
550 
551  while (getline(in, line)) {
552 
553  std::stringstream line_reader(line);
554  std::string token1, token2, token3;
555 
556  line_reader >> token1;
557  line_reader >> token2;
558  line_reader >> token3;
559 
560  if (is_faust_block) {
561  // End of block
562  if ((token1 == "}") && (--brackets == 0)) {
563  is_faust_block = false;
564  cur_faust_block->second = faust_block.str();
565  faust_block.str("");
566  // Start of block (or could be on the previous line)
567  } else if (token1 == "{") {
568  brackets++;
569  continue;
570  } else {
571  faust_block << line;
572  }
573  continue;
574  } else {
575  is_faust_block = (token1 == "faust" && token2 != "");
576  if (is_faust_block) {
577  if (token3 == "{") brackets++;
578  faust_blocks[token2] = "";
579  cur_faust_block = faust_blocks.find(token2);
580  }
581  }
582 
583  // Keep the lines of SOUL file
584  if (!is_faust_block) res_file << line << std::endl;
585  }
586 
587  return faust_blocks;
588  }

627  {
