50 /*
51  The soundfile structure to be used by the DSP code. Soundfile has a MAX_SOUNDFILE_PARTS parts
52  (even a single soundfile or an empty soundfile).
53  fLength, fOffset and fSR fields are filled accordingly by repeating
54  the actual parts if needed.
55 
56  It has to be 'packed' to that the LLVM backend can correctly access it.
57 
58  Index computation:
59  - p is the current part number [0..MAX_SOUNDFILE_PARTS-1] (must be proved by the type system)
60  - i is the current position in the part. It will be constrained between [0..length]
61  - idx(p,i) = fOffset[p] + max(0, min(i, fLength[p]));
62 */
63 
64 PRE_PACKED_STRUCTURE
65 struct Soundfile {
66  FAUSTFLOAT** fBuffers;
67  int* fLength; // length of each part
68  int* fSR; // sample rate of each part
69  int* fOffset; // offset of each part in the global buffer
70  int fChannels; // max number of channels of all concatenated files
71 
72  Soundfile()
73  {
74  fBuffers = nullptr;
75  fChannels = -1;
76  fLength = new int[MAX_SOUNDFILE_PARTS];
77  fSR = new int[MAX_SOUNDFILE_PARTS];
78  fOffset = new int[MAX_SOUNDFILE_PARTS];
79  }
80 
81  ~Soundfile()
82  {
83  // Free the real channels only
84  for (int chan = 0; chan < fChannels; chan++) {
85  delete fBuffers[chan];
86  }
87  delete[] fBuffers;
88  delete[] fLength;
89  delete[] fSR;
90  delete[] fOffset;
91  }
92 
93 } POST_PACKED_STRUCTURE;

114  Soundfile* createSoundfile(int cur_chan, int length, int max_chan)
115  {
116  Soundfile* soundfile = new Soundfile();

223  Soundfile* createSoundfile(const std::vector<std::string>& path_name_list, int max_chan)
224  {
225  try {
226  int cur_chan = 1; // At least one buffer
227  int total_length = 0;
228 
229  // Compute total length and channels max of all files
230  for (int i = 0; i < int(path_name_list.size()); i++) {
231  int chan, length;
232  if (path_name_list[i] == "__empty_sound__") {
233  length = BUFFER_SIZE;
234  chan = 1;
235  } else {
236  getParamsFile(path_name_list[i], chan, length);
237  }
238  cur_chan = std::max<int>(cur_chan, chan);
239  total_length += length;
240  }
241 
242  // Complete with empty parts
243  total_length += (MAX_SOUNDFILE_PARTS - path_name_list.size()) * BUFFER_SIZE;
244 
245  // Create the soundfile
246  Soundfile* soundfile = createSoundfile(cur_chan, total_length, max_chan);
247 
248  // Init offset
249  int offset = 0;
250 
251  // Read all files
252  for (int i = 0; i < int(path_name_list.size()); i++) {
253  if (path_name_list[i] == "__empty_sound__") {
254  emptyFile(soundfile, i, offset);
255  } else {
256  readFile(soundfile, path_name_list[i], i, offset, max_chan);
257  }
258  }
259 
260  // Complete with empty parts
261  for (int i = int(path_name_list.size()); i < MAX_SOUNDFILE_PARTS; i++) {
262  emptyFile(soundfile, i, offset);
263  }
264 
265  // Share the same buffers for all other channels so that we have max_chan channels available
266  for (int chan = cur_chan; chan < max_chan; chan++) {
267  soundfile->fBuffers[chan] = soundfile->fBuffers[chan % cur_chan];
268  }
269 
270  return soundfile;
271 
272  } catch (...) {
273  return nullptr;
274  }
275  }
