425  auto &c_reg = zmm[c_regs[i + idx * 3]];
426  auto &c_load_reg = zmm[c_load_regs[i % 3]];
427  auto c_mem = cox;
428  if (is_unit_inc)

468  * co2 = co1 + ldc;
469  *
470  * auto &alpha_reg = zmm[alpha_load_reg];
471  * if (!is_alpha1) alpha_reg = pload1<vec>(alpha);
472  *
473  * int idx = 0;
474  * for (pow = 1; pow <= 8; pow <<= 1) {
475  *
476  * if (b_unroll >= pow) {
477  * for (count = 1; count < (pow + 1) / 2 + 1; count++) {
478  * if (pow >= 4) co2 += ldc;

491  * if (b_unroll == 1)

522  template <int max_b_unroll, int a_unroll, int b_unroll>
523  EIGEN_ALWAYS_INLINE void c_update(Scalar *&co1, Scalar *&co2) {
524  auto &alpha_reg = zmm[alpha_load_reg];
525 
526  co2 = co1 + ldc;
527  if (!is_alpha1) alpha_reg = pload1<vec>(alpha);
528  if (!is_unit_inc && a_unroll < nelems_in_cache_line) mask = static_cast<umask_t>((1ull << a_unroll) - 1);
529 
530  static_assert(max_b_unroll <= 8, "Unsupported max_b_unroll");
531 
532  if (1 <= max_b_unroll && 1 <= b_unroll) c_update_1pow<1, a_unroll>(co1, co2);
533  if (2 <= max_b_unroll && 2 <= b_unroll) c_update_1pow<2, a_unroll>(co1, co2);
534  if (4 <= max_b_unroll && 4 <= b_unroll) c_update_1pow<4, a_unroll>(co1, co2);
535  if (8 <= max_b_unroll && 8 <= b_unroll) c_update_1pow<8, a_unroll>(co1, co2);
536 
537  if (b_unroll == 1)
538  co1 += ldc;
539  else
540  co1 = co2 + ldc;
541  }

647  const int um_vecs = div_up(a_unroll, nelems_in_cache_line);
648 
649  if (max_b_unroll >= 1)
650  innerkernel_1pow<uk, 1, 0, um_vecs, b_unroll, ktail, fetch_x, c_fetch>(aa, ao, bo, co2, fetchA_idx, fetchB_idx);

653  if (max_b_unroll >= 4)

659  load_a<0, um_vecs, uk, a_unroll, ktail>(ao);

663  * for (int uk = 0; uk < kfactor; uk++) {
664  * int idx = 0;
665  *
666  * for (pow = 1; pow < max_b_unroll << 1; pow <<= 1) {
667  * for (int count = 0; count < (pow + 1) / 2; count++) {
668  * auto &b_reg = zmm[b_regs[idx % 2]];
669  *

671  * if (fetch_x && uk == 3 && idx == 4) aa += 8;
672  *

674  * compute<0, um_vecs, idx, uk, fetchx, ktail>(ao, bo, fetchA_idx, fetchB_idx, b_reg);
675  *
676  * const Scalar *b_addr = bo + b_unroll * uk + idx + 1 + (b_unroll > 1) - b_shift ;
677  * b_load(b_reg, b_addr);
678  * }
679  * idx++;
680  * }
681  *
682  * Maybe prefetch C data.
683  * if (pow == 2 && c_fetch) {
684  * if (uk % 3 == 0 && uk > 0) {
685  * co2 += ldc;
686  * } else {
687  * prefetch_c(co2 + (uk % 3) * nelems_in_cache_line);
688  * }
689  * }
690  * }
691  *
692  * Load A.
693  * load_a<0, um_vecs, uk, ktail, a_unroll>(ao);
694  * }
695  *
696  * Advance A/B pointers after uk-loop.
697  * ao += a_unroll * kfactor;
698  * bo += b_unroll * kfactor;

701  template <int a_unroll, int b_unroll, int k_factor, int max_b_unroll, int max_k_factor, bool c_fetch>
702  EIGEN_ALWAYS_INLINE void innerkernel(const Scalar *&aa, const Scalar *&ao, const Scalar *&bo, Scalar *&co2) {
703  int fetchA_idx = 0;
704  int fetchB_idx = 0;
705 
706  const bool fetch_x = k_factor == max_k_factor;
707  const bool ktail = k_factor == 1;
708 
709  static_assert(k_factor <= 4 && k_factor > 0, "innerkernel maximum k_factor supported is 4");
710 
711  if (k_factor > 0)
712  innerkernel_1uk<0, max_b_unroll, a_unroll, b_unroll, ktail, fetch_x, c_fetch>(aa, ao, bo, co2, fetchA_idx,
713  fetchB_idx);
714  if (k_factor > 1)
715  innerkernel_1uk<1, max_b_unroll, a_unroll, b_unroll, ktail, fetch_x, c_fetch>(aa, ao, bo, co2, fetchA_idx,
716  fetchB_idx);
717  if (k_factor > 2)
718  innerkernel_1uk<2, max_b_unroll, a_unroll, b_unroll, ktail, fetch_x, c_fetch>(aa, ao, bo, co2, fetchA_idx,
719  fetchB_idx);
720  if (k_factor > 3)
721  innerkernel_1uk<3, max_b_unroll, a_unroll, b_unroll, ktail, fetch_x, c_fetch>(aa, ao, bo, co2, fetchA_idx,
722  fetchB_idx);
723 
724  // Advance A/B pointers after uk-loop.
725  ao += a_unroll * k_factor;
726  bo += b_unroll * k_factor;
727  }

729  template <int a_unroll, int b_unroll, int max_b_unroll>
730  EIGEN_ALWAYS_INLINE void kloop(const Scalar *&aa, const Scalar *&ao, const Scalar *&bo, Scalar *&co1, Scalar *&co2) {
731  const int um_vecs = div_up(a_unroll, nelems_in_cache_line);
732  if (!use_less_a_regs)
733  a_loads<0, 2, 0, um_vecs, a_unroll>(ao);
734  else
735  a_loads<0, 1, 0, um_vecs, a_unroll>(ao);
736 
737  b_load(zmm[b_regs[0]], bo - b_shift + 0);
738  if (!use_less_b_regs) b_load(zmm[b_regs[1]], bo - b_shift + 1);
739 
740 #ifndef SECOND_FETCH
741  prefetch_cs<0, max_b_unroll, 0, um_vecs, a_unroll, b_unroll>(co1, co2);
742 #endif // SECOND_FETCH
743 
744  // Unrolling k-loop by a factor of 4.
745  const int max_k_factor = 4;
746  Index loop_count = k / max_k_factor;
747 
748  if (loop_count > 0) {
749 #ifdef SECOND_FETCH
750  loop_count -= SECOND_FETCH;
751 #endif
752  while (loop_count > 0) {
753  innerkernel<a_unroll, b_unroll, max_k_factor, max_b_unroll, max_k_factor, 0>(aa, ao, bo, co2);
754  loop_count--;
755  }
756 #ifdef SECOND_FETCH
757  co2 = co1 + nelems_in_cache_line - 1;
758 
759  loop_count += b_unroll;
760  while (loop_count > 0) {
761  innerkernel<a_unroll, b_unroll, max_k_factor, max_b_unroll, max_k_factor, 1>(aa, ao, bo, co2);
762  loop_count--;
763  }
764 
765  loop_count += SECOND_FETCH - b_unroll;
766  while (loop_count > 0) {
767  innerkernel<a_unroll, b_unroll, max_k_factor, max_b_unroll, max_k_factor, 0>(aa, ao, bo, co2);
768  loop_count--;
769  }
770 #endif
771  }
772 
773  // k-loop remainder handling.
774  loop_count = k % max_k_factor;
775  while (loop_count > 0) {
776  innerkernel<a_unroll, b_unroll, 1, max_b_unroll, max_k_factor, 0>(aa, ao, bo, co2);
777  loop_count--;
778  }
779 
780  // Update C matrix.
781  c_update<max_b_unroll, a_unroll, b_unroll>(co1, co2);
782  }
