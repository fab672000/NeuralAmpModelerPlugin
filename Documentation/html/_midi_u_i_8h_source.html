67  static void analyse(dsp* mono_dsp, bool& midi_sync, int& nvoices)
68  {
69  JSONUI jsonui;
70  mono_dsp->buildUserInterface(&jsonui);
71  std::string json = jsonui.JSON();
72  midi_sync = ((json.find("midi") != std::string::npos) &&
73  ((json.find("start") != std::string::npos) ||
74  (json.find("stop") != std::string::npos) ||
75  (json.find("clock") != std::string::npos) ||
76  (json.find("timestamp") != std::string::npos)));
77 
78  #if defined(NVOICES) && NVOICES!=NUM_VOICES
79  nvoices = NVOICES;
80  #else
81  MidiMeta meta;
82  mono_dsp->metadata(&meta);
83  bool found_voices = false;
84  // If "options" metadata is used
85  std::string options = meta.get("options", "");
86  if (options != "") {
87  std::map<std::string, std::string> metadata;
88  std::string res;
89  MetaDataUI::extractMetadata(options, res, metadata);
90  if (metadata.find("nvoices") != metadata.end()) {
91  nvoices = std::atoi(metadata["nvoices"].c_str());
92  found_voices = true;
93  }
94  }
95  // Otherwise test for "nvoices" metadata
96  if (!found_voices) {
97  std::string numVoices = meta.get("nvoices", "0");
98  nvoices = std::atoi(numVoices.c_str());
99  }
100  nvoices = std::max<int>(0, nvoices);
101  #endif
102  }

104  static bool checkPolyphony(dsp* mono_dsp)
105  {
106  MapUI map_ui;
107  mono_dsp->buildUserInterface(&map_ui);
108  bool has_freq = false;
109  bool has_gate = false;
110  bool has_gain = false;
111  for (int i = 0; i < map_ui.getParamsCount(); i++) {
112  std::string path = map_ui.getParamAddress(i);
113  has_freq |= MapUI::endsWith(path, "/freq");
114  has_gate |= MapUI::endsWith(path, "/gate");
115  has_gain |= MapUI::endsWith(path, "/gain");
116  }
117  return (has_freq && has_gate && has_gain);
118  }

587 /******************************************************************************************
588  * MidiUI : Faust User Interface
589  * This class decodes MIDI metadata and maps incoming MIDI messages to them.
590  * Currently ctrl, keyon/keyoff, keypress, pgm, chanpress, pitchwheel/pitchbend
591  * start/stop/clock meta data are handled.
592  *
593  * Maps associating MIDI event ID (like each ctrl number) with all MIDI aware UI items
594  * are defined and progressively filled when decoding MIDI related metadata.
595  * MIDI aware UI items are used in both directions:
596  * - modifying their internal state when receving MIDI input events
597  * - sending their internal state as MIDI output events
598  *******************************************************************************************/

632  void addGenericZone(FAUSTFLOAT* zone, FAUSTFLOAT min, FAUSTFLOAT max, bool input = true)
633  {
634  if (fMetaAux.size() > 0) {
635  for (size_t i = 0; i < fMetaAux.size(); i++) {
636  unsigned num;
637  unsigned chan;
638  if (fMetaAux[i].first == "midi") {
639  if (gsscanf(fMetaAux[i].second.c_str(), "ctrl %u %u", &num, &chan) == 2) {
640  fCtrlChangeTable[num].push_back(new uiMidiCtrlChange(fMidiHandler, num, this, zone, min, max, input, getScale(zone), chan));
641  } else if (gsscanf(fMetaAux[i].second.c_str(), "ctrl %u", &num) == 1) {
642  fCtrlChangeTable[num].push_back(new uiMidiCtrlChange(fMidiHandler, num, this, zone, min, max, input, getScale(zone)));
643  } else if (gsscanf(fMetaAux[i].second.c_str(), "keyon %u %u", &num, &chan) == 2) {
644  fKeyOnTable[num].push_back(new uiMidiKeyOn(fMidiHandler, num, this, zone, min, max, input, getScale(zone), chan));
645  } else if (gsscanf(fMetaAux[i].second.c_str(), "keyon %u", &num) == 1) {
646  fKeyOnTable[num].push_back(new uiMidiKeyOn(fMidiHandler, num, this, zone, min, max, input, getScale(zone)));
647  } else if (gsscanf(fMetaAux[i].second.c_str(), "keyoff %u %u", &num, &chan) == 2) {
648  fKeyOffTable[num].push_back(new uiMidiKeyOff(fMidiHandler, num, this, zone, min, max, input, getScale(zone), chan));
649  } else if (gsscanf(fMetaAux[i].second.c_str(), "keyoff %u", &num) == 1) {
650  fKeyOffTable[num].push_back(new uiMidiKeyOff(fMidiHandler, num, this, zone, min, max, input, getScale(zone)));
651  } else if (gsscanf(fMetaAux[i].second.c_str(), "key %u %u", &num, &chan) == 2) {
652  fKeyTable[num].push_back(new uiMidiKeyOn(fMidiHandler, num, this, zone, min, max, input, getScale(zone), chan));
653  } else if (gsscanf(fMetaAux[i].second.c_str(), "key %u", &num) == 1) {
654  fKeyTable[num].push_back(new uiMidiKeyOn(fMidiHandler, num, this, zone, min, max, input, getScale(zone)));
655  } else if (gsscanf(fMetaAux[i].second.c_str(), "keypress %u %u", &num, &chan) == 2) {
656  fKeyPressTable[num].push_back(new uiMidiKeyPress(fMidiHandler, num, this, zone, min, max, input, getScale(zone), chan));
657  } else if (gsscanf(fMetaAux[i].second.c_str(), "keypress %u", &num) == 1) {
658  fKeyPressTable[num].push_back(new uiMidiKeyPress(fMidiHandler, num, this, zone, min, max, input, getScale(zone)));
659  } else if (gsscanf(fMetaAux[i].second.c_str(), "pgm %u", &chan) == 1) {
660  fProgChangeTable.push_back(new uiMidiProgChange(fMidiHandler, this, zone, min, max, input, chan));
661  } else if (strcmp(fMetaAux[i].second.c_str(), "pgm") == 0) {
662  fProgChangeTable.push_back(new uiMidiProgChange(fMidiHandler, this, zone, min, max, input));
663  } else if (gsscanf(fMetaAux[i].second.c_str(), "chanpress %u %u", &num, &chan) == 2) {
