183  void Add(const AAX_CString& inAbbreviation)
184  {
185  uint32_t stringLength = inAbbreviation.Length();
186  mAbbreviations[stringLength] = inAbbreviation; //Does a string copy into the map.
187  }
188 
189  const AAX_CString& Get(int32_t inNumCharacters) const
190  {
191  //More characters than the primary string or no specific shortened names.
192  if ((inNumCharacters >= int32_t(mPrimary.Length())) || (mAbbreviations.empty()) || (0 > inNumCharacters))
193  return mPrimary;
194 
195  std::map<uint32_t, AAX_CString>::const_iterator iter = mAbbreviations.upper_bound(static_cast<uint32_t>(inNumCharacters));
196 
197  //If the iterator is already pointing to shortest string, return that.
198  if (iter == mAbbreviations.begin())
199  return iter->second;
200 
201  //lower_bound() will return the iterator that is larger than the desired value, so decrement the iterator.
202  --iter;
203  return iter->second;
204  }
205 
206  void Clear() { mAbbreviations.clear(); }
207 
208 private:
209  AAX_CString mPrimary;
210  std::map<uint32_t, AAX_CString> mAbbreviations;
