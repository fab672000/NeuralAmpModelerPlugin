70  double bench(int run)
71  {
72  // First call with fCount = -1 will be used to estimate fCount by giving the wanted measure duration
73  if (fCount == -1) {
74  measure_dsp_aux<REAL> mes(fDSP, fBufferSize, 5., fTrace, fControl);
75  mes.measure();
76  // fCount is kept from the first duration measure
77  fCount = mes.getCount();
78  return mes.getStats();
79  } else {
80  measure_dsp_aux<REAL> mes(fDSP, fBufferSize, fCount, fTrace, fControl);
81  for (int i = 0; i < run; i++) {
82  mes.measure();
83  if (fTrace) std::cout << mes.getStats() << " " << "(DSP CPU % : " << (mes.getCPULoad() * 100) << ")" << std::endl;
84  FAUSTBENCH_LOG<double>(mes.getStats());
85  }
86  return mes.getStats();
87  }

214  bool computeOne(const std::vector<std::string>& item, int run, double& res)
215  {
216  int argc = 0;
217  const char* argv[64];
218  for (int i = 0; i < item.size(); i++) {
219  argv[argc++] = item[i].c_str();
220  }
221  argv[argc] = nullptr; // NULL terminated argv
222 
223  if (fInput == "") {
224  fFactory = createDSPFactoryFromFile(fFilename.c_str(), argc, argv, fTarget, fError, fOptLevel);
225  } else {
226  fFactory = createDSPFactoryFromString("FaustDSP", fInput, argc, argv, fTarget, fError, fOptLevel);
227  }
228 
229  if (!fFactory) {
230  std::cerr << "Cannot create factory : " << fError;
231  return false;
232  }
233 
234  fDSP = fFactory->createDSPInstance();
235  if (!fDSP) {
236  std::cerr << "Cannot create instance..." << std::endl;
237  return false;
238  }
239 
240  if (fTrace) printItem(item);
241 
242  res = bench(run);
243  // fDSP is deallocated by bench calling measure_dsp
244 
245  deleteDSPFactory(fFactory);
246  fFactory = 0;
247  fDSP = 0;
248 
249  return true;
250  }

252  std::pair<double, std::vector<std::string> > findOptimizedParametersAux(const std::vector<std::vector <std::string> >& options)
253  {
254  std::vector<std::pair<int, double > > table_res;
255  double res = 0.;
256 
257  for (int i = 0; i < options.size(); i++) {
258  if (computeOne(addArgvItems(options[i], fArgc, fArgv), fRun, res)) {
259  table_res.push_back(std::make_pair(i, res));
260  } else {
261  std::cerr << "computeOne error..." << std::endl;
262  }
263  }
264 
265  sort(table_res.begin(), table_res.end(), compareFun);
266  return std::make_pair(table_res[0].second, options[table_res[0].first]);
267  }
268 
269  static bool compareFun(std::pair<int, double> i, std::pair<int, double> j) { return (i.second > j.second); }

303  std::cerr << "computeOne error..." << std::endl;
