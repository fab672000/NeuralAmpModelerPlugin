40  void prepareToPlay(int, double) override
41  {
42  const BigInteger activeInputChannels = deviceManager.getCurrentAudioDevice()->getActiveInputChannels();
43  const BigInteger activeOutputChannels = deviceManager.getCurrentAudioDevice()->getActiveOutputChannels();
44  const int maxInputChannels = activeInputChannels.getHighestBit() + 1;
45  const int maxOutputChannels = activeOutputChannels.getHighestBit() + 1;
46 
47  // Possibly adapt DSP...
48  if (fDSP->getNumInputs() > maxInputChannels || fDSP->getNumOutputs() > maxOutputChannels) {
49  fDSP = new dsp_adapter(fDSP, maxInputChannels, maxOutputChannels, 4096);
50  }
51 
52  fDSP->init(int(deviceManager.getCurrentAudioDevice()->getCurrentSampleRate()));
53  }

58  void getNextAudioBlock(const AudioSourceChannelInfo& bufferToFill) override
59  {
60  AVOIDDENORMALS;
61 
62  const float** inputs = (const float**)alloca(fDSP->getNumInputs() * sizeof(float));
63  for (int i = 0; i < fDSP->getNumInputs(); i++) {
64  inputs[i] = bufferToFill.buffer->getReadPointer(i, bufferToFill.startSample);
65  }
66 
67  float** outputs = (float**)alloca(fDSP->getNumOutputs() * sizeof(float));
68  for (int i = 0; i < fDSP->getNumOutputs(); i++) {
69  outputs[i] = bufferToFill.buffer->getWritePointer(i, bufferToFill.startSample);
70  }
71 
72  // MIDI timestamp is expressed in frames
73  fDSP->compute(-1, bufferToFill.numSamples, (float**)inputs, outputs);
74  }
