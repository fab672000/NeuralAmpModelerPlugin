336 template <typename T>
337 void AUEffectBase::ProcessBufferListsT(
338  AudioUnitRenderActionFlags & ioActionFlags,
339  const AudioBufferList & inBuffer,
340  AudioBufferList & outBuffer,
341  UInt32 inFramesToProcess )
342 {
343  bool ioSilence;
344 
345  bool silentInput = IsInputSilent (ioActionFlags, inFramesToProcess);
346  ioActionFlags |= kAudioUnitRenderAction_OutputIsSilence;
347 
348  // call the kernels to handle either interleaved or deinterleaved
349  if (inBuffer.mNumberBuffers == 1) {
350  for (UInt32 channel = 0; channel < mKernelList.size(); ++channel) {
351  AUKernelBase *kernel = mKernelList[channel];
352 
353  if (kernel == NULL) continue;
354  ioSilence = silentInput;
355 
356  // process each interleaved channel individually
357  kernel->Process(
358  (const T *)inBuffer.mBuffers[0].mData + channel,
359  (T *)outBuffer.mBuffers[0].mData + channel,
360  inFramesToProcess,
361  inBuffer.mBuffers[0].mNumberChannels,
362  ioSilence);
363 
364  if (!ioSilence)
365  ioActionFlags &= ~kAudioUnitRenderAction_OutputIsSilence;
366  }
367  } else {
368  for (UInt32 channel = 0; channel < mKernelList.size(); ++channel) {
369  AUKernelBase *kernel = mKernelList[channel];
370 
371  if (kernel == NULL) continue;
372 
373  ioSilence = silentInput;
374  const AudioBuffer *srcBuffer = &inBuffer.mBuffers[channel];
375  AudioBuffer *destBuffer = &outBuffer.mBuffers[channel];
376 
377  kernel->Process(
378  (const T *)srcBuffer->mData,
379  (T *)destBuffer->mData,
380  inFramesToProcess,
381  1,
382  ioSilence);
383 
384  if (!ioSilence)
385  ioActionFlags &= ~kAudioUnitRenderAction_OutputIsSilence;
386  }
387  }
388 }
