235  inline Scalar& coeffRef(Index row, Index col)
236  {
237  eigen_assert(row>=0 && row<rows() && col>=0 && col<cols());
238  const Index outer = IsRowMajor ? row : col;

332  inline void reserveInnerVectors(const SizesType& reserveSizes)
333  {
334  if(isCompressed())
335  {
336  Index totalReserveSize = 0;
337  for (Index j = 0; j < m_outerSize; ++j) totalReserveSize += internal::convert_index<Index>(reserveSizes[j]);
338 
339  // if reserveSizes is empty, don't do anything!
340  if (totalReserveSize == 0) return;
341 
342  // turn the matrix into non-compressed mode
343  m_innerNonZeros = internal::conditional_aligned_new_auto<StorageIndex, true>(m_outerSize);
344 
345  // temporarily use m_innerSizes to hold the new starting points.
346  StorageIndex* newOuterIndex = m_innerNonZeros;
347 
348  Index count = 0;
349  for(Index j=0; j<m_outerSize; ++j)
350  {
351  newOuterIndex[j] = internal::convert_index<StorageIndex>(count);
352  Index reserveSize = internal::convert_index<Index>(reserveSizes[j]);
353  count += reserveSize + internal::convert_index<Index>(m_outerIndex[j+1]-m_outerIndex[j]);
354  }
355 
356  m_data.reserve(totalReserveSize);
357  StorageIndex previousOuterIndex = m_outerIndex[m_outerSize];
358  for(Index j=m_outerSize-1; j>=0; --j)
359  {
360  StorageIndex innerNNZ = previousOuterIndex - m_outerIndex[j];
361  StorageIndex begin = m_outerIndex[j];
362  StorageIndex end = begin + innerNNZ;
363  StorageIndex target = newOuterIndex[j];
364  internal::smart_memmove(innerIndexPtr() + begin, innerIndexPtr() + end, innerIndexPtr() + target);
365  internal::smart_memmove(valuePtr() + begin, valuePtr() + end, valuePtr() + target);
366  previousOuterIndex = m_outerIndex[j];
367  m_outerIndex[j] = newOuterIndex[j];
368  m_innerNonZeros[j] = innerNNZ;
369  }
370  if(m_outerSize>0)
371  m_outerIndex[m_outerSize] = m_outerIndex[m_outerSize-1] + m_innerNonZeros[m_outerSize-1] + reserveSizes[m_outerSize-1];
372 
373  m_data.resize(m_outerIndex[m_outerSize]);
374  }
375  else
376  {
377  StorageIndex* newOuterIndex = internal::conditional_aligned_new_auto<StorageIndex, true>(m_outerSize + 1);
378 
379  Index count = 0;
380  for(Index j=0; j<m_outerSize; ++j)
381  {
382  newOuterIndex[j] = internal::convert_index<StorageIndex>(count);
383  Index alreadyReserved = internal::convert_index<Index>(m_outerIndex[j+1] - m_outerIndex[j] - m_innerNonZeros[j]);
384  Index reserveSize = internal::convert_index<Index>(reserveSizes[j]);
385  Index toReserve = numext::maxi(reserveSize, alreadyReserved);
386  count += toReserve + internal::convert_index<Index>(m_innerNonZeros[j]);
387  }
388  newOuterIndex[m_outerSize] = internal::convert_index<StorageIndex>(count);
389 
390  m_data.resize(count);
391  for(Index j=m_outerSize-1; j>=0; --j)
392  {
393  StorageIndex innerNNZ = m_innerNonZeros[j];
394  StorageIndex begin = m_outerIndex[j];
395  StorageIndex target = newOuterIndex[j];
396  m_data.moveChunk(begin, target, innerNNZ);
397  }
398 
399  std::swap(m_outerIndex, newOuterIndex);
400  internal::conditional_aligned_delete_auto<StorageIndex, true>(newOuterIndex, m_outerSize + 1);
401  }

509  {
510  Index start = m_outerIndex[j];

530  void makeCompressed()
531  {
532  if (isCompressed()) return;
533 
534  eigen_internal_assert(m_outerIndex!=0 && m_outerSize>0);
535 
536  StorageIndex start = m_outerIndex[1];
537  m_outerIndex[1] = m_innerNonZeros[0];
538  // try to move fewer, larger contiguous chunks
539  Index copyStart = start;
540  Index copyTarget = m_innerNonZeros[0];
541  for (Index j = 1; j < m_outerSize; j++)
542  {
543  StorageIndex end = start + m_innerNonZeros[j];
544  StorageIndex nextStart = m_outerIndex[j + 1];
545  // dont forget to move the last chunk!
546  bool breakUpCopy = (end != nextStart) || (j == m_outerSize - 1);
547  if (breakUpCopy)
548  {
549  Index chunkSize = end - copyStart;
550  if(chunkSize > 0) m_data.moveChunk(copyStart, copyTarget, chunkSize);
551  copyStart = nextStart;
552  copyTarget += chunkSize;
553  }
554  start = nextStart;
555  m_outerIndex[j + 1] = m_outerIndex[j] + m_innerNonZeros[j];
556  }
557  m_data.resize(m_outerIndex[m_outerSize]);
558 
559  // release as much memory as possible
560  internal::conditional_aligned_delete_auto<StorageIndex, true>(m_innerNonZeros, m_outerSize);
561  m_innerNonZeros = 0;
562  m_data.squeeze();
563  }

589  template<typename KeepFunc>
590  void prune(const KeepFunc& keep = KeepFunc())
591  {
592  StorageIndex k = 0;
593  for(Index j=0; j<m_outerSize; ++j)
594  {
595  StorageIndex previousStart = m_outerIndex[j];
596  if (isCompressed())
597  m_outerIndex[j] = k;
598  else
599  k = m_outerIndex[j];
600  StorageIndex end = isCompressed() ? m_outerIndex[j+1] : previousStart + m_innerNonZeros[j];
601  for(StorageIndex i=previousStart; i<end; ++i)
602  {
603  StorageIndex row = IsRowMajor ? StorageIndex(j) : m_data.index(i);
604  StorageIndex col = IsRowMajor ? m_data.index(i) : StorageIndex(j);
605  bool keepEntry = keep(row, col, m_data.value(i));
606  if (keepEntry) {
607  m_data.value(k) = m_data.value(i);
608  m_data.index(k) = m_data.index(i);
609  ++k;
610  } else if (!isCompressed())
611  m_innerNonZeros[j]--;
612  }
613  }
614  if (isCompressed()) {
615  m_outerIndex[m_outerSize] = k;
616  m_data.resize(k, 0);
617  }

628  void conservativeResize(Index rows, Index cols) {
629 
630  // If one dimension is null, then there is nothing to be preserved
631  if (rows == 0 || cols == 0) return resize(rows, cols);
632 
633  Index newOuterSize = IsRowMajor ? rows : cols;
634  Index newInnerSize = IsRowMajor ? cols : rows;
635 
636  Index innerChange = newInnerSize - m_innerSize;
637  Index outerChange = newOuterSize - m_outerSize;
638 
639  if (outerChange != 0) {
640  m_outerIndex = internal::conditional_aligned_realloc_new_auto<StorageIndex, true>(
641  m_outerIndex, newOuterSize + 1, m_outerSize + 1);
642 
643  if (!isCompressed())
644  m_innerNonZeros = internal::conditional_aligned_realloc_new_auto<StorageIndex, true>(
645  m_innerNonZeros, newOuterSize, m_outerSize);
646 
647  if (outerChange > 0) {
648  StorageIndex lastIdx = m_outerSize == 0 ? StorageIndex(0) : m_outerIndex[m_outerSize];
649  std::fill_n(m_outerIndex + m_outerSize, outerChange + 1, lastIdx);
650 
651  if (!isCompressed()) std::fill_n(m_innerNonZeros + m_outerSize, outerChange, StorageIndex(0));
652  }
653  }
654  m_outerSize = newOuterSize;
655 
656  if (innerChange < 0) {
657  for (Index j = 0; j < m_outerSize; j++) {
658  Index start = m_outerIndex[j];
659  Index end = isCompressed() ? m_outerIndex[j + 1] : start + m_innerNonZeros[j];
660  Index lb = m_data.searchLowerIndex(start, end, newInnerSize);
661  if (lb != end) {
662  uncompress();
663  m_innerNonZeros[j] = StorageIndex(lb - start);
664  }
665  }
666  }
667  m_innerSize = newInnerSize;
668 
669  Index newSize = m_outerIndex[m_outerSize];
670  eigen_assert(newSize <= m_data.size());
671  m_data.resize(newSize);
672  }

979  void assignDiagonal(const DiagXpr diagXpr, const Func& assignFunc)
980  {
981 
982  constexpr StorageIndex kEmptyIndexVal(-1);
983  typedef typename ScalarVector::AlignedMapType ValueMap;
984 
985  Index n = diagXpr.size();
986 
987  const bool overwrite = internal::is_same<Func, internal::assign_op<Scalar,Scalar> >::value;
988  if(overwrite)
989  {
990  if((m_outerSize != n) || (m_innerSize != n))
991  resize(n, n);
992  }
993 
994  if(m_data.size()==0 || overwrite)
995  {
996  internal::conditional_aligned_delete_auto<StorageIndex, true>(m_innerNonZeros, m_outerSize);
997  m_innerNonZeros = 0;
998  resizeNonZeros(n);
999  ValueMap valueMap(valuePtr(), n);
1000  std::iota(m_outerIndex, m_outerIndex + n + 1, StorageIndex(0));
1001  std::iota(innerIndexPtr(), innerIndexPtr() + n, StorageIndex(0));
1002  valueMap.setZero();
1003  internal::call_assignment_no_alias(valueMap, diagXpr, assignFunc);
1004  }
1005  else
1006  {
1007  internal::evaluator<DiagXpr> diaEval(diagXpr);
1008 
1009  ei_declare_aligned_stack_constructed_variable(StorageIndex, tmp, n, 0);
1010  typename IndexVector::AlignedMapType insertionLocations(tmp, n);
1011  insertionLocations.setConstant(kEmptyIndexVal);
1012 
1013  Index deferredInsertions = 0;
1014  Index shift = 0;
1015 
1016  for (Index j = 0; j < n; j++) {
1017  Index begin = m_outerIndex[j];
1018  Index end = isCompressed() ? m_outerIndex[j + 1] : begin + m_innerNonZeros[j];
1019  Index capacity = m_outerIndex[j + 1] - end;
1020  Index dst = m_data.searchLowerIndex(begin, end, j);
1021  // the entry exists: update it now
1022  if (dst != end && m_data.index(dst) == StorageIndex(j)) assignFunc.assignCoeff(m_data.value(dst), diaEval.coeff(j));
1023  // the entry belongs at the back of the vector: push to back
1024  else if (dst == end && capacity > 0)
1025  assignFunc.assignCoeff(insertBackUncompressed(j, j), diaEval.coeff(j));
1026  // the insertion requires a data move, record insertion location and handle in second pass
1027  else {
1028  insertionLocations.coeffRef(j) = StorageIndex(dst);
1029  deferredInsertions++;
1030  // if there is no capacity, all vectors to the right of this are shifted
1031  if (capacity == 0) shift++;
1032  }
1033  }
1034 
1035  if (deferredInsertions > 0) {
1036 
1037  m_data.resize(m_data.size() + shift);
1038  Index copyEnd = isCompressed() ? m_outerIndex[m_outerSize]
1039  : m_outerIndex[m_outerSize - 1] + m_innerNonZeros[m_outerSize - 1];
1040  for (Index j = m_outerSize - 1; deferredInsertions > 0; j--) {
1041  Index begin = m_outerIndex[j];
1042  Index end = isCompressed() ? m_outerIndex[j + 1] : begin + m_innerNonZeros[j];
1043  Index capacity = m_outerIndex[j + 1] - end;
1044 
1045  bool doInsertion = insertionLocations(j) >= 0;
1046  bool breakUpCopy = doInsertion && (capacity > 0);
1047  // break up copy for sorted insertion into inactive nonzeros
1048  // optionally, add another criterium, i.e. 'breakUpCopy || (capacity > threhsold)'
1049  // where `threshold >= 0` to skip inactive nonzeros in each vector
1050  // this reduces the total number of copied elements, but requires more moveChunk calls
1051  if (breakUpCopy) {
1052  Index copyBegin = m_outerIndex[j + 1];
1053  Index to = copyBegin + shift;
1054  Index chunkSize = copyEnd - copyBegin;
1055  m_data.moveChunk(copyBegin, to, chunkSize);
1056  copyEnd = end;
1057  }
1058 
1059  m_outerIndex[j + 1] += shift;
1060 
1061  if (doInsertion) {
1062  // if there is capacity, shift into the inactive nonzeros
1063  if (capacity > 0) shift++;
1064  Index copyBegin = insertionLocations(j);
1065  Index to = copyBegin + shift;
1066  Index chunkSize = copyEnd - copyBegin;
1067  m_data.moveChunk(copyBegin, to, chunkSize);
1068  Index dst = to - 1;
1069  m_data.index(dst) = StorageIndex(j);
1070  m_data.value(dst) = Scalar(0);
1071  assignFunc.assignCoeff(m_data.value(dst), diaEval.coeff(j));
1072  if (!isCompressed()) m_innerNonZeros[j]++;
1073  shift--;
1074  deferredInsertions--;
1075  copyEnd = copyBegin;
1076  }
1077  }
1078  }
1079  eigen_assert((shift == 0) && (deferredInsertions == 0));
1080  }

1105 // Creates a compressed sparse matrix from a range of unsorted triplets
1106 // Requires temporary storage to handle duplicate entries
1107 template <typename InputIterator, typename SparseMatrixType, typename DupFunctor>
1108 void set_from_triplets(const InputIterator& begin, const InputIterator& end, SparseMatrixType& mat,
1109  DupFunctor dup_func) {
1110  constexpr bool IsRowMajor = SparseMatrixType::IsRowMajor;
1111  using StorageIndex = typename SparseMatrixType::StorageIndex;
1112  using IndexMap = typename VectorX<StorageIndex>::AlignedMapType;
1113  using TransposedSparseMatrix = SparseMatrix<typename SparseMatrixType::Scalar, IsRowMajor ? ColMajor : RowMajor, StorageIndex>;
1114 
1115  if (begin == end) return;
1116 
1117  // There are two strategies to consider for constructing a matrix from unordered triplets:
1118  // A) construct the 'mat' in its native storage order and sort in-place (less memory); or,
1119  // B) construct the transposed matrix and use an implicit sort upon assignment to `mat` (less time).
1120  // This routine uses B) for faster execution time.
1121  TransposedSparseMatrix trmat(mat.rows(), mat.cols());
1122 
1123  // scan triplets to determine allocation size before constructing matrix
1124  Index nonZeros = 0;
1125  for (InputIterator it(begin); it != end; ++it) {
1126  eigen_assert(it->row() >= 0 && it->row() < mat.rows() && it->col() >= 0 && it->col() < mat.cols());
1127  StorageIndex j = convert_index<StorageIndex>(IsRowMajor ? it->col() : it->row());
1128  if (nonZeros == NumTraits<StorageIndex>::highest()) internal::throw_std_bad_alloc();
1129  trmat.outerIndexPtr()[j + 1]++;
1130  nonZeros++;
1131  }
1132 
1133  std::partial_sum(trmat.outerIndexPtr(), trmat.outerIndexPtr() + trmat.outerSize() + 1, trmat.outerIndexPtr());
1134  eigen_assert(nonZeros == trmat.outerIndexPtr()[trmat.outerSize()]);
1135  trmat.resizeNonZeros(nonZeros);
1136 
1137  // construct temporary array to track insertions (outersize) and collapse duplicates (innersize)
1138  ei_declare_aligned_stack_constructed_variable(StorageIndex, tmp, numext::maxi(mat.innerSize(), mat.outerSize()), 0);
1139  smart_copy(trmat.outerIndexPtr(), trmat.outerIndexPtr() + trmat.outerSize(), tmp);
1140 
1141  // push triplets to back of each vector
1142  for (InputIterator it(begin); it != end; ++it) {
1143  StorageIndex j = convert_index<StorageIndex>(IsRowMajor ? it->col() : it->row());
1144  StorageIndex i = convert_index<StorageIndex>(IsRowMajor ? it->row() : it->col());
1145  StorageIndex k = tmp[j];
1146  trmat.data().index(k) = i;
1147  trmat.data().value(k) = it->value();
1148  tmp[j]++;
1149  }
1150 
1151  IndexMap wi(tmp, trmat.innerSize());
1152  trmat.collapseDuplicates(wi, dup_func);
1153  // implicit sorting
1154  mat = trmat;
1155 }

1157 // Creates a compressed sparse matrix from a sorted range of triplets
1158 template <typename InputIterator, typename SparseMatrixType, typename DupFunctor>
1159 void set_from_triplets_sorted(const InputIterator& begin, const InputIterator& end, SparseMatrixType& mat,
1160  DupFunctor dup_func) {
1161  constexpr bool IsRowMajor = SparseMatrixType::IsRowMajor;
1162  using StorageIndex = typename SparseMatrixType::StorageIndex;
1163 
1164  if (begin == end) return;
1165 
1166  constexpr StorageIndex kEmptyIndexValue(-1);
1167  // deallocate inner nonzeros if present and zero outerIndexPtr
1168  mat.resize(mat.rows(), mat.cols());
1169  // use outer indices to count non zero entries (excluding duplicate entries)
1170  StorageIndex previous_j = kEmptyIndexValue;
1171  StorageIndex previous_i = kEmptyIndexValue;
1172  // scan triplets to determine allocation size before constructing matrix
1173  Index nonZeros = 0;
1174  for (InputIterator it(begin); it != end; ++it) {
1175  eigen_assert(it->row() >= 0 && it->row() < mat.rows() && it->col() >= 0 && it->col() < mat.cols());
1176  StorageIndex j = convert_index<StorageIndex>(IsRowMajor ? it->row() : it->col());
1177  StorageIndex i = convert_index<StorageIndex>(IsRowMajor ? it->col() : it->row());
1178  eigen_assert(j > previous_j || (j == previous_j && i >= previous_i));
1179  // identify duplicates by examining previous location
1180  bool duplicate = (previous_j == j) && (previous_i == i);
1181  if (!duplicate) {
1182  if (nonZeros == NumTraits<StorageIndex>::highest()) internal::throw_std_bad_alloc();
1183  nonZeros++;
1184  mat.outerIndexPtr()[j + 1]++;
1185  previous_j = j;
1186  previous_i = i;
1187  }
1188  }
1189 
1190  // finalize outer indices and allocate memory
1191  std::partial_sum(mat.outerIndexPtr(), mat.outerIndexPtr() + mat.outerSize() + 1, mat.outerIndexPtr());
1192  eigen_assert(nonZeros == mat.outerIndexPtr()[mat.outerSize()]);
1193  mat.resizeNonZeros(nonZeros);
1194 
1195  previous_i = kEmptyIndexValue;
1196  previous_j = kEmptyIndexValue;
1197  Index back = 0;
1198  for (InputIterator it(begin); it != end; ++it) {
1199  StorageIndex j = convert_index<StorageIndex>(IsRowMajor ? it->row() : it->col());
1200  StorageIndex i = convert_index<StorageIndex>(IsRowMajor ? it->col() : it->row());
1201  bool duplicate = (previous_j == j) && (previous_i == i);
1202  if (duplicate) {
1203  mat.data().value(back - 1) = dup_func(mat.data().value(back - 1), it->value());
1204  } else {
1205  // push triplets to back
1206  mat.data().index(back) = i;
1207  mat.data().value(back) = it->value();
1208  previous_j = j;
1209  previous_i = i;
1210  back++;
1211  }
1212  }
1213  eigen_assert(back == nonZeros);
1214  // matrix is finalized
1215 }
