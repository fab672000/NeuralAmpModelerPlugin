123  void CopyBufferContentsTo(AudioBufferList &abl) const {

125  COMPONENT_THROW(-1);

182 // Allocates an array of samples (type T), to be optimally aligned for the processor
184 template <class T>
185 class TAUBuffer {
186 public:
187  enum {
188  kAlignInterval = 0x10,
189  kAlignMask = kAlignInterval - 1
190  };
191 
193  TAUBuffer() : mMemObject(NULL), mAlignedBuffer(NULL), mBufferSizeBytes(0)
194  {
195  }
196 
198  TAUBuffer(UInt32 numElems, UInt32 numChannels) : mMemObject(NULL), mAlignedBuffer(NULL),
199  mBufferSizeBytes(0)
200  {
201  Allocate(numElems, numChannels);
202  }
203 
205  ~TAUBuffer()
206  {
207  Deallocate();
208  }
209 
211  void Allocate(UInt32 numElems) // can also re-allocate
212  {
213  UInt32 reqSize = numElems * sizeof(T);
214 
215  if (mMemObject != NULL && reqSize == mBufferSizeBytes)
216  return; // already allocated
217 
218  mBufferSizeBytes = reqSize;
219  mMemObject = CA_realloc(mMemObject, reqSize);
220  UInt32 misalign = (uintptr_t)mMemObject & kAlignMask;
221  if (misalign) {
222  mMemObject = CA_realloc(mMemObject, reqSize + kAlignMask);
223  mAlignedBuffer = (T *)((char *)mMemObject + kAlignInterval - misalign);
224  } else
225  mAlignedBuffer = (T *)mMemObject;
226  }
227 
229  void Deallocate()
230  {
231  if (mMemObject == NULL) return; // so this method has no effect if we're using
232  // an external buffer
233 
234  free(mMemObject);
235  mMemObject = NULL;
236  mAlignedBuffer = NULL;
237  mBufferSizeBytes = 0;
238  }
239 
241  void AllocateClear(UInt32 numElems) // can also re-allocate
242  {
243  Allocate(numElems);
244  Clear();
245  }
246 
248  void Clear()
249  {
250  memset(mAlignedBuffer, 0, mBufferSizeBytes);
251  }
252 
253  // accessors
254 
256  operator T *() { return mAlignedBuffer; }
257 
258 private:
260  void * mMemObject; // null when using an external buffer
262  T * mAlignedBuffer; // always valid once allocated
264  UInt32 mBufferSizeBytes;
265 };
