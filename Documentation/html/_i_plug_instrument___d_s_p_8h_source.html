53  void ProcessSamplesAccumulating(T** inputs, T** outputs, int nInputs, int nOutputs, int startIdx, int nFrames) override
54  {
55  // inputs to the synthesizer can just fetch a value every block, like this:
56 // double gate = mInputs[kVoiceControlGate].endValue;
57  double pitch = mInputs[kVoiceControlPitch].endValue;
58  double pitchBend = mInputs[kVoiceControlPitchBend].endValue;
59 
60  // or write the entire control ramp to a buffer, like this, to get sample-accurate ramps:
61  mInputs[kVoiceControlTimbre].Write(mTimbreBuffer.Get(), startIdx, nFrames);
62 
63  // convert from "1v/oct" pitch space to frequency in Hertz
64  double osc1Freq = 440. * pow(2., pitch + pitchBend + inputs[kModLFO][0]);
65 
66  // make sound output for each output channel
67  for(auto i = startIdx; i < startIdx + nFrames; i++)
68  {
69  float noise = mTimbreBuffer.Get()[i] * Rand();
70  // an MPE synth can use pressure here in addition to gain
71  outputs[0][i] += (mOSC.Process(osc1Freq) + noise) * mAMPEnv.Process(inputs[kModSustainSmoother][i]) * mGain;
72  outputs[1][i] = outputs[0][i];
73  }
74  }
