366 // Apply binary operations byte-by-byte
367 template<typename T>
368 struct bytewise_bitwise_helper {
369  EIGEN_DEVICE_FUNC static inline T bitwise_and(const T& a, const T& b) {
370  return binary(a, b, bit_and<unsigned char>());
371  }
372  EIGEN_DEVICE_FUNC static inline T bitwise_or(const T& a, const T& b) {
373  return binary(a, b, bit_or<unsigned char>());
374  }
375  EIGEN_DEVICE_FUNC static inline T bitwise_xor(const T& a, const T& b) {
376  return binary(a, b, bit_xor<unsigned char>());
377  }
378  EIGEN_DEVICE_FUNC static inline T bitwise_not(const T& a) {
379  return unary(a,bit_not<unsigned char>());
380  }
381 
382  private:
383  template<typename Op>
384  EIGEN_DEVICE_FUNC static inline T unary(const T& a, Op op) {
385  const unsigned char* a_ptr = reinterpret_cast<const unsigned char*>(&a);
386  T c;
387  unsigned char* c_ptr = reinterpret_cast<unsigned char*>(&c);
388  for (size_t i = 0; i < sizeof(T); ++i) {
389  *c_ptr++ = op(*a_ptr++);
390  }
391  return c;
392  }
393 
394  template<typename Op>
395  EIGEN_DEVICE_FUNC static inline T binary(const T& a, const T& b, Op op) {
396  const unsigned char* a_ptr = reinterpret_cast<const unsigned char*>(&a);
397  const unsigned char* b_ptr = reinterpret_cast<const unsigned char*>(&b);
398  T c;
399  unsigned char* c_ptr = reinterpret_cast<unsigned char*>(&c);
400  for (size_t i = 0; i < sizeof(T); ++i) {
401  *c_ptr++ = op(*a_ptr++, *b_ptr++);
402  }
403  return c;
404  }
405 };

624 template<typename Packet> EIGEN_DEVICE_FUNC inline Packet
625 pload_partial(const typename unpacket_traits<Packet>::type* from, const Index n, const Index offset = 0)
626 {
627  const Index packet_size = unpacket_traits<Packet>::size;
628  eigen_assert(n + offset <= packet_size && "number of elements plus offset will read past end of packet");
629  typedef typename unpacket_traits<Packet>::type Scalar;
630  EIGEN_ALIGN_MAX Scalar elements[packet_size] = { Scalar(0) };
631  for (Index i = offset; i < numext::mini(n+offset,packet_size); i++) {
632  elements[i] = from[i-offset];
633  }
634  return pload<Packet>(elements);
635 }

1242 EIGEN_DEVICE_FUNC EIGEN_ALWAYS_INLINE Packet patan2(const Packet& y, const Packet& x) {
1243  typedef typename internal::unpacket_traits<Packet>::type Scalar;
1244 
1245  // See https://en.cppreference.com/w/cpp/numeric/math/atan2
1246  // for how corner cases are supposed to be handled according to the
1247  // IEEE floating-point standard (IEC 60559).
1248  const Packet kSignMask = pset1<Packet>(-Scalar(0));
1249  const Packet kZero = pzero(x);
1250  const Packet kOne = pset1<Packet>(Scalar(1));
1251  const Packet kPi = pset1<Packet>(Scalar(EIGEN_PI));
1252 
1253  const Packet x_has_signbit = psignbit(x);
1254  const Packet y_signmask = pand(y, kSignMask);
1255  const Packet x_signmask = pand(x, kSignMask);
1256  const Packet result_signmask = pxor(y_signmask, x_signmask);
1257  const Packet shift = por(pand(x_has_signbit, kPi), y_signmask);
1258 
1259  const Packet x_and_y_are_same = pcmp_eq(pabs(x), pabs(y));
1260  const Packet x_and_y_are_zero = pcmp_eq(por(x, y), kZero);
1261 
1262  Packet arg = pdiv(y, x);
1263  arg = pselect(x_and_y_are_same, por(kOne, result_signmask), arg);
1264  arg = pselect(x_and_y_are_zero, result_signmask, arg);
1265 
1266  Packet result = patan(arg);
1267  result = padd(result, shift);
1268  return result;
1269 }
