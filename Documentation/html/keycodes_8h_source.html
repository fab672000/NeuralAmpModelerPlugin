125 inline SMTG_CONSTEXPR14 tchar VirtualKeyCodeToChar (uint8 vKey)
126 {
127  if (vKey >= VKEY_FIRST_ASCII)
128  return (tchar)(vKey - VKEY_FIRST_ASCII + 0x30);
129  else if (vKey == KEY_SPACE)
130  return ' ';
131  return 0;
132 }

134 //------------------------------------------------------------------------------
135 inline SMTG_CONSTEXPR14 uint8 CharToVirtualKeyCode (tchar character)
136 {
137  if ((character >= 0x30 && character <= 0x39) || (character >= 0x41 && character <= 0x5A))
138  return (uint8)(character - 0x30 + VKEY_FIRST_ASCII);
139  if (character == ' ')
140  return (uint8)KEY_SPACE;
141  return 0;
142 }
