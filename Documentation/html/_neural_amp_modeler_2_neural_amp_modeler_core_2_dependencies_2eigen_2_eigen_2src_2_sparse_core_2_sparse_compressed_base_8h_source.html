174  internal::LowerBoundIndex lower_bound(Index row, Index col) const
175  {
176  eigen_internal_assert(row>=0 && row<this->rows() && col>=0 && col<this->cols());
177 
178  const Index outer = Derived::IsRowMajor ? row : col;
179  const Index inner = Derived::IsRowMajor ? col : row;
180 
181  Index start = this->outerIndexPtr()[outer];
182  Index end = this->isCompressed() ? this->outerIndexPtr()[outer+1] : this->outerIndexPtr()[outer] + this->innerNonZeroPtr()[outer];
183  eigen_assert(end>=start && "you are using a non finalized sparse matrix or written coefficient does not exist");
184  internal::LowerBoundIndex p;
185  p.value = std::lower_bound(this->innerIndexPtr()+start, this->innerIndexPtr()+end,inner) - this->innerIndexPtr();
186  p.found = (p.value<end) && (this->innerIndexPtr()[p.value]==inner);
187  return p;

468 template <typename Derived, class Comp, bool IsVector>
469 struct inner_sort_impl {
470  typedef typename Derived::Scalar Scalar;
471  typedef typename Derived::StorageIndex StorageIndex;
472  static inline void run(SparseCompressedBase<Derived>& obj, Index begin, Index end) {
473  const bool is_compressed = obj.isCompressed();
474  for (Index outer = begin; outer < end; outer++) {
475  Index begin_offset = obj.outerIndexPtr()[outer];
476  Index end_offset = is_compressed ? obj.outerIndexPtr()[outer + 1] : (begin_offset + obj.innerNonZeroPtr()[outer]);
477  CompressedStorageIterator<Scalar, StorageIndex> begin_it(begin_offset, obj.innerIndexPtr(), obj.valuePtr());
478  CompressedStorageIterator<Scalar, StorageIndex> end_it(end_offset, obj.innerIndexPtr(), obj.valuePtr());
479  std::sort(begin_it, end_it, Comp());
480  }
481  }
482  static inline Index check(const SparseCompressedBase<Derived>& obj, Index begin, Index end) {
483  const bool is_compressed = obj.isCompressed();
484  for (Index outer = begin; outer < end; outer++) {
485  Index begin_offset = obj.outerIndexPtr()[outer];
486  Index end_offset = is_compressed ? obj.outerIndexPtr()[outer + 1] : (begin_offset + obj.innerNonZeroPtr()[outer]);
487  const StorageIndex* begin_it = obj.innerIndexPtr() + begin_offset;
488  const StorageIndex* end_it = obj.innerIndexPtr() + end_offset;
489  bool is_sorted = std::is_sorted(begin_it, end_it, Comp());
490  if (!is_sorted) return outer;
491  }
492  return end;
493  }
494 };
495 template <typename Derived, class Comp>
496 struct inner_sort_impl<Derived, Comp, true> {
497  typedef typename Derived::Scalar Scalar;
498  typedef typename Derived::StorageIndex StorageIndex;
499  static inline void run(SparseCompressedBase<Derived>& obj, Index, Index) {
500  Index begin_offset = 0;
501  Index end_offset = obj.nonZeros();
502  CompressedStorageIterator<Scalar, StorageIndex> begin_it(begin_offset, obj.innerIndexPtr(), obj.valuePtr());
503  CompressedStorageIterator<Scalar, StorageIndex> end_it(end_offset, obj.innerIndexPtr(), obj.valuePtr());
504  std::sort(begin_it, end_it, Comp());
505  }
506  static inline Index check(const SparseCompressedBase<Derived>& obj, Index, Index) {
507  Index begin_offset = 0;
508  Index end_offset = obj.nonZeros();
509  const StorageIndex* begin_it = obj.innerIndexPtr() + begin_offset;
510  const StorageIndex* end_it = obj.innerIndexPtr() + end_offset;
511  return std::is_sorted(begin_it, end_it, Comp()) ? 1 : 0;
512  }
