160  template <typename T>
161  void iterChildrenAfterCloningIfNeeded(T callback, void* cloneContext) {
162  int i = 0;
163  for (YGNodeRef& child : children_) {
164  if (child->getOwner() != this) {
165  child = config_->cloneNode(child, this, i, cloneContext);
166  child->setOwner(this);
167  }
168  i += 1;
169 
170  callback(child, cloneContext);
171  }
172  }
