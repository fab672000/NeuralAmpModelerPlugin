137 static bool parseDouble(const char*& p, double& x)
138 {
139  std::stringstream reader(p);
140  std::streambuf* pbuf = reader.rdbuf();
141 
142  // Keep position before parsing
143  std::streamsize size1 = pbuf->in_avail();
144 
145  // Parse the number
146  reader >> x;
147 
148  // Keep position after parsing
149  std::streamsize size2 = pbuf->in_avail();
150 
151  // Move from the actual size
152  p += (size1 - size2);
153 
154  // True if the number contains at least one digit
155  return (size1 > size2);
156 }

165 static bool parseString(const char*& p, char quote, std::string& s)
166 {
167  std::string str;
168  skipBlank(p);
169 
170  const char* saved = p; // to restore position if we fail
171  if (*p++ == quote) {
172  while ((*p != 0) && (*p != quote)) {
173  str += *p++;
174  }
175  if (*p++ == quote) {
176  s = str;
177  return true;
178  }
179  }
180  p = saved;
181  return false;
182 }

367 {

386 static bool parseUI(const char*& p, std::vector<itemInfo>& uiItems, int& numItems)
387 {

390

397  if (label == "type") {
398  if (uiItems.size() != 0) {
399  numItems++;
400  }
401  if (parseChar(p, ':') && parseDQString(p, value)) {
402  itemInfo item;
403  item.type = value;
404  uiItems.push_back(item);
405  }
406  }
407 
408  else if (label == "label") {
409  if (parseChar(p, ':') && parseDQString(p, value)) {
410  uiItems[numItems].label = value;
411  }
412  }
413 
414  else if (label == "url") {
415  if (parseChar(p, ':') && parseDQString(p, value)) {

422  uiItems[numItems].address = value;
423  }
424  }
425

435  }
436  }

491 // ---------------------------------------------------------------------
492 // Parse full JSON record describing a JSON/Faust interface :
493 // {"metadatas": "...", "ui": [{ "type": "...", "label": "...", "items": [...], "address": "...","init": "...", "min": "...", "max": "...","step": "..."}]}
494 //
495 // and store the result in map Metadatas and vector containing the items of the interface. Returns true if parsing was successfull.
497 static bool parseJson(const char*& p,
498  std::map<std::string, std::pair<std::string, double> >& metaDatas0,
499  std::map<std::string, std::string>& metaDatas1,
500  std::map<std::string, std::vector<std::string> >& metaDatas2,
501  std::vector<itemInfo>& uiItems)
502 {
503  parseChar(p, '{');
504 
505  do {
506  std::string key;
507  std::string value;
508  double dbl = 0;
509  std::vector<std::string> items;
510  if (parseGlobalMetaData(p, key, value, dbl, metaDatas1, items)) {
511  if (key != "meta") {
512  // keep "name", "inputs", "outputs" key/value pairs
513  if (items.size() > 0) {
514  metaDatas2[key] = items;
515  items.clear();
516  } else if (value != "") {
517  metaDatas0[key].first = value;
518  } else {
519  metaDatas0[key].second = dbl;
520  }
521  }
522  } else if (key == "ui") {
523  int numItems = 0;
524  parseChar(p, '[') && parseUI(p, uiItems, numItems);
525  }
526  } while (tryChar(p, ','));
527 
528  return parseChar(p, '}');
529 }
