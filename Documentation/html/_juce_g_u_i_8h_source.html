273  void drawRotarySlider (Graphics& g, int x, int y, int width, int height, float sliderPos,
274  float rotaryStartAngle, float rotaryEndAngle, Slider& slider) override
275  {
276  const float radius = jmin(width / 2, height / 2) - 4.0f;
277  const float centreX = x + width * 0.5f;
278  const float centreY = y + height * 0.5f;
279  const float rx = centreX - radius;
280  const float ry = centreY - radius;
281  const float rw = radius * 2.0f;
282  const float angle = rotaryStartAngle + sliderPos * (rotaryEndAngle - rotaryStartAngle);
283  const bool isMouseOver = slider.isMouseOverOrDragging() && slider.isEnabled();
284 
285  //Background
286  {
287  g.setColour(Colours::lightgrey.withAlpha (isMouseOver ? 1.0f : 0.7f));
288  Path intFilledArc;
289  intFilledArc.addPieSegment(rx, ry, rw, rw, rotaryStartAngle, rotaryEndAngle, 0.8);
290  g.fillPath(intFilledArc);
291  }
292 
293  if (slider.isEnabled()) {
294  g.setColour(slider.findColour (Slider::rotarySliderFillColourId).withAlpha (isMouseOver ? 1.0f : 0.7f));
295  } else {
296  g.setColour(Colour (0x80808080));
297  }
298 
299  //Render knob value
300  {
301  Path pathArc;
302  pathArc.addPieSegment(rx, ry, rw, rw, rotaryStartAngle, angle, 0.8);
303  g.fillPath(pathArc);
304 
305  Path cursor, cursorShadow;
306  float rectWidth = radius*0.4;
307  float rectHeight = rectWidth/2;
308  float rectX = centreX + radius*0.9 - rectHeight/2;
309  float rectY = centreY - rectWidth/2;
310 
311  cursor.addRectangle(rectX, rectY, rectWidth, rectHeight);
312  cursorShadow.addRectangle(rectX-1, rectY-1, rectWidth+2, rectHeight+2);
313 
314  AffineTransform t = AffineTransform::translation(-rectWidth + 2, rectHeight/2);
315  t = t.rotated((angle - float_Pi/2), centreX, centreY);
316 
317  cursor.applyTransform(t);
318  cursorShadow.applyTransform(t);
319 
320  g.setColour(Colours::black);
321  g.fillPath(cursor);
322 
323  g.setColour(Colours::black .withAlpha(0.15f));
324  g.fillPath(cursorShadow);
325  }
