87 // Exponential function. Works by writing "x = m*log(2) + r" where
88 // "m = floor(x/log(2)+1/2)" and "r" is the remainder. The result is then
89 // "exp(x) = 2^m*exp(r)" where exp(r) is in the range [-1,1).
90 template <>
91 EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS Packet8f
92 pexp<Packet8f>(const Packet8f& _x) {
93  return pexp_float(_x);
94 }

111 // Notice that for newer processors, it is counterproductive to use Newton
112 // iteration for square root. In particular, Skylake and Zen2 processors
113 // have approximately doubled throughput of the _mm_sqrt_ps instruction
114 // compared to their predecessors.
115 template <> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
116 Packet8f psqrt<Packet8f>(const Packet8f& _x) {
117  return _mm256_sqrt_ps(_x);
118 }

120 Packet4d psqrt<Packet4d>(const Packet4d& _x) {
121  return _mm256_sqrt_pd(_x);

125 // Even on Skylake, using Newton iteration is a win for reciprocal square root.
126 #if EIGEN_FAST_MATH
127 template<> EIGEN_DEFINE_FUNCTION_ALLOWING_MULTIPLE_DEFINITIONS
128 Packet8f prsqrt<Packet8f>(const Packet8f& a) {
129  // _mm256_rsqrt_ps returns -inf for negative denormals.
130  // _mm512_rsqrt**_ps returns -NaN for negative denormals. We may want
131  // consistency here.
132  // const Packet8f rsqrt = pselect(pcmp_lt(a, pzero(a)),
133  // pset1<Packet8f>(-NumTraits<float>::quiet_NaN()),
134  // _mm256_rsqrt_ps(a));
135  return generic_rsqrt_newton_step<Packet8f, /*Steps=*/1>::run(a, _mm256_rsqrt_ps(a));
136 }
