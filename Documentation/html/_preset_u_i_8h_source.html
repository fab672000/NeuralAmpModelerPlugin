35 /*
36  Decorates an UI to add preset management:
37  - a 'preset' num entry allows to select a given preset
38  - a 'load' button restores the state of the currently selected preset
39  - a 'save' button saves the state of the currently selected preset
40  - a 'reset' button restores te UI default state
41  Presets are saved in separated text files using the FUI model.
42 */
43

54  new uiCallbackItem(this, &presetui->fSave, PresetUI::save, presetui);
55  new uiCallbackItem(this, &presetui->fReset, PresetUI::reset, presetui);
56  }
57  };

89  void checkOpenFirstBox(const char* label)
90  {
91  if (fGroupCount++ == 0) {
92  // Start of top-level group
93  fUI->openHorizontalBox("Preset manager");
94  fUI->addButton("Save", &fSave);
95  fUI->addNumEntry("Preset", &fPreset, FAUSTFLOAT(0),FAUSTFLOAT(0), FAUSTFLOAT(100), FAUSTFLOAT(1));
96  fUI->addButton("Load", &fLoad);
97  fUI->addButton("Reset", &fReset);
98  fUI->closeBox();
99  }
100  }
