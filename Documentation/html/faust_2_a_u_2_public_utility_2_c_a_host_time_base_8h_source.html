119 inline UInt64 CAHostTimeBase::GetTheCurrentTime()
120 {

124  theTime = mach_absolute_time();
125  #elif TARGET_OS_WIN32
126  LARGE_INTEGER theValue;
127  QueryPerformanceCounter(&theValue);
128  theTime = *((UInt64*)&theValue);

131  #if Track_Host_TimeBase
132  if(sLastTime != 0)

136  DebugMessageN2("CAHostTimeBase::GetTheCurrentTime: the current time is earlier than the last time, now: %qd, then: %qd", theTime, sLastTime);

142  sLastTime = theTime;
