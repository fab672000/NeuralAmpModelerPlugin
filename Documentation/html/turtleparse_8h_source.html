33 class wdl_turtle_pair {
34  enum { MODE_RESOURCE=0, MODE_LIST, MODE_COLLECTION, MODE_BLANK, MODE_LITERAL_TYPELESS };
35 
36  char *verb, *value; // verb can be NULL to indicate "a"
37  UINT_PTR mode; // MODE_* or pointer to string which is @lang or type string (minus ^^)
38 public:
39  wdl_turtle_pair(const char *v, const char *_res, const char *_lit, const char *_lit_type, int _list) // _list=2 for collection
40  {
41  WDL_ASSERT((!!_res) + (!!_lit) + (!!_list) <= 1);
42  verb = v ? strdup(v) : NULL;
43  if (_lit)
44  {
45  value = strdup(_lit);
46  mode = _lit_type && *_lit_type ? (UINT_PTR) strdup(_lit_type) : MODE_LITERAL_TYPELESS;
47  }
48  else if (_res)
49  {
50  value = strdup(_res);
51  mode = MODE_RESOURCE;
52  }
53  else if (_list)
54  {
55  value = (char *)new WDL_PtrList<wdl_turtle_pair>;
56  mode = _list == 2 ? MODE_COLLECTION : MODE_LIST;
57  }
58  else
59  {
60  value = NULL;
61  mode = MODE_BLANK;
62  }
63  }
64  ~wdl_turtle_pair()
65  {
66  free(verb);
67  if (mode > MODE_LITERAL_TYPELESS) free((void *)mode);
68 
69  WDL_PtrList<wdl_turtle_pair> *list = get_list();
70  if (!list) list = get_collection();
71  if (list) { list->Empty(true); delete list; }
72  else free(value);
73  }
74  static void deletePair(wdl_turtle_pair *p) { delete p; }
75 
76  const char *get_verb() { return verb; } // NULL indicates "a"
77  const char *get_resource() { return mode == MODE_RESOURCE ? value : NULL; } // object is a uri <http://foo>, returned as "http://foo", and _not_ URL-decoded (if resolving a path, caller should convert %20 etc)
78  const char *get_literal() { return mode >= MODE_LITERAL_TYPELESS ? value : NULL; } // object is a literal "string" or """string""" or 1.2345 or true/false, etc
79  const char *get_literal_type() { return mode > MODE_LITERAL_TYPELESS ? (const char *)mode : NULL; }
80  WDL_PtrList<wdl_turtle_pair> *get_list() { return mode == MODE_LIST ? ( WDL_PtrList<wdl_turtle_pair>*) value : NULL; } // object is a [] list
81  WDL_PtrList<wdl_turtle_pair> *get_collection() { return mode == MODE_COLLECTION ? ( WDL_PtrList<wdl_turtle_pair>*) value : NULL; } // object is a collection (), ignore verb for these items!
82 };

98  void parse(const char *str, const char *str_end)
99  {
100  // note: specifically does not clear objects, so you can parse multiple files and get the union of them.
101  m_prefixes.DeleteAll();
102  m_base.Set("");
103  m_error_ptr = m_error_msg = NULL;
104  WDL_FastString fs1,fs2;
105  for (;;)
106  {
107  int tok_l;
108  const char *tok = next_tok(&str,str_end, &tok_l);
109  if (!tok) return;
110 
111  if (tok_l == 7 && !memcmp(tok,"@prefix",7))
112  {
113  tok = next_tok(&str,str_end,&tok_l);
114  if (!tok) { on_err("@prefix missing name",str); return; }
115  if (!tok_to_str(tok,tok_l,&fs1,false) || !fs1.GetLength() || fs1.Get()[fs1.GetLength()-1] != ':')
116  {
117  on_err("@prefix parameter must end in :",tok);
118  return;
119  }
120  tok = next_tok(&str,str_end,&tok_l);
121  if (!tok) { on_err("@prefix missing uriref",str); return; }
122  if (tok_to_str(tok,tok_l,&fs2,true) != '<') { on_err("@prefix invalid uriref",tok); return; }
123 
124  m_prefixes.Insert(fs1.Get(),strdup(fs2.Get()));
125 
126  tok = next_tok(&str,str_end,&tok_l);
127  if (!tok || tok_l != 1 || *tok != '.') { on_err("@prefix expected trailing .",tok?tok:str); return; }
128  }
129  else if (tok_l == 5 && !memcmp(tok,"@base",5))
130  {
131  tok = next_tok(&str,str_end,&tok_l);
132  if (!tok) { on_err("@base missing uriref",str); return; }
133  if (tok_to_str(tok,tok_l,&m_base,true) != '<') { on_err("@base invalid uriref",tok); return; }
134  if (!tok || tok_l != 1 || *tok != '.') { on_err("@base expected trailing .",tok?tok:str); return; }
135  }
136  else
137  {
138  if (tok_to_str(tok,tok_l,&fs1,true) != '<') { on_err("expected IRI to begin triple",tok); return; }
139 
140  wdl_turtle_pair *list = objects.Get(fs1.Get());
141  if (!list)
142  {
143  list = new wdl_turtle_pair(NULL,NULL,NULL,NULL,1);
144  objects.Insert(fs1.Get(),list);
145  }
146  if (!parse_list(&str,str_end,list->get_list(),'.')) return;
147  }
148  }

151  static int is_number_tok(const char *p, const char *str_end)
152  {
153  int i = 0, f = 0; // 1=has number before e 2=has . 4=has e/E, 8=has number after e
154  while (p+i < str_end)
155  {
156  if (p[i] >= '0' && p[i] <= '9') f |= f>3 ? 8 : 1;
157  else if (p[i] == '.') { if (f>1) return 0; f|=2; }
158  else if (p[i] == 'e' || p[i] == 'E')
159  {
160  if ((f&4) || !(f&1) || p+i+1 >= str_end) return 0;
161  if (p[i+1] == '+' || p[i+1] == '-') i++;
162  f|=4;
163  }
164  else if (p[i] == '+' || p[i] == '-') { if (i) return 0; }
165  else break;
166  i++;
167  }
168  if (!(f&1) || (f&(4|8)) == 4) return 0;
169  return i;
170  }

186  const char *next_tok(const char **str, const char *str_end, int *toklen)
187  {
188  const char *tok_start = *str;
189  for (;;)
190  {
191  while (tok_start < str_end && is_ws(*tok_start)) tok_start++;
192  if (tok_start >= str_end) { *str = str_end; return NULL; }
193  if (*tok_start != '#') break;
194  while (tok_start < str_end && *tok_start != '\r' && *tok_start != '\n') tok_start++;
195  }
196 
197  if (tok_start[0] == '"' || tok_start[0] == '\'')
198  {
199  const char term_c = tok_start[0], *p = tok_start + 1;
200  if (tok_start+2 < str_end && tok_start[1] == term_c && tok_start[2] == term_c)
201  {
202  p+=2;
203  while (p < str_end)
204  {
205  if (p+2 < str_end && p[0] == term_c && p[1] == term_c && p[2] == term_c)
206  {
207  *toklen = (int) ((*str = p+3) - tok_start);
208  return tok_start;
209  }
210  if (decode_escape(&p,str_end,true) < 0)
211  {
212  on_err("invalid escape sequence in long quote", p);
213  return NULL;
214  }
215  }
216  on_err("unterminated long quote",tok_start);
217  return NULL;
218  }
219  while (p < str_end)
220  {
221  if (p[0] == term_c)
222  {
223  *toklen = (int) ((*str = p+1) - tok_start);
224  return tok_start;
225  }
226  if (p[0] == '\r' || p[0] == '\n' || decode_escape(&p,str_end,true) < 0)
227  {
228  on_err(p[0] == '\\' ? "invalid escape sequence in quote" : "invalid character in quote", p);
229  return NULL;
230  }
231  }
232  on_err("unterminated quote", tok_start);
233  return NULL;
234  }
235 
236  if (tok_start[0] == '<')
237  {
238  const char *p = tok_start + 1;
239  while (p < str_end)
240  {
241  if (p[0] == '>')
242  {
243  *toklen = (int) ((*str = p+1) - tok_start);
244  return tok_start;
245  }
246 
247  if ((p[0] >= 0 && p[0] <= 0x20) || p[0] == '<' || p[0] == '|' ||
248  p[0] == '"' || p[0] == '{' || p[0] == '}' || p[0] == '`' ||
249  decode_escape(&p,str_end,false) < 0
250  )
251  {
252  on_err(p[0] == '\\' ? "invalid escape sequence in IRI" : "invalid character in IRI", p);
253  return NULL;
254  }
255  }
256  on_err("unterminated IRI", tok_start);
257  return NULL;
258  }
259 
260  const char *p = tok_start;
261  int nlen;
262  bool had_colon = false;
263  if ((nlen = is_number_tok(p,str_end))) p+=nlen;
264  else if (is_single_char_tok(*p)) p++;
265  else while (p < str_end && !is_ws(*p))
266  {
267  if (*p == '<') break;
268  else if (*p == ':') had_colon = true;
269  else if (had_colon && *p == '.' && (p+1) < str_end && !is_ws(p[1]) && !is_single_char_tok(p[1])) { }
270  else if (is_single_char_tok(*p)) break;
271  p++;
272  }
273  *toklen = (int)((*str = p) - tok_start);
274  return tok_start;
275  }

277  static int decode_escape(const char **p, const char *endp, bool allow_echar)
278  // if string is not beginning with '\\', returns unsigned raw character value, advances p by 1
279  // if valid escape sequence, returns non-negative value (may be UTF-8 codepoint), advances p by length (2 or more)
280  // if error occurs, returns -1
281  {
282  const char *rd = *p;
283  if (*rd != '\\')
284  {
285  *p += 1;
286  return (int) (unsigned char) *rd;
287  }
288  if (++rd >= endp) return -1;
289 
290  if (*rd == 'u' || *rd == 'U')
291  {
292  const int hexlen = *rd++ == 'u' ? 4 : 8;
293  unsigned int rv = 0;
294  if (rd + hexlen > endp) return -1;
295  for (int x = 0; x < hexlen; x ++)
296  {
297  const char c = *rd++;
298  if (c >= '0' && c <= '9') rv = rv*16 + (c - '0');
299  else if (c >= 'A' && c <= 'F') rv = rv*16 + 10 + (c - 'A');
300  else if (c >= 'a' && c <= 'f') rv = rv*16 + 10 + (c - 'a');
301  else return -1;
302  }
303  if (rv >= 0x200000) return -1;
304  *p += 2 + hexlen;
305  return (int)rv;
306  }
307 
308  if (!allow_echar) return -1;
309 
310  int rv;
311  switch (*rd)
312  {
313  case 'b': rv = '\b'; break;
314  case 'f': rv = '\f'; break;
315  case 't': rv = '\t'; break;
316  case 'n': rv = '\n'; break;
317  case 'r': rv = '\r'; break;
318  case '"': rv = '"'; break;
319  case '\'': rv = '\''; break;
320  case '\\': rv = '\\'; break;
321  default: return -1;
322  }
323  *p += 2;
324  return rv;
325  }

327  int tok_to_str(const char *tok, int toklen, WDL_FastString *fs, bool allow_decode)
328  // returns '"' for string (regardless of whether it was long or single quoted),
329  // or '<' for IRI
330  // or 1 for unquoted literal
331  // 0 for failure
332  {
333  int mode = 1, rv = 1;
334  fs->Set("");
335  if (allow_decode)
336  {
337  if (toklen >= 6 && (tok[0] == '"' || tok[0] == '\'') && tok[1] == tok[0] && tok[2] == tok[0])
338  {
339  rv = mode = tok[0];
340  WDL_ASSERT(tok[toklen-1] == tok[0] && tok[toklen-2] == tok[0] && tok[toklen-3] == tok[0]);
341  tok += 3;
342  toklen -= 6;
343  }
344  else if (toklen >= 2 && (tok[0] == '"' || tok[0] == '\'' || tok[0] == '<'))
345  {
346  rv = mode = tok[0];
347  WDL_ASSERT(tok[toklen-1] == (mode == '<' ? '>' : mode));
348  toklen-=2;
349  tok++;
350  }
351  else
352  {
353  int l;
354  for (l = 0; l < toklen && tok[l] != ':'; l++);
355  if (l < toklen)
356  {
357  WDL_FastString tmp;
358  tmp.Set(tok,l+1);
359  const char *pf = m_prefixes.Get(tmp.Get());
360  if (!pf)
361  {
362  on_err("prefix not found for token",tok);
363  return 0;
364  }
365  fs->Set(pf);
366  toklen -= l+1;
367  tok += l+1;
368  rv = '<';
369  }
370  }
371  }
372 
373  const char *endp = tok+toklen;
374  while (tok < endp)
375  {
376  if (mode > 1 && *tok == '\\')
377  {
378  const int cv = decode_escape(&tok,endp, mode!='<');
379  if (cv < 0)
380  {
381  on_err(mode == '<' ? "invalid escape sequence in IRI" : "invalid escape sequence in quote", tok);
382  return 0;
383  }
384  char tmp[8];
385  const int tmpl = wdl_utf8_makechar(cv,tmp,sizeof(tmp));
386  if (tmpl <= 0)
387  {
388  on_err(mode == '<' ? "invalid unicode codepoint in IRI" : "invalid unicode codepoint in quote", tok);
389  return 0;
390  }
391  fs->Append(tmp,tmpl);
392  }
393  else
394  fs->Append(tok++,1);
395  }
396  if (mode == '<' && !strstr(fs->Get(),"://") && m_base.GetLength()) fs->Insert(m_base.Get(),0);
397  if (rv == '\'') rv = '"';
398  return rv;
399  }

401  bool parse_list(const char **str, const char *str_end, WDL_PtrList<wdl_turtle_pair> *listOut, int endchar)
402  {
403  if (WDL_NOT_NORMALLY(listOut == NULL)) return false;
404  WDL_FastString verb_str, s, langtype;
405  const char *has_verb = NULL;
406  int state = 0, tok_l;
407  for (;;)
408  {
409  const char *tok = next_tok(str,str_end,&tok_l);
410  if (!tok) { on_err("unterminated list",*str); return false; }
411 
412  if (endchar == ')')
413  {
414  if (tok[0] == ')') return true;
415  state=1;
416  }
417  else if (tok[0] == endchar)
418  {
419  if (state!=1) return true;
420  on_err("premature list terminator",tok);
421  return false;
422  }
423  if ((tok_l==1 && tok[0] == '.') || tok[0] == ']' || tok[0] == ')') { on_err("invalid character in list context",tok); return false; }
424 
425  switch (state)
426  {
427  case 0:
428  if (tok_l == 1 && tok[0] == 'a') has_verb = NULL;
429  else if (tok_to_str(tok,tok_l,&verb_str,true) == '<' && verb_str.GetLength()) has_verb = verb_str.Get();
430  else { on_err("invalid verb in list",tok); return false; }
431  state++;
432  break;
433  case 1:
434  if (tok[0] == '[')
435  {
436  wdl_turtle_pair *list = new wdl_turtle_pair(has_verb, NULL,NULL,NULL,1);
437  listOut->Add(list);
438  if (!parse_list(str,str_end,list->get_list(),']')) return false;
439  }
440  else if (tok[0] == '(')
441  {
442  wdl_turtle_pair *list = new wdl_turtle_pair(has_verb, NULL,NULL,NULL,2);
443  listOut->Add(list);
444  if (!parse_list(str,str_end,list->get_collection(),')')) return false;
445  }
446  else
447  {
448  int t = tok_to_str(tok,tok_l,&s,true), has_langtype = 0;
449  if (!t) { on_err("error parsing object",tok); return false; }
450  if (t != '<')
451  {
452  if (t == '"')
453  {
454  const char *peek = *str;
455  if (peek < str_end && peek[0] == '@')
456  {
457  // this is a bit more permissive than the spec. include @ in string
458  tok = next_tok(str,str_end,&tok_l);
459  if (!tok || tok != peek) { on_err("literal has incorrect language specification",peek); return false; }
460  has_langtype++;
461  langtype.Set(tok,tok_l);
462  }
463  else if (peek+1 < str_end && peek[0] == '^' && peek[1] == '^')
464  {
465  peek = (*str += 2); // do not include ^^ in token
466  tok = next_tok(str,str_end,&tok_l);
467  if (!tok || tok != peek || tok_to_str(tok,tok_l,&langtype,true) == '"') { on_err("literal has incorrect type specification",peek); return false; }
468  has_langtype++;
469  }
470  }
471  }
472  listOut->Add(new wdl_turtle_pair(has_verb, t == '<' ? s.Get() : NULL, t == '<' ? NULL : s.Get(), has_langtype ? langtype.Get() : NULL, 0));
473  }
474  state++;
475  break;
476  case 2:
477  if (tok[0] == ';') state=0;
478  else if (tok[0] == ',') state = 1;
479  else { on_err("unknown token in list",tok); return false; }
480  break;
481  }
482  }
483  }
