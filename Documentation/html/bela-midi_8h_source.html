42  static void midiCallback(MidiChannelMessage message, void* arg)
43  {
44  bela_midi* midi = static_cast<bela_midi*>(arg);
45  int type = message.getType(); // which MIDI message, 128-255
46  int channel = message.getChannel(); // which MIDI channel, 1-16
47  double time = 0.;
48 
49  switch (type) {
50  case kmmNoteOff:
51  midi->handleKeyOff(time, channel, message.getDataByte(0), message.getDataByte(1));
52  break;
53  case kmmNoteOn:
54  midi->handleKeyOn(time, channel, message.getDataByte(0), message.getDataByte(1));
55  break;
56  case kmmPolyphonicKeyPressure:
57  midi->handlePolyAfterTouch(time, channel, message.getDataByte(0), message.getDataByte(1));
58  break;
59  case kmmControlChange:
60  midi->handleCtrlChange(time, channel, message.getDataByte(0), message.getDataByte(1));
61  break;
62  case kmmProgramChange:
63  midi->handleProgChange(time, channel, message.getDataByte(0));
64  break;
65  case kmmChannelPressure:
66  midi->handleAfterTouch(time, channel, message.getDataByte(0));
67  break;
68  case kmmPitchBend:
69  midi->handlePitchWheel(time, channel, message.getDataByte(0), message.getDataByte(1));
70  break;
71  case kmmSystem:
72  {
73  // We have to re-build the MIDI message:
74  int status = message.getStatusByte();
75  int systemRealtimeByte = channel | status;
76 
77  switch (systemRealtimeByte)
78  {
79  case MIDI_CLOCK:
80  midi->handleClock(time);
81  break;
82  case MIDI_START:
83  // We can consider start and continue as identical messages
84  case MIDI_CONT:
85  midi->handleStart(time);
86  break;
87  case MIDI_STOP:
88  midi->handleStop(time);
89  break;
90  case MIDI_SYSEX_START:
91  #if 0 // this is not implemented on Bela yet
92  std::vector<unsigned char> sysex;
93  for (unsigned int j = 0; j < message.getNumDataBytes(); j++) {
94  sysex.push_back(message.getDataByte(j));
95  }
96  // Would be nice to do this:
97  // std::vector<unsigned char> sysex(message.getData(), message.getData() + message.getNumDataBytes());
98  midi->handleSysex(time, sysex);
99  #endif
100  break;
101  default:
102  break;
103  }
104  break;
105  }
106  case kmmNone:
107  case kmmAny:
108  default:
109  break;
110  }
111  }
