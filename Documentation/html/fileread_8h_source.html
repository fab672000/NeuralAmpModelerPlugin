104  BOOL HasUTF8(const char *_str)
105  {
106  const unsigned char *str = (const unsigned char *)_str;
107  if (!str) return FALSE;
108  while (*str)
109  {
110  unsigned char c = *str;
111  if (c >= 0xC2)
112  {
113  if (c <= 0xDF && str[1] >=0x80 && str[1] <= 0xBF) return TRUE;
114  else if (c <= 0xEF && str[1] >=0x80 && str[1] <= 0xBF && str[2] >=0x80 && str[2] <= 0xBF) return TRUE;
115  else if (c <= 0xF4 && str[1] >=0x80 && str[1] <= 0xBF && str[2] >=0x80 && str[2] <= 0xBF) return TRUE;
116  }
117  str++;
118  if (((const char *)str-_str) >= 256) return TRUE; // long filenames get converted to wide
119  }
120  return FALSE;
121  }

407  int RunReads()
408  {
409  while (m_pending.GetSize())
410  {
411  WDL_FileRead__ReadEnt *ent=m_pending.Get(0);
412  DWORD s=0;
413 
414  if (!ent->m_size && !GetOverlappedResult(m_fh,&ent->m_ol,&s,FALSE)) break;
415  m_pending.Delete(0);
416  if (!ent->m_size) ent->m_size=s;
417  m_full.Add(ent);
418  }
419 
420 
421  if (m_empties.GetSize()>0)
422  {
423  if (m_async_readpos < m_file_position) m_async_readpos = m_file_position;
424 
425  if (m_async==1) m_async_readpos &= ~((WDL_FILEREAD_POSTYPE) WDL_UNBUF_ALIGN-1);
426 
427  if (m_async_readpos >= m_fsize) return 0;
428 
429  const int rdidx=m_empties.GetSize()-1;
430  WDL_FileRead__ReadEnt *t=m_empties.Get(rdidx);
431 
432  ResetEvent(t->m_ol.hEvent);
433 
434  *(WDL_FILEREAD_POSTYPE *)&t->m_ol.Offset = m_async_readpos;
435 
436  m_async_readpos += m_async_bufsize;
437  DWORD dw;
438  if (ReadFile(m_fh,t->m_buf,m_async_bufsize,&dw,&t->m_ol))
439  {
440  if (!dw) return 1;
441  }
442  else
443  {
444  if (GetLastError() != ERROR_IO_PENDING) return 1;
445  dw=0;
446  }
447  t->m_size=dw;
448  m_empties.Delete(rdidx);
449  m_pending.Add(t);
450  }
451  return 0;
452  }

454  int AsyncRead(char *buf, int maxlen)
455  {
456  char *obuf=buf;
457  int lenout=0;
458  if (m_file_position+maxlen > m_fsize)
459  {
460  maxlen=(int) (m_fsize-m_file_position);
461  }
462  if (maxlen<1) return 0;
463 
464  int errcnt=!!m_async_hashaderr;
465  do
466  {
467  while (m_full.GetSize() > 0)
468  {
469  WDL_FileRead__ReadEnt *ti=m_full.Get(0);
470  WDL_FILEREAD_POSTYPE tiofs=*(WDL_FILEREAD_POSTYPE *)&ti->m_ol.Offset;
471  if (m_file_position >= tiofs && m_file_position < tiofs + ti->m_size)
472  {
473  if (maxlen < 1) break;
474 
475  int l=ti->m_size-(int) (m_file_position-tiofs);
476  if (l > maxlen) l=maxlen;
477 
478  memcpy(buf,(char *)ti->m_buf+m_file_position - tiofs,l);
479  buf += l;
480  m_file_position += l;
481  maxlen -= l;
482  lenout += l;
483  }
484  else
485  {
486  m_empties.Add(ti);
487  m_full.Delete(0);
488  }
489  }
490 
491  if (maxlen > 0 && m_async_readpos != m_file_position)
492  {
493  int x;
494  for (x = 0; x < m_pending.GetSize(); x ++)
495  {
496  WDL_FileRead__ReadEnt *ent=m_pending.Get(x);
497  WDL_FILEREAD_POSTYPE tiofs=*(WDL_FILEREAD_POSTYPE *)&ent->m_ol.Offset;
498  if (m_file_position >= tiofs && m_file_position < tiofs + m_async_bufsize) break;
499  }
500  if (x == m_pending.GetSize())
501  {
502  m_async_readpos=m_file_position;
503  }
504  }
505 
506  errcnt+=RunReads();
507 
508  if (maxlen > 0 && m_pending.GetSize() && !m_full.GetSize())
509  {
510  WDL_FileRead__ReadEnt *ent=m_pending.Get(0);
511  m_pending.Delete(0);
512 
513  if (ent->m_size) m_full.Add(ent);
514  else
515  {
516 // WaitForSingleObject(ent->m_ol.hEvent,INFINITE);
517 
518  DWORD s=0;
519  if (GetOverlappedResult(m_fh,&ent->m_ol,&s,TRUE) && s)
520  {
521  ent->m_size=s;
522  m_full.Add(ent);
523  }
524  else // failed read, set the error flag
525  {
526  errcnt++;
527  ent->m_size=0;
528  m_empties.Add(ent);
529  }
530  }
531  }
532  }
533  while (maxlen > 0 && (m_pending.GetSize()||m_full.GetSize()) && !errcnt);
534  if (!errcnt) RunReads();
535  else m_async_hashaderr=true;
536 
537  return lenout;
538  }

542  void *GetMappedView(int offs, int *len)
543  {
544  if (!m_mmap_view && !m_mmap_totalbufmode) return 0;

549  return (char *)m_mmap_view + offs;

554  int Read(void *buf, int len)
555  {
556  if (m_mmap_view||m_mmap_totalbufmode)
557  {
558  int maxl=(int) (m_fsize-m_file_position);
559  if (maxl > len) maxl=len;
560  if (maxl < 0) maxl=0;
561  if (maxl>0)
562  {
563  if (m_mmap_view)
564  memcpy(buf,(char *)m_mmap_view + (int)m_file_position,maxl);
565  else
566  memcpy(buf,(char *)m_mmap_totalbufmode + (int)m_file_position,maxl);
567 
568  }
569  m_file_position+=maxl;
570  return maxl;
571  }
572 
573  if (m_fsize_maychange) GetSize(); // update m_fsize
574 
575 #ifdef WDL_WIN32_NATIVE_READ
576  if (m_fh == INVALID_HANDLE_VALUE||len<1) return 0;
577 
578  if (m_async>0)
579  {
580  return AsyncRead((char *)buf,len);
581  }
582 #elif defined(WDL_POSIX_NATIVE_READ)
583  if (m_filedes<0 || len<1) return 0;
584 
585 #else
586  if (!m_fp || len<1) return 0;
587 
588 #endif
589 
590  if (m_bufspace.GetSize()>=WDL_UNBUF_ALIGN*2-1)
591  {
592  int rdout=0;
593  int sz=m_bufspace.GetSize()-(WDL_UNBUF_ALIGN-1);
594  char *srcbuf=(char *)m_bufspace.Get(); // read size
595  if (((int)(INT_PTR)srcbuf)&(WDL_UNBUF_ALIGN-1)) srcbuf += WDL_UNBUF_ALIGN-(((int)(INT_PTR)srcbuf)&(WDL_UNBUF_ALIGN-1));
596  while (len > rdout)
597  {
598  int a=m_sync_bufmode_used-m_sync_bufmode_pos;
599  if (a>(len-rdout)) a=(len-rdout);
600  if (a>0)
601  {
602  memcpy((char*)buf+rdout,srcbuf+m_sync_bufmode_pos,a);
603  rdout+=a;
604  m_sync_bufmode_pos+=a;
605  m_file_position+=a;
606  }
607 
608  if (len > rdout)
609  {
610  m_sync_bufmode_used=0;
611  m_sync_bufmode_pos=0;
612 
613  int thissz=sz;
614  if (m_syncrd_firstbuf) // this is a scheduling mechanism to avoid having reads on various files always happening at the same time -- not needed in async modes, only in sync with large buffers
615  {
616  m_syncrd_firstbuf=false;
617  const int blocks = thissz/WDL_UNBUF_ALIGN;
618  if (blocks > 1)
619  {
620  static int rrs; // may not be ideal on multithread, but having it incorrect isnt a big deal.
621  if (blocks>7) thissz >>= (rrs++)&3;
622  else thissz>>= (rrs++)&1;
623  }
