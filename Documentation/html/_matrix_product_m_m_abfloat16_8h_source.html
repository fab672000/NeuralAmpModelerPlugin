297 template<typename DataMapper>
298 void gemmMMAbfloat16(const DataMapper& res, const bfloat16* indexA, const bfloat16* indexB, Index rows, Index depth, Index cols, bfloat16 alpha, Index strideA, Index strideB, Index offsetA, Index offsetB)
299 {
300  float falpha = Eigen::bfloat16_impl::bfloat16_to_float(alpha);
301  const Packet4f pAlpha = pset1<Packet4f>(falpha);
302  ei_declare_aligned_stack_constructed_variable(float, result, cols*rows, 0);
303 
304  convertArrayBF16toF32<DataMapper>(result, cols, rows, res);
305 
306  if( strideA == -1 ) strideA = depth;
307  if( strideB == -1 ) strideB = depth;
308  // Packing is done in blocks.
309  // There's 4 possible sizes of blocks
310  // Blocks of 8 columns with 16 elements (8x16)
311  // Blocks of 8 columns with 8 elements (8x8). This happens when there's 16 > rows >= 8
312  // Blocks of 8 columns with 4 elements (8x4). This happens when there's 8 > rows >= 4
313  // Blocks of 8 columns with < 4 elements. This happens when there's less than 4 remaining rows
314 
315  // Loop for LHS standard block (8x16)
316  Index bigSuffix = (2*8) * (strideA-offsetA);
317  indexB += 4*offsetB;
318  strideB *= 4;
319  offsetB *= 3;
320 
321  Index row = 0;
322  while(row + 16 <= rows){
323  calcColLoops<16>(indexA, row, depth, cols, rows, pAlpha, indexB, strideB, offsetA, offsetB, bigSuffix, result);
324  }
325  // LHS (8x8) block
326  calcColLoops<8>(indexA, row, depth, cols, rows, pAlpha, indexB, strideB, offsetA, offsetB, bigSuffix, result);
327  // LHS (8x4) block
328  calcColLoops<4>(indexA, row, depth, cols, rows, pAlpha, indexB, strideB, offsetA, offsetB, bigSuffix, result);
329  // extra rows
330  if(rows & 3){
331  // This index is the beginning of remaining block.
332  colLoops<4, true>(depth, cols, rows, pAlpha, indexA, indexB, strideB, offsetB, result + row);
333  }
334 
335  // Convert back to bfloat16
336  convertArrayF32toBF16<DataMapper>(result, cols, rows, res);
337 }
