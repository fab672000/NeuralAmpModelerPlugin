60  double scale(double x) const
61  {
62  if (fMinIn < fMaxIn) {
63  // increasing control
64  double z = (x < fMinIn) ? fMinIn : (x > fMaxIn) ? fMaxIn : x;
65  return fMinOut + (z-fMinIn)*(fMaxOut-fMinOut)/(fMaxIn-fMinIn);
66 
67  } else if (fMinIn > fMaxIn) {
68  // reversed control
69  double z = (x < fMaxIn) ? fMaxIn : (x > fMinIn) ? fMinIn : x;
70  return fMinOut + (fMinIn-z)*(fMaxOut-fMinOut)/(fMinIn-fMaxIn);
71 
72  } else {
73  // no control !
74  return (fMinOut+fMaxOut)/2.0f;
75  }
76  }
77 
78  double invscale(double x) const
79  {
80  if (fMinOut < fMaxOut) {
81  // increasing control
82  double z = (x < fMinOut) ? fMinOut : (x > fMaxOut) ? fMaxOut : x;
83  return fMinIn + (z-fMinOut)*(fMaxIn-fMinIn)/(fMaxOut-fMinOut);

85  } else if (fMinOut > fMaxOut) {
86  // reversed control
