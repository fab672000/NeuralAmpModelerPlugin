63 /*
64 
65 Note about URLs and the Faust UI names:
66 ----------------------------------------------------
67 Characters in a url could be:
68 1. Reserved: ; / ? : @ & = + $ ,
69  These characters delimit URL parts.
70 2. Unreserved: alphanum - _ . ! ~ * ' ( )
71  These characters have no special meaning and can be used as is.
72 3. Excluded: control characters, space, < > # % ", { } | \ ^ [ ] `
73 
74 To solve potential conflicts between the Faust UI objects naming scheme and
75 the URL allowed characters, the reserved and excluded characters are replaced
76 with '-' (hyphen).
77 Space or tabulation are replaced with '_' (underscore)
78 */

86  const char* tr(const char* label) const
87  {
88  static char buffer[1024];
89  char * ptr = buffer; int n = 1;
90  while (*label && (n++ < 1024)) {
91  switch (*label) {
92  case ' ': case ' ':
93  *ptr++ = '_';
94  break;
95  case ';': case '/': case '?': case ':': case '@':
96  case '&': case '=': case '+': case '$': case ',':
97  case '<': case '>': case '#': case '%': case '"':
98  case '{': case '}': case '|': case '\\': case '^':
99  case '[': case ']': case '`':
100  *ptr++ = '_';
101  break;
102  default:
103  *ptr++ = *label;
104  }
105  label++;
106  }
107  *ptr = 0;
108  return buffer;

210  static void* UpdateUI(void* arg)
211  {
212  httpdClientUI* ui = static_cast<httpdClientUI*>(arg);
213  std::map<std::string, FAUSTFLOAT*>::iterator it;
214  while (ui->fRunning) {
