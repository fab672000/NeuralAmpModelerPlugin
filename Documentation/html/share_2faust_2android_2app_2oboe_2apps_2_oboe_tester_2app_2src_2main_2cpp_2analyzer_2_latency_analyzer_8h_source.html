56 // Calculate a normalized cross correlation.
57 static double calculateNormalizedCorrelation(const float *a,
58  const float *b,
59  int windowSize)
60 {
61  double correlation = 0.0;
62  double sumProducts = 0.0;
63  double sumSquares = 0.0;
64 
65  // Correlate a against b.
66  for (int i = 0; i < windowSize; i++) {
67  float s1 = a[i];
68  float s2 = b[i];
69  // Use a normalized cross-correlation.
70  sumProducts += s1 * s2;
71  sumSquares += ((s1 * s1) + (s2 * s2));
72  }
73 
74  if (sumSquares >= 1.0e-9) {
75  correlation = (float) (2.0 * sumProducts / sumSquares);
76  }
77  return correlation;
78 }

80 static double calculateRootMeanSquare(float *data, int32_t numSamples) {
81  double sum = 0.0;
82  for (int32_t i = 0; i < numSamples; i++) {
83  float sample = data[i];
84  sum += sample * sample;
85  }
86  return sqrt(sum / numSamples);
87 }

198 static int measureLatencyFromPulse(AudioRecording &recorded,
199  AudioRecording &pulse,
200  int32_t framesPerEncodedBit,
201  LatencyReport *report) {
202 
203  report->latencyInFrames = 0;
204  report->confidence = 0.0;
205 
206  int numCorrelations = recorded.size() - pulse.size();
207  if (numCorrelations < 10) {
208  LOGE("%s() recording too small = %d frames", __func__, recorded.size());
209  return -1;
210  }
211  std::unique_ptr<float[]> correlations= std::make_unique<float[]>(numCorrelations);
212 
213  // Correlate pulse against the recorded data.
214  for (int i = 0; i < numCorrelations; i++) {
215  float correlation = (float) calculateNormalizedCorrelation(&recorded.getData()[i],
216  &pulse.getData()[0],
217  pulse.size());
218  correlations[i] = correlation;
219  }
220 
221  // Find highest peak in correlation array.
222  float peakCorrelation = 0.0;
223  int peakIndex = -1;
224  for (int i = 0; i < numCorrelations; i++) {
225  float value = abs(correlations[i]);
226  if (value > peakCorrelation) {
227  peakCorrelation = value;
228  peakIndex = i;
229  }
230  }
231  if (peakIndex < 0) {
232  LOGE("%s() no signal for correlation", __func__);
233  return -2;
234  }
235 
236  report->latencyInFrames = peakIndex;
237  report->confidence = peakCorrelation;
238 
239  return 0;
240 }
