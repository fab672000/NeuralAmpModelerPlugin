106  BOOL HasUTF8(const char *_str)
107  {
108  const unsigned char *str = (const unsigned char *)_str;
109  if (!str) return FALSE;
110  while (*str)
111  {
112  unsigned char c = *str;
113  if (c >= 0xC2)
114  {
115  if (c <= 0xDF && str[1] >=0x80 && str[1] <= 0xBF) return TRUE;
116  else if (c <= 0xEF && str[1] >=0x80 && str[1] <= 0xBF && str[2] >=0x80 && str[2] <= 0xBF) return TRUE;
117  else if (c <= 0xF4 && str[1] >=0x80 && str[1] <= 0xBF && str[2] >=0x80 && str[2] <= 0xBF) return TRUE;
118  }
119  str++;
120  if (((const char *)str-_str) >= 256) return TRUE; // long filenames get converted to wide
121  }
122  return FALSE;
123  }

156  if (m_async && allow_async == 3 && !wantAppendTo)

158  m_async = 3;
159  bufsize = (bufsize+4095)&~4095;

164  int createFlag= wantAppendTo?OPEN_ALWAYS:CREATE_ALWAYS;
165  int shareFlag = noFileLocking ? (FILE_SHARE_READ|FILE_SHARE_WRITE) : FILE_SHARE_READ;
166  int flag = FILE_ATTRIBUTE_NORMAL;
167 
168  if (m_async)
169  {
170  rwflag |= GENERIC_READ;
171  if (m_async == 3)
172  flag |= FILE_FLAG_OVERLAPPED|FILE_FLAG_NO_BUFFERING|FILE_FLAG_WRITE_THROUGH;
173  else
174  flag |= FILE_FLAG_OVERLAPPED|(allow_async>1 ? FILE_FLAG_WRITE_THROUGH: 0);

178 #ifndef WDL_NO_SUPPORT_UTF8

197  {
198  correctlongpath(wfilename);
199  m_fh = CreateFileW(wfilename,rwflag,shareFlag,NULL,createFlag,flag,NULL);
200  }
201  }
202  }
203 
204  if (m_fh == INVALID_HANDLE_VALUE)
205 #endif

207  }

211  m_async_bufsize=bufsize;
212  m_async_maxbufs=maxbufs;
213  m_async_minbufs=minbufs;
214  int x;
215  for (x = 0; x < m_async_minbufs; x ++)

217  WDL_FileWrite__WriteEnt *t=new WDL_FileWrite__WriteEnt(m_async_bufsize);

233  | O_CLOEXEC

238

272  }

335  int Write(const void *buf, int len)
336  {
337 #ifdef WDL_WIN32_NATIVE_WRITE
338  if (m_fh == INVALID_HANDLE_VALUE) return 0;
339 
340  if (m_async)
341  {
342  int rdpos = 0;
343  while (len > 0)
344  {
345  if (!m_empties.GetSize())
346  {
347  WDL_FileWrite__WriteEnt *ent=m_pending.Get(0);
348  DWORD s=0;
349  if (ent)
350  {
351  bool wasabort=false;
352  DWORD err;
353  if (GetOverlappedResult(m_fh,&ent->m_ol,&s,FALSE)||
354  (wasabort=((err=GetLastError())==ERROR_OPERATION_ABORTED)))
355  {
356  m_pending.Delete(0);
357 
358  if (wasabort)
359  {
360  if (!RunAsyncWrite(ent,false)) m_empties.Add(ent);
361  }
362  else
363  {
364  m_empties.Add(ent);
365  ent->m_bufused=0;
366  }
367  }
368  else if (err != ERROR_IO_PENDING && err != ERROR_IO_INCOMPLETE)
369  {
370  WDL_FILEWRITE_ON_ERROR(err == ERROR_DISK_FULL)
371  }
372  }
373  }
374 
375 
376  WDL_FileWrite__WriteEnt *ent=m_empties.Get(0);
377  if (!ent)
378  {
379  if (m_pending.GetSize()>=m_async_maxbufs)
380  {
381  SyncOutput(false);
382  }
383 
384  if (!(ent=m_empties.Get(0)))
385  m_empties.Add(ent = new WDL_FileWrite__WriteEnt(m_async_bufsize)); // new buffer
386 
387 
388  }
389 
390  int ml=ent->m_bufsz-ent->m_bufused;
391  if (ml>len) ml=len;
392  memcpy(ent->m_bufptr+ent->m_bufused,(const char *)buf + rdpos,ml);
393 
394  ent->m_bufused+=ml;
395  len-=ml;
396  rdpos+=ml;
397 
398  if (ent->m_bufused >= ent->m_bufsz)
399  {
400  if (RunAsyncWrite(ent,true)) m_empties.Delete(0); // if queued remove from list
401  }
402  }
403  return rdpos;
404  }
405  else
406  {
407  DWORD dw=0;
408  if (!WriteFile(m_fh,buf,len,&dw,NULL))
409  {
410  WDL_FILEWRITE_ON_ERROR(GetLastError() == ERROR_DISK_FULL)
411  }
412  m_file_position+=dw;
413  if (m_file_position>m_file_max_position) m_file_max_position=m_file_position;
414  return dw;
415  }
416 #elif defined(WDL_POSIX_NATIVE_WRITE)
417  if (m_bufspace.GetSize()>0)
418  {
419  char *rdptr = (char *)buf;
420  int rdlen = len;
421  while (rdlen>0)
422  {
423  int amt = m_bufspace.GetSize() - m_bufspace_used;
424  if (amt>0)
425  {
426  if (amt>rdlen) amt=rdlen;
427  memcpy((char *)m_bufspace.Get()+m_bufspace_used,rdptr,amt);
428  m_bufspace_used += amt;
429  rdptr+=amt;
430  rdlen -= amt;
431 
432  if (m_file_position+m_bufspace_used > m_file_max_position) m_file_max_position=m_file_position + m_bufspace_used;
433  }
434  if (m_bufspace_used >= m_bufspace.GetSize())
435  {
436  int v=(int)pwrite(m_filedes,m_bufspace.Get(),m_bufspace_used,m_file_position);
437  if (v != m_bufspace_used) { WDL_FILEWRITE_ON_ERROR(v>=0 || errno == EDQUOT || errno == ENOSPC) }
438  if (v>0) m_file_position+=v;
439  m_bufspace_used=0;
440  }
441  }
442  return len;
443  }
444  else
445  {
446  int v=(int)pwrite(m_filedes,buf,len,m_file_position);
447  if (v != len) { WDL_FILEWRITE_ON_ERROR(v>=0 || errno == EDQUOT || errno == ENOSPC) }
448  if (v>0) m_file_position+=v;
449  if (m_file_position > m_file_max_position) m_file_max_position=m_file_position;
450  return v;
451  }
452 #else
453  int written = (int)fwrite(buf,1,len,m_fp);
454  if (written != len) { WDL_FILEWRITE_ON_ERROR(false) }
455  return written;
456 #endif
457

511  bool RunAsyncWrite(WDL_FileWrite__WriteEnt *ent, bool updatePosition) // returns true if ent is added to pending
512  {
513  if (ent && ent->m_bufused>0)
514  {
515  if (updatePosition)
516  {
517  ent->m_last_writepos = m_file_position;
518  m_file_position += ent->m_bufused;
519  if (m_file_position>m_file_max_position) m_file_max_position=m_file_position;
520  }
521 
522  if (m_async == 3 && (ent->m_bufused&4095))
523  {
524  int offs=(ent->m_bufused&4095);
525  char tmp[4096];
526  memset(tmp,0,4096);
527 
528  *(WDL_FILEWRITE_POSTYPE *)&ent->m_ol.Offset = ent->m_last_writepos + ent->m_bufused - offs;
529  ResetEvent(ent->m_ol.hEvent);
530 
531  DWORD dw=0;
532  if (!ReadFile(m_fh,tmp,4096,&dw,&ent->m_ol))
533  {
534  if (GetLastError() == ERROR_IO_PENDING)
535  WaitForSingleObject(ent->m_ol.hEvent,INFINITE);
536  }
537  memcpy(ent->m_bufptr+ent->m_bufused,tmp+offs,4096-offs);
538 
539  ent->m_bufused += 4096-offs;
540  }
541 
542  DWORD d=0;
543 
544  *(WDL_FILEWRITE_POSTYPE *)&ent->m_ol.Offset = ent->m_last_writepos;
545 
546  ResetEvent(ent->m_ol.hEvent);
547 
548  if (!WriteFile(m_fh,ent->m_bufptr,ent->m_bufused,&d,&ent->m_ol))
549  {
550  if (GetLastError()==ERROR_IO_PENDING)
551  {
552  m_pending.Add(ent);
553  return true;
554  }
555  else { WDL_FILEWRITE_ON_ERROR(GetLastError()==ERROR_DISK_FULL) }
556  }
557  ent->m_bufused=0;
558  }
559  return false;
560  }

562  void SyncOutput(bool syncall)
563  {
564  if (syncall)
565  {
566  if (RunAsyncWrite(m_empties.Get(0),true)) m_empties.Delete(0);
567  }
568  for (;;)
569  {
570  WDL_FileWrite__WriteEnt *ent=m_pending.Get(0);
571  if (!ent) break;
572  DWORD s=0;
573  m_pending.Delete(0);
574  BOOL ok = GetOverlappedResult(m_fh,&ent->m_ol,&s,TRUE);
575  int errcode;
576  if (!ok && (errcode=GetLastError())==ERROR_OPERATION_ABORTED)
577  {
578  // rewrite this one
579  if (!RunAsyncWrite(ent,false)) m_empties.Add(ent);
580  }
581  else
582  {
583  if (!ok) { WDL_FILEWRITE_ON_ERROR(errcode==ERROR_DISK_FULL) }
584  m_empties.Add(ent);
585  ent->m_bufused=0;
586  if (!syncall) break;
587  }
588  }
589  }

594  bool SetPosition(WDL_FILEWRITE_POSTYPE pos) // returns 0 on success
595  {
596 #ifdef WDL_WIN32_NATIVE_WRITE
597  if (m_fh == INVALID_HANDLE_VALUE) return true;
598  if (m_async)
599  {
600  SyncOutput(true);
601  m_file_position=pos;
602  if (m_file_position>m_file_max_position) m_file_max_position=m_file_position;
603 
604  if (m_async==3 && (m_file_position&4095))
605  {
606  WDL_FileWrite__WriteEnt *ent=m_empties.Get(0);
607  if (ent)
608  {
609  int psz=(int) (m_file_position&4095);
610 
611  m_file_position -= psz;
612  *(WDL_FILEWRITE_POSTYPE *)&ent->m_ol.Offset = m_file_position;
613  ResetEvent(ent->m_ol.hEvent);
614 
615  DWORD dwo=0;
616  if (!ReadFile(m_fh,ent->m_bufptr,4096,&dwo,&ent->m_ol))
617  {
618  if (GetLastError() == ERROR_IO_PENDING)
619  WaitForSingleObject(ent->m_ol.hEvent,INFINITE);
620  }
621  ent->m_bufused=(int)psz;
622  }
623  }
624  return false;
625  }
626 
627  m_file_position=pos;
628  if (m_file_position>m_file_max_position) m_file_max_position=m_file_position;
629 
630  LONG high=(LONG) (m_file_position>>32);
631  return SetFilePointer(m_fh,(LONG)(m_file_position&((WDL_FILEWRITE_POSTYPE)0xFFFFFFFF)),&high,FILE_BEGIN)==0xFFFFFFFF && GetLastError() != NO_ERROR;
632 #elif defined(WDL_POSIX_NATIVE_WRITE)
633 
634  if (m_filedes < 0) return true;
635  if (m_bufspace.GetSize() > 0 && m_bufspace_used>0)
636  {
637  int v=(int)pwrite(m_filedes,m_bufspace.Get(),m_bufspace_used,m_file_position);
638  if (v>0) m_file_position+=v;
639  if (m_file_position > m_file_max_position) m_file_max_position=m_file_position;
640  m_bufspace_used=0;
641  }
642 
643  m_file_position = pos; // seek!
644  if (m_file_position>m_file_max_position) m_file_max_position=m_file_position;
645  return false;
646 #else
647  if (!m_fp) return true;
648  return !!fseek(m_fp,pos,SEEK_SET);
649 #endif
650  }
