281  bool Descriptions::parse (int ac, char* av[], VariablesMap& result, FilesVector* files) const
282  {
283  using namespace std;
284 
285  for (int i = 1; i < ac; i++)
286  {
287  string current = av[i];
288  if (current[0] == '-')
289  {
290  int pos = current[1] == '-' ? 2 : 1;
291  current = current.substr (pos, string::npos);
292 
293  DescriptionsList::const_iterator found =
294  find (mDescriptions.begin (), mDescriptions.end (), current);
295  if (found != mDescriptions.end ())
296  {
297  result[*found] = "true";
298  if (found->mType != Description::kBool)
299  {
300  if (((i + 1) < ac) && *av[i + 1] != '-')
301  {
302  result[*found] = av[++i];
303  }
304  else
305  {
306  result[*found] = "error!";
307  result.setError ();
308  return false;
309  }
310  }
311  }
312  else
313  {
314  result.setError ();
315  return false;
316  }
317  }
318  else if (files)
319  files->push_back (av[i]);
320  }
321  return true;
322  }

327  void Descriptions::print (std::ostream& os) const
328  {
329  if (!mCaption.empty ())
330  os << mCaption << ":\n";
331 
332  size_t maxLength = 0u;
333  std::for_each (mDescriptions.begin (), mDescriptions.end (),
334  [&] (const Description& d) { maxLength = std::max (maxLength, d.size ()); });
335 
336  for (const Description& opt : mDescriptions)
337  {
338  os << "-" << opt;
339  for (auto s = opt.size (); s < maxLength; ++s)
340  os << " ";
341  os << " | " << opt.mHelp << "\n";
342  }
343  }
