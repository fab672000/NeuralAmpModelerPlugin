313  void GetHSLA(float& h, float& s, float& l, float& a) const
314  {
315  const float fR = R / 255.f;
316  const float fG = G / 255.f;
317  const float fB = B / 255.f;
318  a = A / 255.f;
319 
320  const float fMin = std::min(fR, std::min(fG, fB));
321  const float fMax = std::max(fR, std::max(fG, fB));
322  const float fDiff = fMax - fMin;
323  const float fSum = fMax + fMin;
324 
325  l = 50.f * fSum;
326 
327  if (fMin == fMax) { s = 0.f; h = 0.f; l /= 100.f; return; }
328  else if (l < 50.f) { s = 100.f * fDiff / fSum; }
329  else { s = 100.f * fDiff / (2.f - fDiff); }
330 
331  if (fMax == fR) { h = 60.f * (fG - fB) / fDiff; }
332  if (fMax == fG) { h = 60.f * (fB - fR) / fDiff + 120.f; }
333  if (fMax == fB) { h = 60.f * (fR - fG) / fDiff + 240.f; }
334 
335  if (h < 0.f) { h = h + 360.f; }
336 
337  h /= 360.f;
338  s /= 100.f;
339  l /= 100.f;
340  }

449  static IColor FromHSLA(float h, float s, float l, float a = 1.f)
450  {
451  auto hue = [](float h, float m1, float m2) {
452  if (h < 0) h += 1;
453  if (h > 1) h -= 1;
454  if (h < 1.0f / 6.0f)
455  return m1 + (m2 - m1) * h * 6.0f;
456  else if (h < 3.0f / 6.0f)
457  return m2;
458  else if (h < 4.0f / 6.0f)
459  return m1 + (m2 - m1) * (2.0f / 3.0f - h) * 6.0f;
460  return m1;
461  };
462 
463  IColor col;
464  h = std::fmodf(h, 1.0f);
465  if (h < 0.0f) h += 1.0f;
466  s = Clip(s, 0.0f, 1.0f);
467  l = Clip(l, 0.0f, 1.0f);
468  float m2 = l <= 0.5f ? (l * (1 + s)) : (l + s - l * s);
469  float m1 = 2 * l - m2;
470  col.R = static_cast<int>(Clip(hue(h + 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
471  col.G = static_cast<int>(Clip(hue(h, m1, m2), 0.0f, 1.0f) * 255.f);
472  col.B = static_cast<int>(Clip(hue(h - 1.0f / 3.0f, m1, m2), 0.0f, 1.0f) * 255.f);
473  col.A = static_cast<int>(a * 255.f);
474  return col;
475  }
