62 template<typename MatrixType>
63 void createRandomPIMatrixOfRank(Index desired_rank, Index rows, Index cols, MatrixType& m)
64 {
65  typedef typename internal::traits<MatrixType>::Scalar Scalar;
66  enum { Rows = MatrixType::RowsAtCompileTime, Cols = MatrixType::ColsAtCompileTime };
67 
68  typedef Matrix<Scalar, Dynamic, 1> VectorType;
69  typedef Matrix<Scalar, Rows, Rows> MatrixAType;
70  typedef Matrix<Scalar, Cols, Cols> MatrixBType;
71 
72  if(desired_rank == 0)
73  {
74  m.setZero(rows,cols);
75  return;
76  }
77 
78  if(desired_rank == 1)
79  {
80  // here we normalize the vectors to get a partial isometry
81  m = VectorType::Random(rows).normalized() * VectorType::Random(cols).normalized().transpose();
82  return;
83  }
84 
85  MatrixAType a = MatrixAType::Random(rows,rows);
86  MatrixType d = MatrixType::Identity(rows,cols);
87  MatrixBType b = MatrixBType::Random(cols,cols);
88 
89  // set the diagonal such that only desired_rank non-zero entries remain
90  const Index diag_size = (std::min)(d.rows(),d.cols());
91  if(diag_size != desired_rank)
92  d.diagonal().segment(desired_rank, diag_size-desired_rank) = VectorType::Zero(diag_size-desired_rank);
93 
94  HouseholderQR<MatrixAType> qra(a);
95  HouseholderQR<MatrixBType> qrb(b);
96  m = qra.householderQ() * d * qrb.householderQ();
97 }

106 template<typename PermutationVectorType>
107 void randomPermutationVector(PermutationVectorType& v, Index size)
108 {
109  typedef typename PermutationVectorType::Scalar Scalar;
110  v.resize(size);
111  for(Index i = 0; i < size; ++i) v(i) = Scalar(i);
112  if(size == 1) return;
113  for(Index n = 0; n < 3 * size; ++n)
114  {
115  Index i = internal::random<Index>(0, size-1);
116  Index j;
117  do j = internal::random<Index>(0, size-1); while(j==i);
118  std::swap(v(i), v(j));
119  }
120 }

132 template<typename MatrixType>
133 MatrixType generateRandomUnitaryMatrix(const Index dim)
134 {
135  typedef typename internal::traits<MatrixType>::Scalar Scalar;
136  typedef Matrix<Scalar, Dynamic, 1> VectorType;
137 
138  MatrixType v = MatrixType::Identity(dim, dim);
139  VectorType h = VectorType::Zero(dim);
140  for (Index i = 0; i < dim; ++i)
141  {
142  v.col(i).tail(dim - i - 1) = VectorType::Random(dim - i - 1);
143  h(i) = 2 / v.col(i).tail(dim - i).squaredNorm();
144  }
145 
146  const Eigen::HouseholderSequence<MatrixType, VectorType> HSeq(v, h);
147  return MatrixType(HSeq);
148 }

177 template<typename MatrixType, typename RealScalarVectorType>
178 void generateRandomMatrixSvs(const RealScalarVectorType &svs, const Index rows, const Index cols, MatrixType& M)
179 {
180  enum { Rows = MatrixType::RowsAtCompileTime, Cols = MatrixType::ColsAtCompileTime };
181  typedef typename internal::traits<MatrixType>::Scalar Scalar;
182  typedef Matrix<Scalar, Rows, Rows> MatrixAType;
183  typedef Matrix<Scalar, Cols, Cols> MatrixBType;
184 
185  const Index min_dim = (std::min)(rows, cols);
186 
187  const MatrixAType U = generateRandomUnitaryMatrix<MatrixAType>(rows);
188  const MatrixBType V = generateRandomUnitaryMatrix<MatrixBType>(cols);
189 
190  M = U.block(0, 0, rows, min_dim) * svs.asDiagonal() * V.block(0, 0, cols, min_dim).transpose();
191 }

209 template<typename VectorType, typename RealScalar>
210 VectorType setupRandomSvs(const Index dim, const RealScalar max)
211 {
212  VectorType svs = max / RealScalar(2) * (VectorType::Random(dim) + VectorType::Ones(dim));
213  std::sort(svs.begin(), svs.end(), std::greater<RealScalar>());
214  return svs;

235 template<typename VectorType, typename RealScalar>
236 VectorType setupRangeSvs(const Index dim, const RealScalar min, const RealScalar max)
237 {
238  VectorType svs = VectorType::Random(dim);
239  if(dim == 0)
240  return svs;
241  if(dim == 1)
242  {
243  svs(0) = min;
244  return svs;
245  }
246  std::sort(svs.begin(), svs.end(), std::greater<RealScalar>());
247 
248  // scale to range [min, max]
249  const RealScalar c_min = svs(dim - 1), c_max = svs(0);
250  svs = (svs - VectorType::Constant(dim, c_min)) / (c_max - c_min);
251  return min * (VectorType::Ones(dim) - svs) + max * svs;
252 }
