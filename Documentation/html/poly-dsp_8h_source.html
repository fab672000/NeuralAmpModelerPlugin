158 
159  int fNote; // Playing note actual pitch

175  extractPaths(fGatePath, fFreqPath, fGainPath);

180  void extractPaths(std::vector<std::string>& gate, std::vector<std::string>& freq, std::vector<std::string>& gain)
181  {
182  // Keep gain, freq and gate labels
183  std::map<std::string, FAUSTFLOAT*>::iterator it;
184  for (it = getMap().begin(); it != getMap().end(); it++) {
185  std::string path = (*it).first;
186  if (endsWith(path, "/gate")) {
187  gate.push_back(path);
188  } else if (endsWith(path, "/freq")) {
189  freq.push_back(path);
190  } else if (endsWith(path, "/gain")) {
191  gain.push_back(path);
192  }
193  }
194  }
195 
196  // MIDI velocity [0..127]
197  void keyOn(int pitch, int velocity, bool trigger)
198  {
199  keyOn(pitch, float(velocity)/127.f, trigger);
200  }
201 
202  // Normalized MIDI velocity [0..1]
203  void keyOn(int pitch, float velocity, bool trigger)
204  {
205  // So that DSP state is always re-initialized
206  fDSP->instanceClear();
207 
208  for (size_t i = 0; i < fFreqPath.size(); i++) {
209  setParamValue(fFreqPath[i], midiToFreq(pitch));
210  }
211  for (size_t i = 0; i < fGatePath.size(); i++) {
212  setParamValue(fGatePath[i], FAUSTFLOAT(1));
213  }
214  for (size_t i = 0; i < fGainPath.size(); i++) {
215  setParamValue(fGainPath[i], velocity);
216  }
217 
218  fNote = pitch;
219  }
220 
221  void keyOff(bool hard = false)
222  {
223  // No use of velocity for now...
224  for (size_t i = 0; i < fGatePath.size(); i++) {
225  setParamValue(fGatePath[i], FAUSTFLOAT(0));
226  }
227 
228  if (hard) {
229  // Immediately stop voice
230  fNote = kFreeVoice;
231  } else {
232  // Release voice
233  fRelease = fMinRelease;
234  fNote = kReleaseVoice;
235  }
236  }

457  FAUSTFLOAT mixCheckVoice(int count, FAUSTFLOAT** outputBuffer, FAUSTFLOAT** mixBuffer)
458  {
459  FAUSTFLOAT level = 0;
460  for (int i = 0; i < getNumOutputs(); i++) {
461  FAUSTFLOAT* mixChannel = mixBuffer[i];
462  FAUSTFLOAT* outChannel = outputBuffer[i];
463  for (int j = 0; j < count; j++) {
464  level = std::max<FAUSTFLOAT>(level, (FAUSTFLOAT)fabs(outChannel[j]));
465  mixChannel[j] += outChannel[j];
466  }
467  }
468  return level;
469  }

472  {
473  for (int i = 0; i < getNumOutputs(); i++) {
474  FAUSTFLOAT* mixChannel = mixBuffer[i];
475  FAUSTFLOAT* outChannel = outputBuffer[i];
476  for (int j = 0; j < count; j++) {
477  mixChannel[j] += outChannel[j];
478  }
479  }

489  int getPlayingVoice(int pitch)
490  {
491  int voice_playing = kNoVoice;
492  int oldest_date_playing = INT_MAX;
493 
494  for (size_t i = 0; i < fVoiceTable.size(); i++) {
495  if (fVoiceTable[i]->fNote == pitch) {
496  // Keeps oldest playing voice
497  if (fVoiceTable[i]->fDate < oldest_date_playing) {
498  oldest_date_playing = fVoiceTable[i]->fDate;
499  voice_playing = int(i);
500  }
501  }
502  }
503 
504  return voice_playing;
505  }
506 
507  // Always returns a voice
508  int getFreeVoice()
509  {
510  int voice = kNoVoice;
511 
512  // Looks for the first available voice
513  for (size_t i = 0; i < fVoiceTable.size(); i++) {
514  if (fVoiceTable[i]->fNote == kFreeVoice) {
515  voice = int(i);
516  goto result;
517  }
518  }
519 
520  {
521  // Otherwise steal one
522  int voice_release = kNoVoice;
523  int voice_playing = kNoVoice;
524 
525  int oldest_date_release = INT_MAX;
526  int oldest_date_playing = INT_MAX;
527 
528  // Scan all voices
529  for (size_t i = 0; i < fVoiceTable.size(); i++) {
530  if (fVoiceTable[i]->fNote == kReleaseVoice) {
531  // Keeps oldest release voice
532  if (fVoiceTable[i]->fDate < oldest_date_release) {
533  oldest_date_release = fVoiceTable[i]->fDate;
534  voice_release = int(i);
535  }
536  } else {
537  // Otherwise keeps oldest playing voice
538  if (fVoiceTable[i]->fDate < oldest_date_playing) {
539  oldest_date_playing = fVoiceTable[i]->fDate;
540  voice_playing = int(i);
541  }
542  }
543  }
544 
545  // Then decide which one to steal
546  if (oldest_date_release != INT_MAX) {
547  std::cout << "Steal release voice : voice_date " << fVoiceTable[voice_release]->fDate << " cur_date = " << fDate << " voice = " << voice_release << std::endl;
548  voice = voice_release;
549  goto result;
550  } else if (oldest_date_playing != INT_MAX) {
551  std::cout << "Steal playing voice : voice_date " << fVoiceTable[voice_playing]->fDate << " cur_date = " << fDate << " voice = " << voice_playing << std::endl;
552  voice = voice_playing;
553  goto result;
554  } else {
555  assert(false);
556  return kNoVoice;
557  }
558  }
559 
560  result:
561  fVoiceTable[voice]->fDate = fDate++;
562  fVoiceTable[voice]->fNote = kActiveVoice;
563  return voice;
564  }

692  void compute(int count, FAUSTFLOAT** inputs, FAUSTFLOAT** outputs)
693  {
694  assert(count <= MIX_BUFFER_SIZE);
695 
696  // First clear the outputs
697  clearOutput(count, outputs);
698 
699  if (fVoiceControl) {
700  // Mix all playing voices
701  for (size_t i = 0; i < fVoiceTable.size(); i++) {
702  dsp_voice* voice = fVoiceTable[i];
703  if (voice->fNote != kFreeVoice) {
704  voice->compute(count, inputs, fMixBuffer);
705  // Mix it in result
706  voice->fLevel = mixCheckVoice(count, fMixBuffer, outputs);
707  // Check the level to possibly set the voice in kFreeVoice again
708  voice->fRelease -= count;
709  if ((voice->fNote == kReleaseVoice)
710  && (voice->fRelease < 0)
711  && (voice->fLevel < VOICE_STOP_LEVEL)) {
712  voice->fNote = kFreeVoice;
713  }
714  }
715  }
716  } else {
717  // Mix all voices
718  for (size_t i = 0; i < fVoiceTable.size(); i++) {
719  fVoiceTable[i]->compute(count, inputs, fMixBuffer);
720  mixVoice(count, fMixBuffer, outputs);
721  }
722  }
723  }
