69  double getSignalToNoiseDB() {
70  static const double threshold = 1.0e-14;
71  if (mMeanSquareSignal < threshold || mMeanSquareNoise < threshold) {
72  return 0.0;
73  } else {
74  double signalToNoise = mMeanSquareSignal / mMeanSquareNoise; // power ratio
75  double signalToNoiseDB = 10.0 * log(signalToNoise);
76  if (signalToNoiseDB < MIN_SNRATIO_DB) {
77  LOGD("ERROR - signal to noise ratio is too low! < %d dB. Adjust volume.",
78  MIN_SNRATIO_DB);
79  setResult(ERROR_VOLUME_TOO_LOW);
80  }
81  return signalToNoiseDB;

115  double calculateMagnitude(double *phasePtr = NULL) {
116  if (mFramesAccumulated == 0) {
117  return 0.0;
118  }
119  double sinMean = mSinAccumulator / mFramesAccumulated;
120  double cosMean = mCosAccumulator / mFramesAccumulated;
121  double magnitude = 2.0 * sqrt( (sinMean * sinMean) + (cosMean * cosMean ));
122  if( phasePtr != NULL )
123  {
124  double phase = M_PI_2 - atan2( sinMean, cosMean );
125  *phasePtr = phase;
126  }
127  return magnitude;
128  }

134  result_code processInputFrame(float *frameData, int channelCount) override {
135  result_code result = RESULT_OK;
136 
137  float sample = frameData[0];
138  float peak = mPeakFollower.process(sample);
139  mInfiniteRecording.write(sample);
140 
141  // Force a periodic glitch!
142  if (mForceGlitchDuration > 0) {
143  if (mForceGlitchCounter == 0) {
144  LOGE("%s: force a glitch!!", __func__);
145  mForceGlitchCounter = getSampleRate();
146  } else if (mForceGlitchCounter <= mForceGlitchDuration) {
147  sample += (sample > 0.0) ? -0.5f : 0.5f;
148  }
149  --mForceGlitchCounter;
150  }
151 
152  mStateFrameCounters[mState]++; // count how many frames we are in each state
153 
154  switch (mState) {
155  case STATE_IDLE:
156  mDownCounter--;
157  if (mDownCounter <= 0) {
158  mState = STATE_IMMUNE;
159  mDownCounter = IMMUNE_FRAME_COUNT;
160  mInputPhase = 0.0; // prevent spike at start
161  mOutputPhase = 0.0;
162  }
163  break;
164 
165  case STATE_IMMUNE:
166  mDownCounter--;
167  if (mDownCounter <= 0) {
168  mState = STATE_WAITING_FOR_SIGNAL;
169  }
170  break;
171 
172  case STATE_WAITING_FOR_SIGNAL:
173  if (peak > mThreshold) {
174  mState = STATE_WAITING_FOR_LOCK;
175  //LOGD("%5d: switch to STATE_WAITING_FOR_LOCK", mFrameCounter);
176  resetAccumulator();
177  }
178  break;
179 
180  case STATE_WAITING_FOR_LOCK:
181  mSinAccumulator += sample * sinf(mInputPhase);
182  mCosAccumulator += sample * cosf(mInputPhase);
183  mFramesAccumulated++;
184  // Must be a multiple of the period or the calculation will not be accurate.
185  if (mFramesAccumulated == mSinePeriod * PERIODS_NEEDED_FOR_LOCK) {
186  double phaseOffset = 0.0;
187  setMagnitude(calculateMagnitude(&phaseOffset));
188 // LOGD("%s() mag = %f, offset = %f, prev = %f",
189 // __func__, mMagnitude, mPhaseOffset, mPreviousPhaseOffset);
190  if (mMagnitude > mThreshold) {
191  if (abs(phaseOffset) < kMaxPhaseError) {
192  mState = STATE_LOCKED;
193 // LOGD("%5d: switch to STATE_LOCKED", mFrameCounter);
194  }
195  // Adjust mInputPhase to match measured phase
196  mInputPhase += phaseOffset;
197  }
198  resetAccumulator();
199  }
200  incrementInputPhase();
201  break;
202 
203  case STATE_LOCKED: {
204  // Predict next sine value
205  float predicted = sinf(mInputPhase) * mMagnitude;
206  float diff = predicted - sample;
207  float absDiff = fabs(diff);
208  mMaxGlitchDelta = std::max(mMaxGlitchDelta, absDiff);
209  if (absDiff > mScaledTolerance) {
210  result = ERROR_GLITCHES;
211  onGlitchStart();
212 // LOGI("diff glitch detected, absDiff = %g", absDiff);
213  } else {
214  mSumSquareSignal += predicted * predicted;
215  mSumSquareNoise += diff * diff;
216  // Track incoming signal and slowly adjust magnitude to account
217  // for drift in the DRC or AGC.
218  mSinAccumulator += sample * sinf(mInputPhase);
219  mCosAccumulator += sample * cosf(mInputPhase);
220  mFramesAccumulated++;
221  // Must be a multiple of the period or the calculation will not be accurate.
222  if (mFramesAccumulated == mSinePeriod) {
223  const double coefficient = 0.1;
224  double phaseOffset = 0.0;
225  double magnitude = calculateMagnitude(&phaseOffset);
226  // One pole averaging filter.
227  setMagnitude((mMagnitude * (1.0 - coefficient)) + (magnitude * coefficient));
228 
229  mMeanSquareNoise = mSumSquareNoise * mInverseSinePeriod;
230  mMeanSquareSignal = mSumSquareSignal * mInverseSinePeriod;
231  resetAccumulator();
232 
233  if (abs(phaseOffset) > kMaxPhaseError) {
234  result = ERROR_GLITCHES;
235  onGlitchStart();
236  LOGD("phase glitch detected, phaseOffset = %g", phaseOffset);
237  } else if (mMagnitude < mThreshold) {
238  result = ERROR_GLITCHES;
239  onGlitchStart();
240  LOGD("magnitude glitch detected, mMagnitude = %g", mMagnitude);
241  }
242  }
243  }
244  incrementInputPhase();
245  } break;
246 
247  case STATE_GLITCHING: {
248  // Predict next sine value
249  mGlitchLength++;
250  float predicted = sinf(mInputPhase) * mMagnitude;
251  float diff = predicted - sample;
252  float absDiff = fabs(diff);
253  mMaxGlitchDelta = std::max(mMaxGlitchDelta, absDiff);
254  if (absDiff < mScaledTolerance) { // close enough?
255  // If we get a full sine period of non-glitch samples in a row then consider the glitch over.
256  // We don't want to just consider a zero crossing the end of a glitch.
257  if (mNonGlitchCount++ > mSinePeriod) {
258  onGlitchEnd();
259  }
260  } else {
261  mNonGlitchCount = 0;
262  if (mGlitchLength > (4 * mSinePeriod)) {
263  relock();
264  }
265  }
266  incrementInputPhase();
267  } break;
268 
269  case NUM_STATES: // not a real state
270  break;
271  }
272 
273  mFrameCounter++;
274 
275  return result;
276  }
