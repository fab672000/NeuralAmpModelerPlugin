52  static double PlotResponse(double freqCPS, double Q, EMode mode, double x, double gain = 0., double minHz = 1., double maxHz = 20000)
53  {
54  using cdouble = std::complex<double>;
55 
56  const double g = freqCPS;
57  const double k = 1.0 / Q;
58  const double A = std::pow(10., gain / 40.);
59 
60  static double minLogHz = std::log10(minHz);
61  static double maxLogHz = std::log10(maxHz);
62  static const cdouble _1j = std::sqrt(cdouble(-1, 0));
63 
64  const double w = std::pow(10., (x * (maxLogHz - minLogHz)) + minHz);
65  const cdouble s = _1j * w;
66  cdouble result;
67 
68  switch (mode)
69  {
70  case kLowPass: result = g * g / (g * g + g * k * s + s * s); break;
71  case kBandPass: result = g * s / (g * g + g * k * s + s * s); break;
72  case kHighPass: result = s * s / (g * g + g * k * s + s * s); break;
73  // case kNotch: result = (g * g) + (s * s) / (g * g + g * k * s + s * s); break;
74  case kPeak: result = (g - s) * (g + s) / (g * g + g * k * s + s * s); break;
75  // case kBell: result = A * (1. + A * k * s + (s * s)) / A + k * s + A * (s * s); break;
76  // case kLowPassShelf: result = A * (A + std::sqrt(A) * k * s + (s * s)) / 1. + std::sqrt(A) * k * s + A * (s * s); break;
77  // case kHighPassShelf: result = A * (1. + std::sqrt(A) * k * s + A * (s * s)) / A + std::sqrt(A) * k * s + (s * s); break;
78  default: return 0.;
79  }
80 
81  const double magnitude = 20. * std::log10(std::abs(std::real(result)));
82 
83  //DBGMSG("%f\n", magnitude);
84 
85  return magnitude;
86  }

98  void ProcessBlock(T** inputs, T** outputs, int nChans, int nFrames)
99  {
100  assert(nChans <= NC);
101 
102  if(mState != mNewState)
103  UpdateCoefficients();
104 
105  for (auto c = 0; c < nChans; c++)
106  {
107  for (auto s = 0; s < nFrames; s++)
108  {
109  const double v0 = (double) inputs[c][s];
110 
111  mV3[c] = v0 - mIc2eq[c];
112  mV1[c] = m_a1 * mIc1eq[c] + m_a2*mV3[c];
113  mV2[c] = mIc2eq[c] + m_a2 * mIc1eq[c] + m_a3 * mV3[c];
114  mIc1eq[c] = 2. * mV1[c] - mIc1eq[c];
115  mIc2eq[c] = 2. * mV2[c] - mIc2eq[c];
116 
117  outputs[c][s] = (T) m_m0 * v0 + m_m1 * mV1[c] + m_m2 * mV2[c];
118  }
119  }
120  }

135  void UpdateCoefficients()
136  {
137  mState = mNewState;
138 
139  const double w = std::tan(PI * mState.freq/mState.sampleRate);
140 
141  switch(mState.mode)
142  {
143  case kLowPass:
144  {
145  const double g = w;
146  const double k = 1. / mState.Q;
147  m_a1 = 1./(1. + g * (g + k));
148  m_a2 = g * m_a1;
149  m_a3 = g * m_a2;
150  m_m0 = 0;
151  m_m1 = 0;
152  m_m2 = 1.;
153  break;
154  }
155  case kHighPass:
156  {
157  const double g = w;
158  const double k = 1. / mState.Q;
159  m_a1 = 1./(1. + g * (g + k));
160  m_a2 = g * m_a1;
161  m_a3 = g * m_a2;
162  m_m0 = 1.;
163  m_m1 = -k;
164  m_m2 = -1.;
165  break;
166  }
167  case kBandPass:
168  {
169  const double g = w;
170  const double k = 1. / mState.Q;
171  m_a1 = 1./(1. + g * (g + k));
172  m_a2 = g * m_a1;
173  m_a3 = g * m_a2;
174  m_m0 = 0.;
175  m_m1 = 1.;
176  m_m2 = 0.;
177  break;
178  }
179  case kNotch:
180  {
181  const double g = w;
182  const double k = 1. / mState.Q;
183  m_a1 = 1./(1. + g * (g + k));
184  m_a2 = g * m_a1;
185  m_a3 = g * m_a2;
186  m_m0 = 1.;
187  m_m1 = -k;
188  m_m2 = 0.;
189  break;
190  }
191  case kPeak:
192  {
193  const double g = w;
194  const double k = 1. / mState.Q;
195  m_a1 = 1./(1. + g * (g + k));
196  m_a2 = g * m_a1;
197  m_a3 = g * m_a2;
198  m_m0 = 1.;
199  m_m1 = -k;
200  m_m2 = -2.;
201  break;
202  }
203  case kBell:
204  {
205  const double A = std::pow(10., mState.gain/40.);
206  const double g = w;
207  const double k = 1 / mState.Q;
208  m_a1 = 1./(1. + g * (g + k));
209  m_a2 = g * m_a1;
210  m_a3 = g * m_a2;
211  m_m0 = 1.;
212  m_m1 = k * (A * A - 1.);
213  m_m2 = 0.;
214  break;
215  }
216  case kLowPassShelf:
217  {
218  const double A = std::pow(10., mState.gain/40.);
219  const double g = w / std::sqrt(A);
220  const double k = 1. / mState.Q;
221  m_a1 = 1./(1. + g * (g + k));
222  m_a2 = g * m_a1;
223  m_a3 = g * m_a2;
224  m_m0 = 1.;
225  m_m1 = k * (A - 1.);
226  m_m2 = (A * A - 1.);
227  break;
228  }
229  case kHighPassShelf:
230  {
231  const double A = std::pow(10., mState.gain/40.);
232  const double g = w / std::sqrt(A);
233  const double k = 1. / mState.Q;
234  m_a1 = 1./(1. + g * (g + k));
235  m_a2 = g * m_a1;
236  m_a3 = g * m_a2;
237  m_m0 = A*A;
238  m_m1 = k*(1. - A)*A;
239  m_m2 = (1. - A*A);
240  break;
241  }
242  default:
243  break;
244  }
