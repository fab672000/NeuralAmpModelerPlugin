166 static dsp_poly_factory* readWasmPolyDSPFactoryFromMachineFile(const std::string& machine_code_path, std::string& error_msg)
167 {
168  std::string process_path = machine_code_path + "_machine_process";
169  std::string effect_path = machine_code_path + "_machine_effect";
170  wasm_dsp_factory* process_factory = readWasmDSPFactoryFromMachineFile(process_path, error_msg);
171  wasm_dsp_factory* effect_factory = readWasmDSPFactoryFromMachineFile(effect_path, error_msg);
172  if (process_factory) {
173  return new dsp_poly_factory(process_factory, effect_factory);
174  } else {
175  wasm_dsp_factory* process_factory = readWasmDSPFactoryFromMachineFile(machine_code_path, error_msg);
176  return (process_factory) ? new dsp_poly_factory(process_factory, NULL) : NULL;
177  }
