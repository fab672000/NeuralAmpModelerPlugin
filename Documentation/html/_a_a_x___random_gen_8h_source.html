39 inline int32_t GetInt32RPDF(int32_t* iSeed)
40 {
41  // Requirement: iSeed param must be a static in the calling function.
42 
43  int32_t k = int32_t((*iSeed)*cSeedDivisor);
44  *iSeed = 16807 * (*iSeed - k * 127773) - 2836 * k + 7395;
45  if (*iSeed < 0)
46  *iSeed += 2147483647;
47  return (*iSeed - 1073741824) * 2; // -2147483647..+2147483647
