44  {
45  int status = msg.StatusMsg();
46  double time = 0.;
47 
48  switch (status) {
49  case IMidiMsg::kNoteOn:
50  handleKeyOn(time, msg.Channel(), msg.mData1, msg.mData2);
51  break;
52  case IMidiMsg::kNoteOff:
53  handleKeyOff(time, msg.Channel(), msg.mData1, msg.mData2);
54  break;
55  case IMidiMsg::kPolyAftertouch:
56  handlePolyAfterTouch(time, msg.Channel(), msg.mData1, msg.mData2);
57  break;
58  case IMidiMsg::kControlChange:
59  handleCtrlChange(time, msg.Channel(), msg.mData1, msg.mData2);
60  break;
61  case IMidiMsg::kProgramChange:
62  handleProgChange(time, msg.Channel(), msg.mData1);
63  break;
64  case IMidiMsg::kChannelAftertouch:
65  handleAfterTouch(time, msg.Channel(), msg.mData1);
66  break;
67  case IMidiMsg::kPitchWheel:
68  handlePitchWheel(time, msg.Channel(), msg.mData1, msg.mData2);
69  break;

73  }
