202 static void JNL_get_mime_type_for_file(const char *fn, char *strout, int stroutsz)
203 {
204  const char *ext = fn;
205  while (*ext) ext++;
206  while (ext > fn && *ext != '.' && *ext != '/' && *ext != '\\') ext--;
207 
208  const char *type = "application/octet-stream";
209 
210  if (!stricmp(ext,".jpg")) type = "image/jpeg";
211  else if (!stricmp(ext,".png")) type = "image/png";
212  else if (!stricmp(ext,".gif")) type = "image/gif";
213  else if (!stricmp(ext,".txt")) type = "text/plain";
214  else if (!strnicmp(ext,".htm",4)) type = "text/html";
215  else if (!stricmp(ext,".js")) type = "application/javascript";
216  else if (!stricmp(ext,".css")) type = "text/css";
217  else if (!stricmp(ext,".xml")) type = "text/xml";
218  else if (!stricmp(ext,".svg")) type = "image/svg+xml";
219 
220  lstrcpyn_safe(strout,type,stroutsz);
221 }

223 static void JNL_Format_RFC1123(time_t t, char *buf)
224 {
225 
226  buf[0]=0;
227  static const char days[] = { "SunMonTueWedThuFriSat" };
228  static const char mons[] = { "JanFebMarAprMayJunJulAugSepOctNovDec" };
229 
230  struct tm *tm = gmtime(&t);
231  if (!tm) return;
232  memcpy(buf, days + (tm->tm_wday%7)*3, 3);
233  strcpy(buf+3,", ");
234  char *p=buf+5;
235  strftime(p, 64, "%d xxx %Y %H:%M:%S GMT", tm);
236  while (*p && *p != 'x') p++;
237  if (*p) memcpy(p, mons + (tm->tm_mon%12)*3, 3);
238 }
