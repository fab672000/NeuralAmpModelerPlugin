81  // Add a generic parameter
82  virtual void addParameter(const char* label,
83  FAUSTFLOAT* zone,
84  FAUSTFLOAT init,
85  FAUSTFLOAT min,
86  FAUSTFLOAT max,
87  FAUSTFLOAT step,
88  ItemType type)
89  {
90  std::string path = buildPath(label);
91  fPathMap[path] = fLabelMap[label] = fNumParameters++;
92  fPaths.push_back(path);
93  fLabels.push_back(label);
94  fZone.push_back(zone);
95  fInit.push_back(init);
96  fMin.push_back(min);
97  fMax.push_back(max);
98  fStep.push_back(step);
99  fItemType.push_back(type);
100 
101  // handle scale metadata
102  switch (fCurrentScale) {
103  case kLin:
104  fConversion.push_back(new LinearValueConverter(0, 1, min, max));
105  break;
106  case kLog:
107  fConversion.push_back(new LogValueConverter(0, 1, min, max));
108  break;
109  case kExp: fConversion.push_back(new ExpValueConverter(0, 1, min, max));
110  break;
111  }
112  fCurrentScale = kLin;
113 
114  if (fCurrentAcc.size() > 0 && fCurrentGyr.size() > 0) {
115  std::cerr << "warning : 'acc' and 'gyr' metadata used for the same " << label << " parameter !!\n";
116  }
117 
118  // handle acc metadata "...[acc : <axe> <curve> <amin> <amid> <amax>]..."
119  if (fCurrentAcc.size() > 0) {
120  std::istringstream iss(fCurrentAcc);
121  int axe, curve;
122  double amin, amid, amax;
123  iss >> axe >> curve >> amin >> amid >> amax;
124 
125  if ((0 <= axe) && (axe < 3) &&
126  (0 <= curve) && (curve < 4) &&
127  (amin < amax) && (amin <= amid) && (amid <= amax))
128  {
129  fAcc[axe].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, min, init, max));
130  } else {
131  std::cerr << "incorrect acc metadata : " << fCurrentAcc << std::endl;
132  }
133  fCurrentAcc = "";
134  }
135 
136  // handle gyr metadata "...[gyr : <axe> <curve> <amin> <amid> <amax>]..."
137  if (fCurrentGyr.size() > 0) {
138  std::istringstream iss(fCurrentGyr);
139  int axe, curve;
140  double amin, amid, amax;
141  iss >> axe >> curve >> amin >> amid >> amax;
142 
143  if ((0 <= axe) && (axe < 3) &&
144  (0 <= curve) && (curve < 4) &&
145  (amin < amax) && (amin <= amid) && (amid <= amax))
146  {
147  fGyr[axe].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, min, init, max));
148  } else {
149  std::cerr << "incorrect gyr metadata : " << fCurrentGyr << std::endl;
150  }
151  fCurrentGyr = "";
152  }
153 
154  // handle screencolor metadata "...[screencolor:red|green|blue|white]..."
155  if (fCurrentColor.size() > 0) {
156  if ((fCurrentColor == "red") && (fRedReader == 0)) {
157  fRedReader = new ZoneReader(zone, min, max);
158  fHasScreenControl = true;
159  } else if ((fCurrentColor == "green") && (fGreenReader == 0)) {
160  fGreenReader = new ZoneReader(zone, min, max);
161  fHasScreenControl = true;
162  } else if ((fCurrentColor == "blue") && (fBlueReader == 0)) {
163  fBlueReader = new ZoneReader(zone, min, max);
164  fHasScreenControl = true;
165  } else if ((fCurrentColor == "white") && (fRedReader == 0) && (fGreenReader == 0) && (fBlueReader == 0)) {
166  fRedReader = new ZoneReader(zone, min, max);
167  fGreenReader = new ZoneReader(zone, min, max);
168  fBlueReader = new ZoneReader(zone, min, max);
169  fHasScreenControl = true;
170  } else {
171  std::cerr << "incorrect screencolor metadata : " << fCurrentColor << std::endl;
172  }
173  }
174  fCurrentColor = "";
175 
176  fMetaData.push_back(fCurrentMetadata);
177  fCurrentMetadata.clear();
178  }

189  void setConverter(std::vector<ZoneControl*>* table, int p, int val, int curve, double amin, double amid, double amax)
190  {
191  int id1 = getZoneIndex(table, p, 0);
192  int id2 = getZoneIndex(table, p, 1);
193  int id3 = getZoneIndex(table, p, 2);
194 
195  // Deactivates everywhere..
196  if (id1 != -1) table[0][id1]->setActive(false);
197  if (id2 != -1) table[1][id2]->setActive(false);
198  if (id3 != -1) table[2][id3]->setActive(false);
199 
200  if (val == -1) { // Means: no more mapping...
201  // So stay all deactivated...
202  } else {
203  int id4 = getZoneIndex(table, p, val);
204  if (id4 != -1) {
205  // Reactivate the one we edit...
206  table[val][id4]->setMappingValues(curve, amin, amid, amax, fMin[p], fInit[p], fMax[p]);
207  table[val][id4]->setActive(true);
208  } else {
209  // Allocate a new CurveZoneControl which is 'active' by default
210  FAUSTFLOAT* zone = fZone[p];
211  table[val].push_back(new CurveZoneControl(zone, curve, amin, amid, amax, fMin[p], fInit[p], fMax[p]));
212  }
213  }
214  }

216  void getConverter(std::vector<ZoneControl*>* table, int p, int& val, int& curve, double& amin, double& amid, double& amax)
217  {
218  int id1 = getZoneIndex(table, p, 0);
219  int id2 = getZoneIndex(table, p, 1);
220  int id3 = getZoneIndex(table, p, 2);
221 
222  if (id1 != -1) {
223  val = 0;
224  curve = table[val][id1]->getCurve();
225  table[val][id1]->getMappingValues(amin, amid, amax);
226  } else if (id2 != -1) {
227  val = 1;
228  curve = table[val][id2]->getCurve();
229  table[val][id2]->getMappingValues(amin, amid, amax);
230  } else if (id3 != -1) {
231  val = 2;
232  curve = table[val][id3]->getCurve();
233  table[val][id3]->getMappingValues(amin, amid, amax);
234  } else {
235  val = -1; // No mapping
236  curve = 0;
237  amin = -100.;
238  amid = 0.;
239  amax = 100.;
240  }
