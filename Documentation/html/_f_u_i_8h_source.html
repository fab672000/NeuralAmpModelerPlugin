40  * FUI : used to save and recall the state of the user interface
41  * This class provides essentially two new methods saveState() and recallState()
42  * used to save on file and recall from file the state of the user interface.

62  }

70  // -- Save and recall methods
71 
72  // save the zones values and full names
73  virtual void saveState(const char* filename)
74  {
75  std::ofstream file(filename);
76  std::map<std::string, FAUSTFLOAT*>::iterator it;
77 
78  if (file.is_open()) {
79  for (it = fName2Zone.begin(); it != fName2Zone.end(); ++it) {
80  file << *(*it).second << ' ' << (*it).first << std::endl;
81  }
82 
83  file << std::endl;
84  file.close();
85  } else {
86  std::cerr << "Error opening " << filename << " file\n";
87  }
88  }
89 
90  // recall the zones values and full names
91  virtual void recallState(const char* filename)
92  {
93  std::ifstream file(filename);
94  FAUSTFLOAT value;
95  std::string path1, path2;
96  while (file.good()) {
97  file >> value >> path1;
98  path2 = "/" + path1;
99  if (fName2Zone.count(path1) > 0) { // Old path system
100  *(fName2Zone[path1]) = value;
101  } else if (fName2Zone.count(path2) > 0) { // New path system with the starting '/'
102  *(fName2Zone[path2]) = value;
103  } else if (path1.size() > 0) {
104  std::cerr << "recallState : parameter not found : " << path1 << " with value : " << value << std::endl;
105  }
106  }
107  file.close();
108  }
