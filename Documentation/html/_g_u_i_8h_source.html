363  fDelete = true;

370  {
371  ztimedmap::iterator it;
372  if (fDelete && ((it = GUI::gTimedZoneMap.find(fZone)) != GUI::gTimedZoneMap.end())) {
373  ringbuffer_free((*it).second);
374  GUI::gTimedZoneMap.erase(it);
375  }
376  }
