60 template <typename T>
61 bool RectiFi_UnitDisplayDelegateDecorator<T>::ValueToString(T value, int32_t maxNumChars, AAX_CString* valueString) const
62 {
63  int32_t maxLength = maxNumChars;
64  value = value + 0.5;
65 
66  if (maxLength >= 8)
67  {
68  valueString->AppendNumber(static_cast<int>(value));
69  if (maxLength >= 11)
70  valueString->Append(" Hertz");
71  else
72  valueString->Append(" Hz");
73  }
74  else if (value >= 1000)
75  {
76  valueString->AppendNumber(static_cast<int>(value/1000));
77  valueString->Append(".");
78  valueString->AppendNumber(((int)value % 1000) / 100);
79  valueString->Append("k");
80  }
81  else
82  {
83  valueString->AppendNumber(static_cast<int>(value));
84  if (maxLength >= 5)
85  valueString->Append("Hz");
86  }
87 
88  double normalizedValue = double(value) - double(kMinFilterFrequency);
89  normalizedValue = normalizedValue / (double(kMaxFilterFrequency) - double(kMinFilterFrequency));
90 
91  int32_t longVal = NormalizedToInt32(normalizedValue);
92  double maxCheck = LongControlToNewRange(longVal, 0, 1000);
93  if ( maxCheck == 1000 )
94  {
95  valueString->Clear();
96  valueString->Append("Thru");
97  return true;
98  }
99 
100  return true;
101 }

103 template <typename T>
104 bool RectiFi_UnitDisplayDelegateDecorator<T>::StringToValue(const AAX_CString& valueString, T* value) const
105 {
106  //Just call through if there is obviously no unit string.
107  if (valueString.Length() <= mUnitString.Length())
108  return AAX_IDisplayDelegateDecorator<T>::StringToValue(valueString, value);
109 
110  //Just call through if the end of this string does not match the unit string.
111  AAX_CString unitSubString;
112  valueString.SubString(valueString.Length() - mUnitString.Length(), mUnitString.Length(), &unitSubString);
113  if (unitSubString != mUnitString)
114  return AAX_IDisplayDelegateDecorator<T>::StringToValue(valueString, value);
115 
116  //Call through with the stripped down value string.
117  AAX_CString valueSubString;
118  valueString.SubString(0, valueString.Length() - mUnitString.Length(), &valueSubString);
119  return AAX_IDisplayDelegateDecorator<T>::StringToValue(valueSubString, value);
120 }
