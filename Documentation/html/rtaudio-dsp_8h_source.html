61  virtual int processAudio(double streamTime, void* inbuf, void* outbuf, unsigned long frames)
62  {
63  AVOIDDENORMALS;
64 
65  float* inputs[fDsp->getNumInputs()];
66  float* outputs[fDsp->getNumOutputs()];
67 
68  for (int i = 0; i < fDsp->getNumInputs(); i++) {
69  inputs[i] = &(static_cast<float*>(inbuf))[i * frames];
70  }
71  for (int i = 0; i < fDsp->getNumOutputs(); i++) {
72  outputs[i] = &(static_cast<float*>(outbuf))[i * frames];
73  }
74 
75  // process samples
76  fDsp->compute(streamTime * 1000000., frames, inputs, outputs);
77  return 0;
78  }
79

81  unsigned int nBufferFrames,

83  void* drv)
84  {

86  }

114  bool init(const char* /*name*/, int numInputs, int numOutputs)
115  {
116  if (fAudioDAC.getDeviceCount() < 1) {
117  std::cout << "No audio devices found!\n";
118  return false;
119  }
120 
121  RtAudio::DeviceInfo info_in = fAudioDAC.getDeviceInfo(fAudioDAC.getDefaultInputDevice());
122  RtAudio::DeviceInfo info_out = fAudioDAC.getDeviceInfo(fAudioDAC.getDefaultOutputDevice());
123  RtAudio::StreamParameters iParams, oParams;
124 
125  iParams.deviceId = fAudioDAC.getDefaultInputDevice();
126  fDevNumInChans = info_in.inputChannels;
127  iParams.nChannels = fDevNumInChans;
128  iParams.firstChannel = 0;
129 
130  oParams.deviceId = fAudioDAC.getDefaultOutputDevice();
131  fDevNumOutChans = info_out.outputChannels;
132  oParams.nChannels = fDevNumOutChans;
133  oParams.firstChannel = 0;
134 
135  RtAudio::StreamOptions options;
136  options.flags |= RTAUDIO_NONINTERLEAVED;
137 
138  try {
139  fAudioDAC.openStream(((numOutputs > 0) ? &oParams : NULL),
140  ((numInputs > 0) ? &iParams : NULL), FORMAT,
141  fSampleRate, &fBufferSize, audioCallback, this, &options);
142  } catch (RtAudioError& e) {
143  std::cout << '\n' << e.getMessage() << '\n' << std::endl;
144  return false;
145  }
146 
147  return true;
148  }
149

155  printf("DSP has %d inputs and %d outputs, physical inputs = %d physical outputs = %d \n",
156  fDsp->getNumInputs(), fDsp->getNumOutputs(),
157  fDevNumInChans, fDevNumOutChans);
158  fDsp = new dsp_adapter(fDsp, fDevNumInChans, fDevNumOutChans, fBufferSize);
159  }
160 
161  fDsp->init(fSampleRate);
162  }
