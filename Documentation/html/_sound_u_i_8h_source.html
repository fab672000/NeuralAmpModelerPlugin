113  // -- soundfiles
114  virtual void addSoundfile(const char* label, const char* url, Soundfile** sf_zone)
115  {
116  const char* saved_url = url; // 'url' is consumed by parseMenuList2
117  std::vector<std::string> file_name_list;
118 
119  bool menu = parseMenuList2(url, file_name_list, true);
120  // If not a list, we have as single file
121  if (!menu) { file_name_list.push_back(saved_url); }
122 
123  // Parse the possible list
124  if (fSoundfileMap.find(saved_url) == fSoundfileMap.end()) {
125  // Check all files and get their complete path
126  std::vector<std::string> path_name_list = fSoundReader->checkFiles(fSoundfileDir, file_name_list);
127  // Read them and create the Soundfile
128  Soundfile* sound_file = fSoundReader->createSoundfile(path_name_list, MAX_CHAN);
129  if (sound_file) {
130  fSoundfileMap[saved_url] = sound_file;
131  } else {
132  // If failure, use 'defaultsound'
133  std::cerr << "addSoundfile : soundfile for " << saved_url << " cannot be created !" << std::endl;
134  *sf_zone = defaultsound;
135  return;
136  }
137  }
138 
139  // Get the soundfile
140  *sf_zone = fSoundfileMap[saved_url];
141  }
