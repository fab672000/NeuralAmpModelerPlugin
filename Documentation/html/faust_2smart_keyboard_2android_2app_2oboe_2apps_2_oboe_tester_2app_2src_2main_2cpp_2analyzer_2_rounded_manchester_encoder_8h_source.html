32  RoundedManchesterEncoder(int samplesPerPulse)
33  : ManchesterEncoder(samplesPerPulse) {
34  int rampSize = samplesPerPulse / 4;
35  mZeroAfterZero = std::make_unique<float[]>(samplesPerPulse);
36  mZeroAfterOne = std::make_unique<float[]>(samplesPerPulse);
37 
38  int i = 0;
39  for (int j = 0; j < rampSize; j++) {
40  float phase = (j + 1) * M_PI / rampSize;
41  float sample = -cosf(phase);
42  mZeroAfterZero[i] = sample;
43  mZeroAfterOne[i] = 1.0f;
44  i++;
45  }
46  for (int j = 0; j < rampSize; j++) {
47  mZeroAfterZero[i] = 1.0f;
48  mZeroAfterOne[i] = 1.0f;
49  i++;
50  }
51  for (int j = 0; j < rampSize; j++) {
52  float phase = (j + 1) * M_PI / rampSize;
53  float sample = cosf(phase);
54  mZeroAfterZero[i] = sample;
55  mZeroAfterOne[i] = sample;
56  i++;
57  }
58  for (int j = 0; j < rampSize; j++) {
59  mZeroAfterZero[i] = -1.0f;
60  mZeroAfterOne[i] = -1.0f;
61  i++;
62  }
63  }
