55 /*
56  BufferedOutputBus
57 
58  This class provides a prepareOutputBufferList method to copy the internal buffer pointers
59  to the output buffer list in case the client passed in null buffer pointers.
60  */
61 struct BufferedOutputBus: BufferedAudioBus {
62  void prepareOutputBufferList(AudioBufferList* outBufferList, AVAudioFrameCount frameCount, bool zeroFill) {
63  UInt32 byteSize = frameCount * sizeof(float);
64  for (UInt32 i = 0; i < outBufferList->mNumberBuffers; ++i) {
65  outBufferList->mBuffers[i].mNumberChannels = originalAudioBufferList->mBuffers[i].mNumberChannels;
66  outBufferList->mBuffers[i].mDataByteSize = byteSize;
67  if (outBufferList->mBuffers[i].mData == nullptr) {
68  outBufferList->mBuffers[i].mData = originalAudioBufferList->mBuffers[i].mData;
69  }
70  if (zeroFill) {
71  memset(outBufferList->mBuffers[i].mData, 0, byteSize);
72  }
73  }
74  }
75 };

81  BufferedInputBus
82 
83  This class manages a buffer into which an audio unit with input busses can
84  pull its input data.
85  */
86 struct BufferedInputBus : BufferedAudioBus {
87  /*
88  Gets input data for this input by preparing the input buffer list and pulling
89  the pullInputBlock.
90  */
91  AUAudioUnitStatus pullInput(AudioUnitRenderActionFlags *actionFlags,
92  AudioTimeStamp const* timestamp,
93  AVAudioFrameCount frameCount,
94  NSInteger inputBusNumber,
95  AURenderPullInputBlock pullInputBlock) {
96  if (pullInputBlock == nullptr) {
97  return kAudioUnitErr_NoConnection;
98  }
99 
100  /*
101  Important:
102  The Audio Unit must supply valid buffers in (inputData->mBuffers[x].mData) and mDataByteSize.
103  mDataByteSize must be consistent with frameCount.
104 
105  The AURenderPullInputBlock may provide input in those specified buffers, or it may replace
106  the mData pointers with pointers to memory which it owns and guarantees will remain valid
107  until the next render cycle.
108 
109  See prepareInputBufferList()
110  */
111 
112  prepareInputBufferList(frameCount);
113 
114  return pullInputBlock(actionFlags, timestamp, frameCount, inputBusNumber, mutableAudioBufferList);
115  }
116 
117  /*
118  prepareInputBufferList populates the mutableAudioBufferList with the data
119  pointers from the originalAudioBufferList.
120 
121  The upstream audio unit may overwrite these with its own pointers, so each
122  render cycle this function needs to be called to reset them.
123  */
124  void prepareInputBufferList(UInt32 frameCount) {
125  UInt32 byteSize = std::min(frameCount, maxFrames) * sizeof(float);
126  mutableAudioBufferList->mNumberBuffers = originalAudioBufferList->mNumberBuffers;
127 
128  for (UInt32 i = 0; i < originalAudioBufferList->mNumberBuffers; ++i) {
129  mutableAudioBufferList->mBuffers[i].mNumberChannels = originalAudioBufferList->mBuffers[i].mNumberChannels;
130  mutableAudioBufferList->mBuffers[i].mData = originalAudioBufferList->mBuffers[i].mData;
131  mutableAudioBufferList->mBuffers[i].mDataByteSize = byteSize;
132  }
133  }
