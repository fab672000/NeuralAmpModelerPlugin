45 // DBL_INT_ADD treats two unsigned ints a and b as one 64-bit integer and adds c to it
46 #define ROTLEFT(a, b) ((a << b) | (a >> (32 - b)))
47 #define DBL_INT_ADD(a, b, c) \
48  if (a > 0xffffffff - c) ++b; \
49  a += c;

59 inline void sha1_transform(SHA1_CTX* ctx, uchar data[])
60 {
61  uint a, b, c, d, e, i, j, t, m[80];
62 
63  for (i = 0, j = 0; i < 16; ++i, j += 4) {
64  m[i] = (data[j] << 24) + (data[j + 1] << 16) + (data[j + 2] << 8) + (data[j + 3]);
65  }
66  for (; i < 80; ++i) {
67  m[i] = (m[i - 3] ^ m[i - 8] ^ m[i - 14] ^ m[i - 16]);
68  m[i] = (m[i] << 1) | (m[i] >> 31);
69  }
70 
71  a = ctx->state[0];
72  b = ctx->state[1];
73  c = ctx->state[2];
74  d = ctx->state[3];
75  e = ctx->state[4];
76 
77  for (i = 0; i < 20; ++i) {
78  t = ROTLEFT(a, 5) + ((b & c) ^ (~b & d)) + e + ctx->k[0] + m[i];
79  e = d;
80  d = c;
81  c = ROTLEFT(b, 30);
82  b = a;
83  a = t;
84  }
85  for (; i < 40; ++i) {
86  t = ROTLEFT(a, 5) + (b ^ c ^ d) + e + ctx->k[1] + m[i];
87  e = d;
88  d = c;
89  c = ROTLEFT(b, 30);
90  b = a;
91  a = t;
92  }
93  for (; i < 60; ++i) {
94  t = ROTLEFT(a, 5) + ((b & c) ^ (b & d) ^ (c & d)) + e + ctx->k[2] + m[i];
95  e = d;
96  d = c;
97  c = ROTLEFT(b, 30);
98  b = a;
99  a = t;
100  }
101  for (; i < 80; ++i) {
102  t = ROTLEFT(a, 5) + (b ^ c ^ d) + e + ctx->k[3] + m[i];
103  e = d;
104  d = c;
105  c = ROTLEFT(b, 30);
106  b = a;
107  a = t;
108  }
109 
110  ctx->state[0] += a;
111  ctx->state[1] += b;
112  ctx->state[2] += c;
113  ctx->state[3] += d;
114  ctx->state[4] += e;
115 }

133 inline void sha1_update(SHA1_CTX* ctx, uchar data[], uint len)
134 {
135  uint i;
136 
137  for (i = 0; i < len; ++i) {
138  ctx->data[ctx->datalen] = data[i];
139  ctx->datalen++;
140  if (ctx->datalen == 64) {
141  sha1_transform(ctx, ctx->data);
142  DBL_INT_ADD(ctx->bitlen[0], ctx->bitlen[1], 512);
143  ctx->datalen = 0;
144  }
145  }
146 }

148 static void sha1_final(SHA1_CTX* ctx, uchar hash[])
149 {
150  uint i;
151 
152  i = ctx->datalen;
153 
154  // Pad whatever data is left in the buffer.
155  if (ctx->datalen < 56) {
156  ctx->data[i++] = 0x80;
157  while (i < 56) {
158  ctx->data[i++] = 0x00;
159  }
160  } else {
161  ctx->data[i++] = 0x80;
162  while (i < 64) {
163  ctx->data[i++] = 0x00;
164  }
165  sha1_transform(ctx, ctx->data);
166  memset(ctx->data, 0, 56);
167  }
168 
169  // Append to the padding the total message's length in bits and transform.
170  DBL_INT_ADD(ctx->bitlen[0], ctx->bitlen[1], 8 * ctx->datalen);
171  ctx->data[63] = ctx->bitlen[0];
172  ctx->data[62] = ctx->bitlen[0] >> 8;
173  ctx->data[61] = ctx->bitlen[0] >> 16;
174  ctx->data[60] = ctx->bitlen[0] >> 24;
175  ctx->data[59] = ctx->bitlen[1];
176  ctx->data[58] = ctx->bitlen[1] >> 8;
177  ctx->data[57] = ctx->bitlen[1] >> 16;
178  ctx->data[56] = ctx->bitlen[1] >> 24;
179  sha1_transform(ctx, ctx->data);
180 
181  // Since this implementation uses little endian byte ordering and MD uses big endian,
182  // reverse all the bytes when copying the final state to the output hash.
183  for (i = 0; i < 4; ++i) {
184  hash[i] = (ctx->state[0] >> (24 - i * 8)) & 0x000000ff;
185  hash[i + 4] = (ctx->state[1] >> (24 - i * 8)) & 0x000000ff;
186  hash[i + 8] = (ctx->state[2] >> (24 - i * 8)) & 0x000000ff;
187  hash[i + 12] = (ctx->state[3] >> (24 - i * 8)) & 0x000000ff;
188  hash[i + 16] = (ctx->state[4] >> (24 - i * 8)) & 0x000000ff;
189  }
190 }

199 LIBEXPORT inline void generateCSHA1(const char* data, char* sha_key)
200 {
201  SHA1_CTX ctx;
202  unsigned char obuf[20] = {0};
203 
204  // Hash one
205  sha1_init(&ctx);
206  sha1_update(&ctx, (unsigned char*)data, strlen(data));
207  sha1_final(&ctx, obuf);
208 
209  // convert SHA1 key into hexadecimal string
210  for (int i = 0; i < 20; i++) {
211  const char* H = "0123456789ABCDEF";
212  char c1 = H[(obuf[i] >> 4)];
213  char c2 = H[(obuf[i] & 15)];
214  sha_key += c1;
215  sha_key += c2;
216  }
217 }
