30 /*
31  A 'MemoryReader' object can be used to prepare a set of sound resources in memory, to be used by SoundUI::addSoundfile.
32 
33  A Soundfile* object will have to be filled with a list of sound resources: the fLength, fOffset, fSampleRate and fBuffers fields
34  have to be completed with the appropriate values, and will be accessed in the DSP object while running.
35  *
36  */

81  virtual void readFile(Soundfile* soundfile, const std::string& path_name, int part, int& offset, int max_chan)
82  {
83  soundfile->fLength[part] = SOUND_LENGTH;
84  soundfile->fSR[part] = SOUND_SR;
85  soundfile->fOffset[part] = offset;
86 
87  // Audio frames have to be written for each chan
88  for (int sample = 0; sample < SOUND_LENGTH; sample++) {
89  for (int chan = 0; chan < SOUND_CHAN; chan++) {
90  soundfile->fBuffers[chan][offset + sample] = 0.f;
91  }
92  }
93 
94  // Update offset
95  offset += SOUND_LENGTH;
96  }
97 
98 };
