78  tresult PLUGIN_API getClassInfo (int32 index, PClassInfo* info) override
79  {
80  if (index < 0 || index >= static_cast<int32> (T::classInfos.size ()) || info == nullptr)
81  return kInvalidArgument;
82  *info = {};
83  const auto& ci = T::classInfos[index];
84  copyTUID (info->cid, ci.cid);
85  info->cardinality = ci.cardinality;
86  if (ci.category)
87  strncpy (info->category, ci.category, PClassInfo::kCategorySize);
88  if (ci.name)
89  strncpy (info->name, ci.name, PClassInfo::kNameSize);
90  return kResultTrue;
91  }
92  tresult PLUGIN_API createInstance (FIDString cid, FIDString iid, void** obj) override
93  {
94  for (const auto& e : T::classInfos)
95  {
96  if (FUnknownPrivate::iidEqual (e.cid, cid))
97  {
98  auto instance = e.create (e.context);
99  if (instance)
100  {
101  if (instance->queryInterface (iid, obj) == kResultOk)
102  {
103  instance->release ();
104  return kResultOk;
105  }
106  else
107  instance->release ();
108  }
109  }
110  }
111  *obj = nullptr;
112  return kNoInterface;
113  }
