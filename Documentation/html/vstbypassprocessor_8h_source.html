110 template <typename T>
111 static bool delay (int32 sampleFrames, T* inStream, T* outStream, T* delayBuffer, int32 bufferSize,
112  int32 bufferInPos, int32 bufferOutPos)
113 {
114  // delay inStream
115  int32 remain, inFrames, outFrames;
116  T* bufIn;
117  T* bufOut;
118 
119  remain = sampleFrames;
120  while (remain > 0)
121  {
122  bufIn = delayBuffer + bufferInPos;
123  bufOut = delayBuffer + bufferOutPos;
124 
125  if (bufferInPos > bufferOutPos)
126  inFrames = bufferSize - bufferInPos;
127  else
128  inFrames = bufferOutPos - bufferInPos;
129 
130  outFrames = bufferSize - bufferOutPos;
131 
132  if (inFrames > remain)
133  inFrames = remain;
134 
135  if (outFrames > inFrames)
136  outFrames = inFrames;
137 
138  // order important for in-place processing!
139  memcpy (bufIn, inStream, inFrames * sizeof (T)); // copy to buffer
140  memcpy (outStream, bufOut, outFrames * sizeof (T)); // copy from buffer
141 
142  inStream += inFrames;
143  outStream += outFrames;
144 
145  bufferInPos += inFrames;
146  if (bufferInPos >= bufferSize)
147  bufferInPos -= bufferSize;
148  bufferOutPos += outFrames;
149  if (bufferOutPos >= bufferSize)
150  bufferOutPos -= bufferSize;
151 
152  if (inFrames > outFrames)
153  {
154  // still some output to copy
155  bufOut = delayBuffer + bufferOutPos;
156  outFrames = inFrames - outFrames;
157 
158  memcpy (outStream, bufOut, outFrames * sizeof (T)); // copy from buffer
159 
160  outStream += outFrames;
161 
162  bufferOutPos += outFrames;
163  if (bufferOutPos >= bufferSize)
164  bufferOutPos -= bufferSize;
165  }
166 
167  remain -= inFrames;
168  }
169 
170  return true;
171 }

206  SpeakerArray inArray (inputArr);
207  SpeakerArray outArray (outputArr);
208 
209  // security check (todo)
210  if (outArray.total () >= kMaxChannelsSupported)
211  return;
212 
213  for (int32 i = 0; i < outArray.total (); i++)
214  {
215  if (outArray.at (i) == Vst::kSpeakerL)
216  {
217  if (inArray.total () == 1 && inArray.at (0) == Vst::kSpeakerM)
218  {

264  void process (ProcessData& data)
265  {
266  // flush
267  if (data.numInputs == 0 || data.numOutputs == 0)
268  return;
269 
270  AudioBusBuffers& inBus = data.inputs[0];
271  AudioBusBuffers& outBus = data.outputs[0];
272 
273  if (data.symbolicSampleSize == kSample32)
274  {
275  if (!outBus.channelBuffers32)
276  return;
277  }
278  else if (!outBus.channelBuffers64)
279  return;
280 
281  if (mMainIOBypass)
282  {
283  for (int32 channel = 0; channel < outBus.numChannels; channel++)
284  {
285  T* src = nullptr;
286  bool silent = true;
287  T* dst;
288  if (data.symbolicSampleSize == kSample32)
289  dst = (T*)outBus.channelBuffers32[channel];
290  else
291  dst = (T*)outBus.channelBuffers64[channel];
292  if (!dst)
293  continue;
294 
295  int inputChannel = mInputPinLookup[channel];
296  if (inputChannel != -1)
297  {
298  silent = (inBus.silenceFlags & (1ll << inputChannel)) != 0;
299  if (data.symbolicSampleSize == kSample32)
300  src = (T*)inBus.channelBuffers32[inputChannel];
301  else
302  src = (T*)inBus.channelBuffers64[inputChannel];
303  }
304 
305  if (mDelays[channel]->process (src, dst, data.numSamples, silent))
306  {
307  outBus.silenceFlags |= (1ll << channel);
308  }
309  else
310  {
311  outBus.silenceFlags = 0;
312  }
313  }
314  }
315 
316  // clear all other outputs
317  for (int32 outBusIndex = mMainIOBypass ? 1 : 0; outBusIndex < data.numOutputs;
318  outBusIndex++)
319  {
320  outBus = data.outputs[outBusIndex];
321 
322  for (int32 channel = 0; channel < outBus.numChannels; channel++)
323  {
324  T* dst;
325  if (data.symbolicSampleSize == kSample32)
326  dst = (T*)outBus.channelBuffers32[channel];
327  else
328  dst = (T*)outBus.channelBuffers64[channel];
329  if (dst)
330  {
331  memset (dst, 0, data.numSamples * sizeof (T));
332  outBus.silenceFlags |= 1ll << channel;
333  }
334  }
335  }
336  }

341  struct Delay
342  {
343  AudioBuffer<T> mDelayBuffer;
344  int32 mDelaySamples;
345  int32 mInPos;
346  int32 mOutPos;
347 
348  Delay (int32 maxSamplesPerBlock, int32 delaySamples)
349  : mDelaySamples (delaySamples), mInPos (-1), mOutPos (-1)
350  {
351  if (mDelaySamples > 0)
352  mDelayBuffer.resize (maxSamplesPerBlock + mDelaySamples);
353  }
354 
355  bool hasDelay () const { return mDelaySamples > 0; }
356  int32 getBufferSamples () const { return mDelayBuffer.getMaxSamples (); }
357 
358  bool process (T* src, T* dst, int32 numSamples, bool silentIn)
359  {
360  bool silentOut = false;
361 
362  if (hasDelay () && src)
363  {
364  int32 bufferSize = getBufferSamples ();
365  delay<T> (numSamples, src, dst, mDelayBuffer, bufferSize, mInPos, mOutPos);
366 
367  // update inPos, outPos
368  mInPos += numSamples;
369  if (mInPos >= bufferSize)
370  mInPos -= bufferSize;
371  mOutPos += numSamples;
372  if (mOutPos >= bufferSize)
373  mOutPos -= bufferSize;
374  }
375  else
376  {
377  if (src != dst)
378  {
379  if (src && !silentIn)
380  {
381  memcpy (dst, src, numSamples * sizeof (T));
382  }
383  else
384  {
385  memset (dst, 0, numSamples * sizeof (T));
386  silentOut = true;
387  }
388  }
389  else
390  {
391  silentOut = silentIn;
392  }
393  }
394  return silentOut;
395  }
396 
397  void flush ()
398  {
399  mDelayBuffer.clearAll ();
400 
401  mInPos = mOutPos = 0;
402  if (hasDelay ())
403  mOutPos = getBufferSamples () - mDelaySamples; // must be != inPos
404  }
405  };
