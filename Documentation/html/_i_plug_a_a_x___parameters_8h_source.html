166 private:
167  struct SParamValList
168  {
169  static const int32_t sCap = 4*kSynchronizedParameterQueueSize;
170 
171  TParamValPair* mElem[sCap];
172  int32_t mSize;
173 
174  SParamValList()
175  {
176  Clear();
177  }
178 
179  void Add(TParamValPair* inElem)
180  {
181  AAX_ASSERT(sCap > mSize);
182  if (sCap > mSize)
183  {
184  mElem[mSize++] = inElem;
185  }
186  }
187 
188  void Append(const SParamValList& inOther)
189  {
190  AAX_ASSERT(sCap >= mSize + inOther.mSize);
191  for (int32_t i = 0; i < inOther.mSize; ++i)
192  {
193  Add(inOther.mElem[i]);
194  }
195  }
196 
197  void Append(const std::list<TParamValPair*>& inOther)
198  {
199  AAX_ASSERT(sCap >= mSize + (int64_t)inOther.size());
200  for (std::list<TParamValPair*>::const_iterator iter = inOther.begin(); iter != inOther.end(); ++iter)
201  {
202  Add(*iter);
203  }
204  }
205 
206  void Merge(AAX_IPointerQueue<TParamValPair>& inOther)
207  {
208  do
209  {
210  TParamValPair* const val = inOther.Pop();
211  if (NULL == val) { break; }
212  Add(val);
213  } while (1);
214  }
215 
216  void Clear()
217  {
218  std::memset(mElem, 0x0, sizeof(mElem));
219  mSize = 0;
220  }
221  };
222 
223  typedef std::set<const AAX_IParameter*> TParamSet;
224  typedef std::pair<int64_t, std::list<TParamValPair*> > TNumberedParamStateList;
225  typedef AAX_CAtomicQueue<TNumberedParamStateList, 256> TNumberedStateListQueue;
226  typedef AAX_CAtomicQueue<const TParamValPair, 16*kSynchronizedParameterQueueSize> TParamValPairQueue;
227 
228 
229  SParamValList GetUpdatesForState(int64_t inTargetStateNum);
230  void DeleteUsedParameterChanges();
231
