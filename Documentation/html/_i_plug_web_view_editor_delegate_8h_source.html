83  void OnMessageFromWebView(const char* jsonStr) override
84  {
85  auto json = nlohmann::json::parse(jsonStr, nullptr, false);
86 
87  if(json["msg"] == "SPVFUI")
88  {
89  SendParameterValueFromUI(json["paramIdx"], json["value"]);
90  }
91  else if (json["msg"] == "BPCFUI")
92  {
93  BeginInformHostOfParamChangeFromUI(json["paramIdx"]);
94  }
95  else if (json["msg"] == "EPCFUI")
96  {
97  EndInformHostOfParamChangeFromUI(json["paramIdx"]);
98  }
99  else if (json["msg"] == "SAMFUI")
100  {
101  std::vector<unsigned char> base64;
102 
103  if(json.count("data") > 0 && json["data"].is_string())
104  {
105  auto dStr = json["data"].get<std::string>();
106  int dSize = static_cast<int>(dStr.size());
107 
108  // calculate the exact size of the decoded base64 data
109  int numPaddingBytes = 0;
110 
111  if(dSize >= 2 && dStr[dSize-2] == '=')
112  numPaddingBytes = 2;
113  else if(dSize >= 1 && dStr[dSize-1] == '=')
114  numPaddingBytes = 1;
115 
116 
117  base64.resize((dSize * 3) / 4 - numPaddingBytes);
118  wdl_base64decode(dStr.c_str(), base64.data(), static_cast<int>(base64.size()));
119  }
120 
121  SendArbitraryMsgFromUI(json["msgTag"], json["ctrlTag"], static_cast<int>(base64.size()), base64.data());
122  }
123  else if(json["msg"] == "SMMFUI")
124  {
125  IMidiMsg msg {0, json["statusByte"].get<uint8_t>(),
126  json["dataByte1"].get<uint8_t>(),
127  json["dataByte2"].get<uint8_t>()};
128  SendMidiMsgFromUI(msg);
129  }
130  }
