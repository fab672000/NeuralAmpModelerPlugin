105  uint64 rdtsc(void)
106  {
107  #ifdef TARGET_OS_IPHONE
108  return (uint64)(mach_absolute_time() * (double)fTimeInfo.numer / (double)fTimeInfo.denom);
109  #else
110  union {
111  uint32 i32[2];
112  uint64 i64;
113  } count;
114  __asm__ __volatile__("rdtsc" : "=a" (count.i32[0]), "=d" (count.i32[1]));
115  return count.i64;
116  #endif
117  }
118 
122  double rdtscpersec()
123  {
124  // If the environment variable CLOCKSPERSEC is defined
125  // we use it instead of our own measurement
126  char* str = getenv("CLOCKSPERSEC");
127  if (str) {
128  int64 cps = (int64)atoll(str);
129  if (cps > 1000000000) {
130  return cps;
131  }
132  }
133 
134  return double(fLastRDTSC - fFirstRDTSC)
135  / (((double(fTv2.tv_sec) * 1000000 + double(fTv2.tv_usec)) - (double(fTv1.tv_sec) * 1000000 + double(fTv1.tv_usec)))
136  / 1000000);
137  }

165  uint64 meanValue(std::vector<uint64>::const_iterator a, std::vector<uint64>::const_iterator b)
166  {

222  double getStats(int bsize, int ichans, int ochans)
223  {
224  assert(fMeasure > fCount);
225  std::vector<uint64> V(fCount);
226 
227  for (int i = 0; i < fCount; i++) {
228  V[i] = fStops[i] - fStarts[i];
229  }
230 
231  sort(V.begin(), V.end());
232 
233  // Mean of 10 best values (gives relatively stable results)
234  uint64 meavalx = meanValue(V.begin(), V.begin() + 10);
235  return megapersec(bsize, ichans + ochans, meavalx);

241  void printStats(const char* applname, int bsize, int ichans, int ochans)
242  {

244  std::vector<uint64> V(fCount);
245 
246  for (int i = 0; i < fCount; i++) {
247  V[i] = fStops[i] - fStarts[i];
248  }
249 
250  sort(V.begin(), V.end());
251 
252  // Mean of 10 best values (gives relatively stable results)
253  uint64 meaval00 = meanValue(V.begin(), V.begin()+ 5);
254  uint64 meaval25 = meanValue(V.begin() + fCount / 4 - 2, V.begin()+fCount / 4 + 3);
255  uint64 meaval50 = meanValue(V.begin() + fCount / 2 - 2, V.begin()+fCount / 2 + 3);
256  uint64 meaval75 = meanValue(V.begin() + 3 * fCount / 4 - 2, V.begin() + 3 * fCount / 4 + 3);
257  uint64 meaval100 = meanValue(V.end() - 5, V.end());
258 
259  // Printing
260  std::cout << applname
261  << '\t' << megapersec(bsize, ichans+ochans, meaval00)
262  << '\t' << megapersec(bsize, ichans+ochans, meaval25)
263  << '\t' << megapersec(bsize, ichans+ochans, meaval50)
264  << '\t' << megapersec(bsize, ichans+ochans, meaval75)
265  << '\t' << megapersec(bsize, ichans+ochans, meaval100)
266  << std::endl;
267  }

377  fAllInputs[i] = new REAL[fBufferSize * NBV];
378  fInputs[i] = fAllInputs[i];

382  int R0temp0 = (12345 + (1103515245 * R0_0));
383  fAllInputs[i][j] = REAL(4.656613e-10f * R0temp0);
384  R0_0 = R0temp0;
385  }
386  }
387

389  fAllOutputs = new REAL*[fDSP->getNumOutputs()];
390  for (int i = 0; i < fDSP->getNumOutputs(); i++) {
391  fAllOutputs[i] = new REAL[fBufferSize * NBV];
392  fOutputs[i] = fAllOutputs[i];
393  // Write zero in outputs
394  memset(fAllOutputs[i], 0, sizeof(REAL) * fBufferSize * NBV);
395  }

398  bool setRealtimePriority()
399  {
400  struct passwd* pw;
401  struct sched_param param;
402  int policy;
403  uid_t uid = getuid();
404  pw = getpwnam("root");
405  setuid(pw->pw_uid);
406 
407  int err = pthread_getschedparam(pthread_self(), &policy, &param);
408  if (err != 0) {
409  std::cerr << "setRealtimePriority : pthread_getschedparam res = %d" << err << std::endl;
410  }
411  policy = SCHED_RR;
412  param.sched_priority = 80;
413  err = pthread_setschedparam(pthread_self(), policy, &param);
414  if (err != 0) {
415  std::cerr << "setRealtimePriority : pthread_setschedparam res = %d" << err << std::endl;
416  }

418  setuid(uid);
