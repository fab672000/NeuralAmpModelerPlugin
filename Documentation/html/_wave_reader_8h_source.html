334  {
335  Reader* reader = new FileReader(path_name);
336  assert(reader);

339  soundfile->fLength[part] = (reader->fWave->subchunk_2_size * 8) / (reader->fWave->num_channels * reader->fWave->bits_per_sample);
340  soundfile->fSR[part] = reader->fWave->sample_rate;
341  soundfile->fOffset[part] = offset;
342 
343  // Audio frames have to be written for each chan
344  if (reader->fWave->bits_per_sample == 16) {
345  float factor = 1.f/32767.f;
346  for (int sample = 0; sample < soundfile->fLength[part]; sample++) {
347  short* frame = (short*)&reader->fWave->data[reader->fWave->block_align * sample];
348  for (int chan = 0; chan < reader->fWave->num_channels; chan++) {
349  soundfile->fBuffers[chan][offset + sample] = frame[chan] * factor;
350  }
351  }
352  } else if (reader->fWave->bits_per_sample == 32) {
353  std::cerr << "readFile : not implemented \n";
354  }
355 
356  // Update offset
357  offset += soundfile->fLength[part];
358  delete reader;
359  }
