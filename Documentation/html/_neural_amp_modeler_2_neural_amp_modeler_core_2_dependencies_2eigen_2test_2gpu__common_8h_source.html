40 void run_on_gpu(const Kernel& ker, int n, const Input& in, Output& out)
41 {
42  typename Input::Scalar* d_in;
43  typename Output::Scalar* d_out;
44  std::ptrdiff_t in_bytes = in.size() * sizeof(typename Input::Scalar);
45  std::ptrdiff_t out_bytes = out.size() * sizeof(typename Output::Scalar);
46 
47  gpuMalloc((void**)(&d_in), in_bytes);
48  gpuMalloc((void**)(&d_out), out_bytes);
49 
50  gpuMemcpy(d_in, in.data(), in_bytes, gpuMemcpyHostToDevice);
51  gpuMemcpy(d_out, out.data(), out_bytes, gpuMemcpyHostToDevice);
52 
53  // Simple and non-optimal 1D mapping assuming n is not too large
54  // That's only for unit testing!
55  dim3 Blocks(128);
56  dim3 Grids( (n+int(Blocks.x)-1)/int(Blocks.x) );
57 
58  gpuDeviceSynchronize();
59 
60 #ifdef EIGEN_USE_HIP
61  hipLaunchKernelGGL(HIP_KERNEL_NAME(run_on_gpu_meta_kernel<Kernel,
62  typename std::decay<decltype(*d_in)>::type,
63  typename std::decay<decltype(*d_out)>::type>),
64  dim3(Grids), dim3(Blocks), 0, 0, ker, n, d_in, d_out);
65 #else
66  run_on_gpu_meta_kernel<<<Grids,Blocks>>>(ker, n, d_in, d_out);
67 #endif
68  // Pre-launch errors.
69  gpuError_t err = gpuGetLastError();
70  if (err != gpuSuccess) {
71  printf("%s: %s\n", gpuGetErrorName(err), gpuGetErrorString(err));
72  gpu_assert(false);
73  }
74 
75  // Kernel execution errors.
76  err = gpuDeviceSynchronize();
77  if (err != gpuSuccess) {
78  printf("%s: %s\n", gpuGetErrorName(err), gpuGetErrorString(err));
79  gpu_assert(false);
80  }
81 
82 
83  // check inputs have not been modified
84  gpuMemcpy(const_cast<typename Input::Scalar*>(in.data()), d_in, in_bytes, gpuMemcpyDeviceToHost);
85  gpuMemcpy(out.data(), d_out, out_bytes, gpuMemcpyDeviceToHost);
86
