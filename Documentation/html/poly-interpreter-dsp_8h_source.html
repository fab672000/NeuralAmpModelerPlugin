55  }

112 static dsp_poly_factory* readInterpreterPolyDSPFactoryFromMachineFile(const std::string& bit_code_path, std::string& error_msg)
113 {
114  std::string process_path = bit_code_path + "_bitcode_process.fbc";
115  std::string effect_path = bit_code_path + "_bicode_effect.fbc";
116  interpreter_dsp_factory* process_factory = readInterpreterDSPFactoryFromBitcodeFile(process_path, error_msg);
117  interpreter_dsp_factory* effect_factory = readInterpreterDSPFactoryFromBitcodeFile(effect_path, error_msg);
118  if (process_factory) {
119  return new dsp_poly_factory(process_factory, effect_factory);
120  } else {
121  interpreter_dsp_factory* process_factory = readInterpreterDSPFactoryFromBitcodeFile(bit_code_path, error_msg);
122  return (process_factory) ? new dsp_poly_factory(process_factory, NULL) : NULL;
123  }
124 }
