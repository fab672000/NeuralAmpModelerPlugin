60  void processMidi()
61  {
62  double time = 0;
63  uint8_t data[RX_BUF_SIZE];
64  MIDIParser parser;
65  jdksMIDIMessage message;
66 
67  while (true) {
68  int rxBytes = uart_read_bytes(UART_NUM_1, data, RX_BUF_SIZE, 1);
69  if (rxBytes > 0) {
70  for (int i = 0; i < rxBytes; i++) {
71  if (parser.Parse((uchar)data[i], &message)) {
72  unsigned char status = message.GetStatus();
73  if (status < 0xF0)
74  { // channel/system message discriminator.
75  unsigned char type = message.GetType();
76  switch (type)
77  {
78  case 0x80: // Note Off
79  handleKeyOff(time, message.GetChannel(), message.GetNote(), message.GetVelocity());
80  break;
81  case 0x90: // Note On
82  handleKeyOn(time, message.GetChannel(), message.GetNote(), message.GetVelocity());
83  break;
84  case 0xA0: // Poly Key Pressure
85  handlePolyAfterTouch(time, message.GetChannel(), message.GetNote(), message.GetByte2());
86  break;
87  case 0xB0: // Control Change
88  handleCtrlChange(time, message.GetChannel(), message.GetController(), message.GetControllerValue());
89  break;
90  case 0xC0: // Program Change // No Bank Select in faust?
91  handleProgChange(time, message.GetChannel(), message.GetPGValue());
92  break;
93  case 0xD0: // Channel Pressure
94  handleAfterTouch(time, message.GetChannel(), message.GetChannelPressure());
95  break;
96  case 0xE0: // Pitch Bend
97  handlePitchWheel(time, message.GetChannel(), message.GetByte1(), message.GetByte2());
98  break;
99  default:
100  break;
101  }
102  }
103  else
104  {
105  switch (status)
106  {
107  case 0xF8: // Timing Clock
108  handleClock(time);
109  break;
110  // We can consider start and continue as identical messages.
111  case 0xFA: // Start
112  case 0xFB: // Continue
113  handleStart(time);
114  break;
115  case 0xFC: // Stop
116  handleStop(time);
117  break;
118  case 0xF0: // SysEx Start
119  // TODO
120  break;
121  case 0xF7: // SysEx Stop
122  // TODO
123  break;
124  default:
125  break;
126  }
127  }
128  }
129  // Synchronize all GUI controllers
130  GUI::updateAllGuis();
131  }
132  }
133  }
134  }
